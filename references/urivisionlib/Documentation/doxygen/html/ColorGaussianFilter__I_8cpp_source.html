207 void ColorGaussianFilter_I::applyInto(const RasterImage* imgIn, const ImageRect* theRect, RasterImage* imgOut)
208 {
209  //----------------------------------------------------------------
210  // STEP 0: Test for null or invalid type parameters
211  //----------------------------------------------------------------
212  FAIL_CONDITION( imgIn == NULL,
213  kNullRasterImageError,
214  "NULL input raster image passed as parameter to ColorGaussianFilter_I::applyInto");
215 
216  FAIL_CONDITION( theRect == NULL,
217  kNullRectangleError,
218  "NULL ImageRect parameter in ColorGaussianFilter_I::applyInto");
219 
220  FAIL_CONDITION( imgOut == NULL,
221  kNullRasterImageError,
222  "NULL output raster image passed as parameter to ColorGaussianFilter_I::applyInto");
223 
224  // so far I only accept a RasterImage_RGBa or RasterImage_RGBa_F as a source
225  FAIL_CONDITION( imgOut->getBaseRasterType() != kRGBa32Raster,
226  kInvalidRasterType,
227  "ColorGaussianFilter_F::applyInto only accepts RasterImage_RGBa or RasterImage_RGBa_F images as source");
228 
229  // so far I only accept a RasterImage_RGBa or RasterImage_RGBa_F as a destination
230  FAIL_CONDITION( imgOut->getBaseRasterType() != kRGBa32Raster,
231  kInvalidRasterType,
232  "ColorGaussianFilter_F::applyInto only accepts RasterImage_RGBa or RasterImage_RGBa_F images as destination");
233 
234 
235  //----------------------------------------------------------------
236  // STEP 1: Test for rectangle intersection
237  //----------------------------------------------------------------
238  // determine the output rectangle that the input image can produce.
239  // Here, because we apply a filter of width&height (2m+1), we
240  // "lose m all around".
241  ImageRect *usableRect = new ImageRect(imgIn->getValidRect());
242  const int m = getWidth() / 2;
243  usableRect->removeAllAround(m);
244  // ... and determine its intersection with the user-specified output rectangle
245  ImageRect *destRect = ImageRect::intersection(usableRect, theRect);
246  delete usableRect;
247  FAIL_CONDITION( destRect == NULL,
248  kNullRectangleError,
249  "Destination rectangle does not intersect valid data rectangle in ColorGaussianFilter_I::applyInto");
250  // if the output image does not contain the destination rect, it must be resized.
251  if (!imgOut->contains(destRect))
252  imgOut->setBoundRect(destRect);
253 
254 
255  //----------------------------------------------------------------
256  // STEP 2: Perform the calculations
257  //----------------------------------------------------------------
258  // allocate a 2D raster for intermediate calculations if needed
259  bool newTempRaster = allocateTempRaster_(imgIn->getValidRect());
260 
261  switch (getWidth())
262  {
263  case 3:
264  applyInto_3x3_(imgIn, destRect, imgOut);
265  break;
266 
267  case 5:
268  applyInto_5x5_(imgIn, destRect, imgOut);
269  break;
270 
271  case 7:
272  applyInto_7x7_(imgIn, destRect, imgOut);
273  break;
274 
275  case 9:
276  applyInto_9x9_(imgIn, destRect, imgOut);
277  break;
278 
279  case 11:
280  applyInto_11x11_(imgIn, destRect, imgOut);
281  break;
282 
283  case 13:
284  applyInto_13x13_(imgIn, destRect, imgOut);
285  break;
286 
287  default:
288  applyInto_(imgIn, destRect, imgOut);
289  break;
290  }
291 
292  //----------------------------------------------------------------
293  // STEP 4: Set rectangle
294  //----------------------------------------------------------------
295  // Very important: At the end of the computation, don't forget to set the
296  // output's valid data rectangle. In "safe" mode, the library has no way to guess
297  // what the valid data rectangle should be and sets it to its minimum value... empty
298  imgOut->setValidRect(destRect);
299  if (imgOut->hasFloatRaster())
300  (static_cast<RasterImage_RGBa_F*>(imgOut))->setFloatConversionMode(getFloatConversionMode());
301 
302  //----------------------------------------------------------------
303  // STEP 5: Free temporary storage (if needed)
304  //----------------------------------------------------------------
305 // if (localRGBa)
306 // delete rgbaImg;
307  delete destRect;
308  if (newTempRaster)
309  freeTempRaster_();
310 
311 }

428 void ColorGaussianFilter_I::initializeFilter_(void)
429 {
430  const int m = getWidth()/2;
431  const double scale = getScale();
432  const double sigma2Scale = 1.L / (2.L * scale * scale);
433 
434  ig_ = new int[m+1];
435  FAIL_CONDITION( ig_ == NULL,
436  kFilterAllocationError,
437  "filter allocation failed");
438 
439  // the scaling denominator is the inverse of the smallest value of the
440  // "floating point filter"
441  iDenum_ = static_cast<int>((sqrt(2*PI)*scale) / exp(-m*m*sigma2Scale));
442 
443  for (int r=0; r<=m; r++)
444  ig_[r] = static_cast<int>(exp((m*m - -r*r)*sigma2Scale));
445 
446 }
