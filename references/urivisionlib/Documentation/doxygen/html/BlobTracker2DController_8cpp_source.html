102 {
103 //---------------------------------------------------
104 // do background subtraction, thresholding
105 //---------------------------------------------------
106  remover_->removeBackground(img, differImg_);
107  thresholder_->applyInto(differImg_, grayParams_, label_);
108 
109 //---------------------------------------------------
110 // tracking control
111 //---------------------------------------------------
112  BlobList::iterator blobIter;
113  list<const ImageRect*>::iterator rectIter;
114  BlobTracker2DList::iterator trackerIter;
115  const ImagePoint_F *center = NULL;
116  BlobTrackingState *trackingState = NULL;
117 
118  // initial setup
119  if(frameCount_ == 0)
120  {
121  blobFinder_->findAllBlobs(label_, grayParams_->nbModes, target_, blobMinSize_, blobList_);
122 
123  blobIter = blobList_[0].begin();
124  trackerIter = trackerList_.begin();
125 
126  // create tracker each detected blob
127  for( ; blobIter!=blobList_[0].end(); blobIter++)
128  {
129  // reject small noise blobs
130  if((*blobIter)->getNbPixels() > noiseSize_)
131  {
132  center = (*blobIter)->getCentroid();
133  cout << "x, y = " << center->getX() << " , " << center->getY() << endl;
134  trackingState = new BlobTrackingState(center->getX(),center->getY(),5);
135 
136  // push into trackerList_
137  trackerList_.push_back(new BlobTracker2D(new BlobFinder(), 0.f, 0.f, trackingState));
138 
139  delete center;
140  delete trackingState;
141  }
142  }
143 
144  //---------------------------------------------------
145  // update background using dynamic backgroundremover.
146  //---------------------------------------------------
147  for( ; blobIter!=blobList_[0].end(); blobIter++)
148  rectList_.push_back((*blobIter)->getBoundRect());
149  if(rectList_.empty())
150  rectList_.push_back(NULL);
151  remover_->updateBackground(img, rectList_);
152 
153  // blobList is supposed to store the wrongly found blobs, here it is only temporarily
154  // used in setup phase, therefore we clear it after temporary use
155  for (blobIter != blobList_[0].begin(); blobIter != blobList_[0].end(); blobIter++)
156  delete (*blobIter);
157  for (rectIter != rectList_.begin(); rectIter != rectList_.end(); rectIter++)
158  delete (*rectIter);
159  blobList_[0].clear();
160  rectList_.clear();
161 
162  frameCount_++;
163  PeriodOfDetection_++;
164  }
165 
166  // continuous tracking
167  else
168  {
169  float xTracker,
170  yTracker,
171  xBlob,
172  yBlob,
173  xTemp,
174  yTemp,
175  dx,
176  dy;
177  int distanceRange;
178 
179  // create a temporary variable in order to prevent label_ from being changed
180  LabelRaster *tempLabel = new LabelRaster(label_);
181 
182  blobIter = blobList_[0].begin();
183  trackerIter = trackerList_.begin();
184 
185  // check each detected blob
186  if((*trackerIter) != NULL)
187  {
188  for( ; trackerIter!=trackerList_.end(); trackerIter++)
189  {
190  trackingState = (BlobTrackingState *) (*trackerIter)->track(&blobList_[0], label_);
191 
192  xTracker = (*trackerIter)->getCurrentBlob()->getCentroid()->getX();
193  yTracker = (*trackerIter)->getCurrentBlob()->getCentroid()->getY();
194  distanceRange = (*trackerIter)->distanceRange_;
195 
196  switch (trackingState->getResult())
197  {
198  // if find a wrong one, push into blobList_[0]
199  case WRONG_TARGET_FOUND:
200  blobList_[0].push_back((*trackerIter)->getCurrentBlob());
201  // the tracker lost its targets
202  if(trackingState->getTargetGone() == true)
203  trackerList_.erase(trackerIter);
204  break;
205 
206  // if match one in blobList_[0], delete the matched one
207  case TARGET_MATCH_LIST:
208  if((*blobIter) != NULL)
209  {
210  for( ; blobIter!=blobList_[0].end(); blobIter++)
211  {
212  xBlob = (*blobIter)->getCentroid()->getX();
213  yBlob = (*blobIter)->getCentroid()->getY();
214  dx = xTracker - xBlob;
215  dy = yTracker - yBlob;
216  if(dx+distanceRange>0 && dx-distanceRange<0
217  && dy+distanceRange>0 && dy-distanceRange<0)
218  {
219  blobList_[0].erase(blobIter);
220  }
221  }
222  }
223  break;
224 
225  // if no target found, check if the tracker lost its target
226  case TARGET_NOT_FOUND:
227  if(trackingState->getTargetGone() == true)
228  trackerList_.erase(trackerIter);
229  break;
230 
231  default:
232  break;
233  }
234 
235  }
236  }
237 
238  frameCount_++; // increasing frameCount_
239  PeriodOfDetection_++; // increasing PeriodOfDetection_
240 
241  ///* PENDING
242  // new object detection and handling (T = 5)
243  if(PeriodOfDetection_ == 5)
244  {
245  // once a new object found
246  BlobList *tempList = new BlobList[1];
247 
248  // (maybe we can use NormalFlow instead of/together with BackgroundRemover)
249  blobFinder_->findAllBlobs(tempLabel, grayParams_->nbModes, target_, blobMinSize_, tempList);
250 
251  // compare two list, if a new object is found, send it a new Tracker2D
252  /* put it into the NewBlobList, meanwhile, update NewBlobList*/
253  BlobList::iterator tempIter = tempList[0].begin();
254  //NewBlobList::iterator nbIter = newBlobList_.begin();
255  bool match;
256  for( ; tempIter != tempList[0].end(); tempIter++)
257  {
258  xTemp = (*tempIter)->getCentroid()->getX();
259  yTemp = (*tempIter)->getCentroid()->getY();
260 
261  match = false;
262  if((*trackerIter) != NULL)
263  {
264  for(trackerIter=trackerList_.begin(); trackerIter!=trackerList_.end(); trackerIter++)
265  {
266  xTracker = (*trackerIter)->getCurrentBlob()->getCentroid()->getX();
267  yTracker = (*trackerIter)->getCurrentBlob()->getCentroid()->getY();
268  distanceRange = (*trackerIter)->distanceRange_;
269  dx = xTracker - xTemp;
270  dy = yTracker - yTemp;
271  if(dx+distanceRange>0 && dx-distanceRange<0
272  && dy+distanceRange>0 && dy-distanceRange<0)
273  {
274  match = true;
275  break;
276  }
277 
278  }
279 
280  if(match == false)
281  {
282  /*
283  for( ; nbIter != newBlobList_.end(); nbIter++)
284  {
285  if((*nbIter)->blob->getCentroid()->getX() != (*tempIter)->getCentroid()->getX()){
286  NewBlob* nb;
287  nb->blob = new Blob(*tempIter);
288  nb->count = 0;
289  newBlobList_.push_back(nb);
290  }
291  */
292  // reject small noise blobs
293  if((*tempIter)->getNbPixels() > noiseSize_)
294  {
295  trackingState = new BlobTrackingState(xTemp,yTemp,5);
296 
297  // push into trackerList_
298  trackerList_.push_back(new BlobTracker2D(new BlobFinder(), 0.f, 0.f, trackingState));
299  }
300  }
301  }
302  else
303  {
304  for (trackerIter != trackerList_.begin(); trackerIter != trackerList_.end(); trackerIter++)
305  delete (*trackerIter);
306  trackerList_.clear();
307  trackingState = new BlobTrackingState(xTemp,yTemp,5);
308 
309  // push into trackerList_
310  trackerList_.push_back(new BlobTracker2D(new BlobFinder(), 0.f, 0.f, trackingState));
311  }
312  }
313 
314  /*
315  // check every member's count in the NewBlobList, if it is not recognized
316  // for T times, send a new tracker to it and erase it from the NewBlobList
317  */
318 
319  for (blobIter != tempList[0].begin(); blobIter != tempList[0].end(); blobIter++)
320  delete (*blobIter);
321  tempList[0].clear();
322  delete []tempList;
323 
324  PeriodOfDetection_ = 0;
325  }
326 
327  delete tempLabel;
328 
329  //---------------------------------------------------
330  // update background using dynamic backgroundremover.
331  //---------------------------------------------------
332  // iterate blobList_[0]
333  for(blobIter=blobList_[0].begin(); blobIter!=blobList_[0].end(); blobIter++)
334  rectList_.push_back((*blobIter)->getBoundRect());
335 
336  // iterate trackerList_
337  Blob* blob = NULL;
338  for(trackerIter=trackerList_.begin(); trackerIter!=trackerList_.end(); trackerIter++)
339  {
340  blob = (*trackerIter)->getCurrentBlob();
341  if(blob != NULL)
342  rectList_.push_back(blob->getBoundRect());
343  }
344 
345  if(rectList_.empty())
346  rectList_.push_back(NULL);
347  remover_->updateBackground(img, rectList_);
348  for (rectIter != rectList_.begin(); rectIter != rectList_.end(); rectIter++)
349  delete (*rectIter);
350  rectList_.clear();
351  }
