191 bool ContourForce::applyForceToContour(void){
192 
193  float avgDistance = 0.f,
194  maxDisp = 0.f,
195  minDisp = 100.f,
196  stoppingThreshold = .80f,
197  maxDispThreshold = 2.f;
198  int numPts = contourSubject2Force_->getNbPoints();
199  bool convergence = false;
200 
201  const list<ImagePoint_F*> *contourList = contourSubject2Force_->getPoints();
202  list<Vector2D_F*>::iterator iterF = force_.begin();
203 
204  for(list<ImagePoint_F*>::const_iterator iter = contourList->begin(); iter != contourList->end(); iterF++, iter++)
205  {
206  float distancePtMoved = 0.f;
207  distancePtMoved = sqrtf((*iterF)->getX()*(*iterF)->getX() + (*iterF)->getY()*(*iterF)->getY()) ;
208  avgDistance += distancePtMoved;
209 
210  if(distancePtMoved > maxDisp)
211  {
212  maxDisp = distancePtMoved;
213  }
214 
215  if(distancePtMoved < minDisp)
216  {
217  minDisp = distancePtMoved;
218  }
219 
220  (*iter)->translate((*iterF)->getX(), (*iterF)->getY());
221  }
222 
223  avgDistance /= numPts;
224 
225  // bottoms out at around .82f
226  if((avgDistance < stoppingThreshold) && (maxDisp < maxDispThreshold)){
227  convergence = true;
228  }
229 
230  contourSubject2Force_->pointsHaveMoved();
231  return convergence;
232 }

244 void ContourForce::render_(int scaledLeft, int scaledTop, int scaledRight, int scaledBottom,
245  float theZ, bool shifted, const ImagePoint* origin) const
246 {
247  //--------------------------------------------------------------
248  // Part 1: render the contour
249  //--------------------------------------------------------------
250  contourSubject2Force_->render(theZ, shifted, origin);
251 
252  //--------------------------------------------------------------
253  // Part 2: render the contour force's needles
254  //--------------------------------------------------------------
255  glLoadIdentity();
256  //
257  glPushMatrix();
258  glLoadIdentity();
259  glTranslatef(scaledLeft, scaledTop, theZ);
260  glScalef(getRenderingScaleX(), -getRenderingScaleY(), 1.f);
261 
262  const list<ImagePoint_F*> *contourList = contourSubject2Force_->getPoints();
263  list<ImagePoint_F*>::const_iterator endList = contourList->end();
264  list<Vector2D_F*>::const_iterator iterF = force_.begin();
265 
266  for(list<ImagePoint_F*>::const_iterator iter = contourList->begin(); iter != endList; iterF++, iter++)
267  {
268  glPushMatrix();
269  glTranslatef((*iter)->getX(), (*iter)->getY(), 0.f);
270 
271  glColor3f(needleColor_[0], needleColor_[1], needleColor_[2]);
272  glBegin(GL_LINES);
273  glVertex2f(0, 0);
274  glVertex2f(((*iterF)->getX() * VECTORSCALE),
275  ((*iterF)->getY() * VECTORSCALE));
276  glEnd();
277 
278  glPopMatrix(); // restore the state that had been pushed
279  }
280 
281  glPopMatrix();
282 }
