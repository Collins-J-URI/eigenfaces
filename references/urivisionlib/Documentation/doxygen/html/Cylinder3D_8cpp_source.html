80 Cylinder3D::Cylinder3D(GLfloat* origin, GLfloat* dirAxis, GLfloat radius, bool isClosed)
81  : GraphicComponent3D(),
82  //
83  vertex_(new GLfloat** [kVeryHighDefSlabs+1]),
84  normal_(GlUtils::glfMatrix(0, kVeryHighDefCircPts-1, 0, 2)),
85  circStep_(kFineResolution),
86  slabStep_(kMediumResolution),
87  closedCyl_(isClosed)
88 {
89  if (!initCtSt_)
90  initStatic_();
91 
92  GLfloat **axes, w[3];
93 
94  axes = GlUtils::glfMatrix(0, 2, 0, 2);
95 
96  try {
97  // dirAxis is the direction axis of the cylinder
98  axes[2][0] = dirAxis[0];
99  axes[2][1] = dirAxis[1];
100  axes[2][2] = dirAxis[2];
101 
102  // For the major axis, we need to build a vector orthogonal to dirAxis.
103  if ((dirAxis[0]!= 0.f) || (dirAxis[2]!= 0.f)) // this is not the local Y axis
104  {
105  w[0] = 0.f; w[1] = 1.f; w[2] = 0.f;
106  commonNormalVector(w, dirAxis, axes[0]); // "X" = "Y" x "Z"
107  commonNormalVector(dirAxis, axes[0], axes[1]); // "Y" = "Z" x "X"
108  //
109  // and now we can multiply by the radius
110  for (int i=0; i<3; i++)
111  {
112  axes[0][i] *= radius;
113  axes[1][i] *= radius;
114  }
115  }
116  // if dirAxis was the local Y axis, then I can take Z and X as major and minor
117  else
118  {
119  axes[0][0] = axes[0][1] = axes[1][1] = axes[1][2] = 0.f;
120  axes[0][2] = axes[1][0] = radius;
121  }
122  }
123  catch (...)
124  {
125  deleteArrays_();
126  GlUtils::free_glfMatrix(axes, 0, 2, 0, 2);
127  FAIL_CONDITION( true,
128  kInvalidParameterError,
129  "Invalid array passed to Cylinder3D constructor");
130  }
131 
132  for (int i=0; i<=kVeryHighDefSlabs; i++)
133  vertex_[i] = GlUtils::glfMatrix(0, kVeryHighDefCircPts-1, 0, 2);
134 
135  try {
136  init_(origin, axes);
137  }
138  catch (...)
139  {
140  deleteArrays_();
141  GlUtils::free_glfMatrix(axes, 0, 2, 0, 2);
142  FAIL_CONDITION( true,
143  kInvalidParameterError,
144  "Invalid array passed to Cylinder3D constructor");
145  }
146 
147  GlUtils::free_glfMatrix(axes, 0, 2, 0, 2);
148 }

303  // I draw the cylinders as a series of one-slab-high crowns
304  // Each crown is defined as a quad strip.
305  // It might be a better idea to draw then the other way around (as a ring
306  // of linear quad strips). --> check
307  for (int i=0; i<=kVeryHighDefSlabs-slabStep_; i+=slabStep_)
308  {
309  glBegin(GL_TRIANGLE_STRIP);
310  for (int j=0; j<kVeryHighDefCircPts; j+=circStep_)
311  {
312  glNormal3fv(normal_[j]);
313  glVertex3fv(vertex_[i+slabStep_][j]);
314  glNormal3fv(normal_[j]);
315  glVertex3fv(vertex_[i][j]);
316  }
317  glNormal3fv(normal_[0]);
318  glVertex3fv(vertex_[i+slabStep_][0]);
319  glNormal3fv(normal_[0]);
320  glVertex3fv(vertex_[i][0]);
321 
322  glEnd();
323  }
324 
325  // If this is a closed cylinder, then we also draw the top and bottom
326  // sides at the same resolution as the rest)
327  if (closedCyl_)
328  {
329  glNormal3fv(normalTop_);
330  glBegin(GL_POLYGON);
331  for (int j=0; j<kVeryHighDefCircPts; j+=circStep_)
332  glVertex3fv(vertex_[kVeryHighDefSlabs][j]);
333  glEnd();
334  //
335  glNormal3fv(normalBottom_);
336  glBegin(GL_POLYGON);
337  for (int j=kVeryHighDefCircPts-circStep_; j>=0; j-=circStep_)

352 | Function: void init(GLfloat* origin, GLfloat** a) |
353 | |
354 | Status: Public |
355 | |
356 | Parameters: origin coordinates of the origin (bottom' center) |
357 | a coordinates of the major axis, minor axis, and |
358 | direction axis in a 3x3 0-based array. |
359 | |
360 | Returns: Nothing |
361 | |
362 | Description: Initializes the array of vertex and normal coordinates |
363 | |
364 +----------------------------------------------------------------------------------*/
365 void Cylinder3D::init_(GLfloat* origin, GLfloat** a)
366 {
367  GLfloat dirStep = 1.f/kVeryHighDefSlabs,
368  dir[kVeryHighDefSlabs+1][3],
369  // X, Y, and Z coordinates of the radial and normal directions.
370  radial_j[3],
371  majCross[3], minCross[3];
372 
373  for (int i=0; i<=kVeryHighDefSlabs; i++)
374  vertex_[i] = GlUtils::glfMatrix(0, kVeryHighDefCircPts-1, 0, 2);
375 
376  // majCross = a[0] x a[2] (major axis x direction)
377  crossProduct(a[0], a[2], majCross);
378  // minCross = a[1] x a[2] (minor axis x direction)
379  crossProduct(a[1], a[2], minCross);
380 
381  // First, we precompute the graduation steps for the slabs along
382  // the "direction" axis (a[2]).
383  for (int i=0; i<=kVeryHighDefSlabs; i++)
384  {
385  dir[i][0] = i*dirStep*a[2][0];
386  dir[i][1] = i*dirStep*a[2][1];
387  dir[i][2] = i*dirStep*a[2][2];
388  }
389 
390  for (int j=0; j< kVeryHighDefCircPts; j++)
391  {
392  radial_j[0] = ct_[j]*a[0][0] + st_[j]*a[1][0];
393  normal_[j][0] = ct_[j]*minCross[0] - st_[j]*majCross[0];
394  //
395  radial_j[1] = ct_[j]*a[0][1] + st_[j]*a[1][1];
396  normal_[j][1] = ct_[j]*minCross[1] - st_[j]*majCross[1];
397  //
398  radial_j[2] = ct_[j]*a[0][2] + st_[j]*a[1][2];
399  normal_[j][2] = ct_[j]*minCross[2] - st_[j]*majCross[2];
400 
401  normalizeVector(normal_[j]);
402 
403  for (int i=0; i<=kVeryHighDefSlabs; i++)
404  {
405  // M = origin + cos[u] major + sin[u] minor + v direction
406  vertex_[i][j][0] = origin[0] + radial_j[0] + dir[i][0];
407  vertex_[i][j][1] = origin[1] + radial_j[1] + dir[i][1];
408  vertex_[i][j][2] = origin[2] + radial_j[2] + dir[i][2];
409  }
410  }
411 
412  // if this is a closed cylinder, then we must also compute the normal
413  // vector for the top and bottom faces.
414  if (closedCyl_)
415  {
416  // topNormal = major x minor / ||major x minor||
417  commonNormalVector(a[0], a[1], normalTop_);
418  normalBottom_[0] = -normalTop_[0];
419  normalBottom_[1] = -normalTop_[1];
420  normalBottom_[2] = -normalTop_[2];
421  }
422 }
