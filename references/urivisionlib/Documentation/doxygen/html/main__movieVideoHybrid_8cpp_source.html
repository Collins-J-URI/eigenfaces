418 //---------------------------------------------------------------------------------
419 // setupOrthoCam
420 //
421 // This function maps world coordinates to screen coordinates. This is where
422 // the projection mode would be defined. Here I only want to have a 2D
423 // rectangle where to draw 2D stuff on top of my image, so I define my camera
424 // as an orthographic camera, and I set the world coordinates to coincide with
425 // the pixel coordinates (this is what the gluOrtho2D call does).
426 //---------------------------------------------------------------------------------
427 void setupOrthoCam(int w, int h)
428 {
429  glMatrixMode(GL_PROJECTION);
430  glLoadIdentity();
431  gluOrtho2D( 0.0f, (GLfloat) w, 0.0f, (GLfloat) h);
432 
433  glMatrixMode(GL_MODELVIEW);
434  glLoadIdentity();
435  glViewport(0, 0, w, h);
436 }
