235 void ActiveContourDetector_GVF::computeGradientVectorFlow_(void)
236 {
237 
238  //floats and remove underscore
239  float **pGradXYSqrd = (float**)normSqrdOfGrad_->getShiftedGrayF2D(),
240  **pGradientX = (float**)imageGradient_->getShiftedRasterX2D(),
241  **pGradientY = (float**)imageGradient_->getShiftedRasterY2D(),
242  **gvfVectorU, **gvfVectorV, **gvfUPlus1, **gvfVPlus1,
243  **c1 = (float**)c_->getShiftedRasterX2D(),
244  **c2 = (float**)c_->getShiftedRasterY2D(),
245  normUV = 0.0f,
246  tempAlpha, tempBeta;
247 
248  //jyh
249  // The norm of the gradient should be the drawable object with the smallest
250  // valid data rectangle. Check on that though, if you use the gradient of
251  // that image later....
252  const ImageRect *workRect = normSqrdOfGrad_->getValidRect();
253  int iLow = workRect->getTop(),
254  iHigh = workRect->getBottom(),
255  jLow = workRect->getLeft(),
256  jHigh = workRect->getRight();
257  int gvfIterationMax = 2;
258 
259  VectorField_F *tempVec = new VectorField_F(workRect);
260 //jyh --> unused
261 // float** tempVecU = (float** ) tempVec->getShiftedRasterX2D();
262 // float** tempVecV = (float** ) tempVec->getShiftedRasterY2D();
263 
264 
265  for(int i = iLow; i < iHigh; i++)
266  {
267  for(int j = jLow; j < jHigh; j++)
268  {
269  c1[i][j] = pGradXYSqrd[i][j] * pGradientX[i][j];
270  c2[i][j] = pGradXYSqrd[i][j] * pGradientY[i][j];
271  }
272  }
273 
274  tempBeta = (MU * timeStep_);
275 // theTimeStep_ = theTimeStep_;
276 
277  iLow++;
278  iHigh--;
279  jLow++;
280  jHigh--;
281  for(int iterationNum = 0; iterationNum < gvfIterationMax; iterationNum++)
282  {
283  gvfVectorU = (float** )gvfVector_->getShiftedRasterX2D();
284  gvfVectorV = (float** )gvfVector_->getShiftedRasterY2D();
285  gvfUPlus1 = (float** )gvfVectorPlus1_->getShiftedRasterX2D();
286  gvfVPlus1 = (float** )gvfVectorPlus1_->getShiftedRasterY2D();
287 
288  for(int i = iLow; i <= iHigh; i++)
289  {
290  for(int j = jLow; j <= jHigh; j++)
291  {
292 
293  tempAlpha = (1 - pGradXYSqrd[i][j] * timeStep_);
294 
295  gvfUPlus1[i][j] = (tempAlpha * gvfVectorU[i][j])
296  + (tempBeta
297  * (gvfVectorU[i+1][j] + gvfVectorU[i][j+1]
298  + gvfVectorU[i-1][j] + gvfVectorU[i][j-1]
299  - (4 * gvfVectorU[i][j])))
300  + c1[i][j] * timeStep_;
301  gvfVPlus1[i][j] = (tempAlpha * gvfVectorV[i][j])
302  + (tempBeta
303  * (gvfVectorV[i+1][j] + gvfVectorV[i][j+1]
304  + gvfVectorV[i-1][j] + gvfVectorV[i][j-1]
305  - (4 * gvfVectorV[i][j])))
306  + c2[i][j] * timeStep_;
307  normUV = 0.f;
308  normUV = sqrtf((gvfUPlus1[i][j] * gvfUPlus1[i][j])
309  + (gvfVPlus1[i][j] * gvfVPlus1[i][j]));
310 
311  if(normUV < 0.00001f)
312  {
313  gvfUPlus1[i][j] = 0.f;
314  gvfVPlus1[i][j] = 0.f;
315  }
316  else
317  {
318  gvfUPlus1[i][j] = gvfUPlus1[i][j] / normUV;
319  gvfVPlus1[i][j] = gvfVPlus1[i][j] / normUV;
320  }
321  }
322  }
323 
324  float **swapPointer;
325 
326  swapPointer = gvfVectorU;
327  gvfVectorU = gvfUPlus1;
328  gvfUPlus1 = swapPointer;
329 
330  swapPointer = gvfVectorV;
331  gvfVectorV = gvfVPlus1;
332  gvfVPlus1 = swapPointer;
333  }
334 
335  delete tempVec;
336 
337  ImageRect *outRect = new ImageRect(workRect);
338  outRect->removeAllAround(1);
339  gvfVector_->setValidRect(outRect);
340  gvfVector_->setRasterAsReference();
341  gvfVector_->forceRasterUpdate();
342  delete outRect;
343 }
