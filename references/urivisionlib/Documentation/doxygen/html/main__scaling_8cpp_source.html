165  // Image 0 is created by computation --> fill the array with RGB values
166  //-------------------------------------------------------------
167  n = 256;
168  //
169  tempBuf = new unsigned char[n * n * 3];
170 
171  if (tempBuf !=NULL)
172  {
173  for (int i=0; i<n; i++)
174  {
175  idx = 3 * n * i;
176 
177  for (int j=0; j<n; j++)
178  {
179  tempBuf[idx++] = (unsigned char) i;
180  tempBuf[idx++] = (unsigned char) 255-j;
181  tempBuf[idx++] = (unsigned char) 0;
182  }
183  }
184  gMyImage[0] = new RasterImage_RGB(n, n, tempBuf);
185  delete tempBuf;
186  }

191  }

193 
194  //-------------------------------------------------------------
195  // Image 1 is created by computation --> fill the array with RGB values
196  //-------------------------------------------------------------
197  n = 256;
198  // I create an image with n rows and 2n columns
199  //
200  tempBuf = new unsigned char[2 * n * n * 3];
201 
202  if (tempBuf !=NULL)
203  {
204  for (int i=0; i<n; i++)
205  {
206  idx = 6 * n * i;
207 
208  for (int j=0; j<2*n; j++)

210

349  );

353  gMyImage[gCurrentImg]->setRenderingScale(++gScaleIndex/10.f);

357  );
