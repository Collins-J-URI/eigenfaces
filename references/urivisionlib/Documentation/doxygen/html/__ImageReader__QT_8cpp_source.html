297 RasterImage* uriVL::simpleGetImageFromImporter_(GraphicsImportComponent importer)
298 {
299  Rect rect;
300  RasterImage *img = NULL;
301  OSErr err = noErr;
302 
303  //----------------------------------------------------------------------
304  /// Fix the rectangle of the importer (origin at (0, 0)
305  //----------------------------------------------------------------------
306  GraphicsImportGetNaturalBounds(importer, &rect);
307  MacOffsetRect(&rect, static_cast<short>(-rect.left), static_cast<short>(-rect.top));
308 
309  //----------------------------------------------------------------------
310  // Get from the importer a description record for the image
311  //----------------------------------------------------------------------
312  ImageDescriptionHandle desc = NULL;
313  err = GraphicsImportGetImageDescription( importer, &desc );
314  FAIL_CONDITION( (err!= noErr) || (desc == NULL) || (*desc == NULL),
315  kFileIOError,
316  "Could not get a description for the selected file");
317 
318  //----------------------------------------------------------------------
319  // Extract fields from the record
320  //----------------------------------------------------------------------
321  int nbCols = (*desc)->width;
322  int nbRows = (*desc)->height;
323  int pixelDepth = (*desc)->depth;
324  int clutID = (*desc)->clutID;
325 
326  //----------------------------------------------------------------------
327  // Allocate the appropriate raster image
328  //----------------------------------------------------------------------
329  switch (pixelDepth)
330  {
331  // binary image
332  case 1: {
333  //CTabHandle colorTable = NULL;
334  //err = GetImageDescriptionCTable(desc, &colorTable);
335  //DisposeCTable( colorTable );
336  FAIL_CONDITION( true,
337  kUnsupportedImageFormatError,
338  "Not ready yet to ready binary images");
339  }
340  break;
341 
342  // 8-bit gray-level image
343  case 40:
344  // so far we only handle "normal" gray-level images
345  FAIL_CONDITION( (clutID != -1) && (clutID != 40),
346  kUnsupportedImageFormatError,
347  "Not ready yet to ready binary images");
348  img = new RasterImage_gray(nbRows, nbCols);
349  // set importer to invert grays
350  err = GraphicsImportSetGraphicsMode(importer, notSrcCopy, NULL);
351  break;
352 
353  // 16-bit color image
354  case 16: {
355  //CTabHandle colorTable = NULL;
356  //err = GetImageDescriptionCTable(desc, &colorTable);
357  //DisposeCTable( colorTable );
358  FAIL_CONDITION( true,
359  kUnsupportedImageFormatError,
360  "Not ready yet to ready binary images");
361  }
362  break;
363 
364  // 24-bit color image and 32-bit color images both are read into
365  // a RasterImage_RGBa object
366  case 24:
367  case 32:
368  img = new RasterImage_RGBa(nbRows, nbCols);
369  break;
370 
371  // Other formats not handled yet
372  default:
373  FAIL_CONDITION( true,
374  kUnsupportedImageFormatError,
375  "Image format not supported yet.");
376  }
377 
378  //----------------------------------------------------------------------
379  // And now we read the image into the new RasterImage's graphworld
380  //----------------------------------------------------------------------
381  img->setAsImporterGraphWorld(importer);
382  // draw the image
383  err = GraphicsImportDraw(importer);
384  // close the importer instance
385  CloseComponent(importer);
386 
387  // let's not forget to set the image's valid data rect before returning it
388  img->setValidRect(0, 0, nbCols, nbRows);
389  return img;
390 }
