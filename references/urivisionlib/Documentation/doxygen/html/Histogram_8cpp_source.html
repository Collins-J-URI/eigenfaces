318 void Histogram::render_(int scaledLeft, int scaledTop, int scaledRight, int scaledBottom,
319  float theZ, bool shifted, const ImagePoint* origin) const
320 {
321  if (!isRenderReady_)
322  computeRenderingParameters_();
323 
324  //----------------------------------------------------------------------
325  // Actual drawing
326  //----------------------------------------------------------------------
327  glPushMatrix();
328  glTranslatef(scaledLeft + horPad_, scaledTop + vertPad_, 0.f);
329 
330  // If we set a background color, then use it.
331  if (backgroundColor_ != NULL)
332  {
333  glColor3fv(backgroundColor_);
334  glBegin(GL_POLYGON);
335  glVertex2i(0, 0);
336  glVertex2i(renderWidth_, 0);
337  glVertex2i(renderWidth_, renderHeight_);
338  glVertex2i(0, renderHeight_);
339  glEnd();
340  }
341 
342  // Now apply the scale so that the histogram will fill the space
343  glPushMatrix();
344  glScalef(horScale_, vertScale_, 1.f);
345 
346  switch (renderMode_)
347  {
348  case LIN_LIN:
349 
350  glColor4fv(getColor());
351  //
352  for (int k=0; k<histVals_; k++)
353  {
354  glBegin(GL_POLYGON);
355  glVertex2f(k, 0);
356  glVertex2f(k+1, 0);
357  glVertex2f(k+1, hist_[k]);
358  glVertex2f(k, hist_[k]);
359  glEnd();
360  }
361  break;
362 
363  case LIN_LOG:
364 
365  glColor4fv(getColor());
366  //
367  for (int k=0; k<histVals_; k++)
368  {
369  const float logHist = logf(1.f + hist_[k]);
370  glBegin(GL_POLYGON);
371  glVertex2f(k, 0);
372  glVertex2f(k+1, 0);
373  glVertex2f(k+1, logHist);
374  glVertex2f(k, logHist);
375  glEnd();
376  }
377  break;
378 
379  case LIN_POLAR:
380  case LOG_POLAR:
381  break;
382 
383  }
384 
385  // draw the axes, if needed
386  glPopMatrix();
387  if (drawAxes_)
388  {
389  if (axesColor_ != NULL)
390  glColor3fv(axesColor_);
391  else
392  glColor4f(0.f, 0.f, 0.f, 1.f);
393 
394  glBegin(GL_LINE_STRIP);
395  glVertex2i(0, renderHeight_);
396  glVertex2i(0, 0);
397  glVertex2i(renderWidth_, 0);
398  glEnd();
399  }
400 
401  glPopMatrix();
402 }

404 void Histogram::computeRenderingParameters_(void) const
405 {
406  //----------------------------------------------------------------------
407  // Determine the width and height dimensions for the rendering.
408  //----------------------------------------------------------------------
409  const int winHeight = getWindowHeight(),
410  winWidth = getWindowWidth();
411 
412  // use only part of the window for rendering
413  renderHeight_ = static_cast<int>((1.f-PAD_FACTOR) * winHeight),
414  renderWidth_ = static_cast<int>((1.f-PAD_FACTOR) * winWidth);
415  // the unused part is horizontal and vertical "padding"
416  horPad_ = (winWidth - renderWidth_) / 2;
417  vertPad_ = (winHeight - renderHeight_) / 2;
418 
419  // we are going to need the max value in the histogram to compute rendering scales
420  long histMax = -1;
421  for (int k=0; k<histVals_; k++)
422  if (hist_[k] > histMax)
423  histMax = hist_[k];
424 
425  //----------------------------------------------------------------------
426  // Determine the horizontal and vertical rendering scales
427  //----------------------------------------------------------------------
428  switch (renderMode_)
429  {
430  case LIN_LIN:
431  horScale_ = (1.f * renderWidth_) / histVals_;
432  vertScale_ = (1.f * renderHeight_) / histMax;
433  break;
434 
435  case LIN_LOG:
436  horScale_ = (1.f * renderWidth_) / histVals_;
437  vertScale_ = (1.f * renderHeight_) / logf(1.f+histMax);
438  break;
439 
440 
441  default:
442  FAIL_CONDITION( true,
443  kFunctionNotImplemented,
444  "Render mode not open for business yet");
445 
446  }
447 
448  isRenderReady_ = true;
449 }
