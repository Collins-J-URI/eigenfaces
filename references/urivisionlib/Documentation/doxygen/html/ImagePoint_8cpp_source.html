217 void ImagePoint::rotate(float angle, const ImagePoint* rotCenter)
218 {
219  const float cT = cosf(angle),
220  sT = sinf(angle);
221 
222  // if the rotation center is null we rotate around the origin
223  if (rotCenter == NULL)
224  {
225  const int x = static_cast<int>(cT*x_ - sT*y_);
226  y_ = static_cast<int>(sT*x_ + cT*y_);
227  x_ = x;
228  }
229  else
230  {
231  if (typeid(*rotCenter) == typeid(uriVL::ImagePoint_F))
232  {
233  const ImagePoint_F *rotCenterF = (ImagePoint_F*) rotCenter;
234  const float xcf = rotCenterF->getXF();
235  const float ycf = rotCenterF->getYF();
236  const float dx = x_ - xcf;
237  const float dy = y_ - ycf;
238  x_ = static_cast<int>(xcf + cT*dx - sT*dy);
239  y_ = static_cast<int>(ycf + sT*dx + cT*dy);
240  }
241  else
242  {
243  const int xc = rotCenter->x_;
244  const int yc = rotCenter->y_;
245  const float dx = x_ - xc;
246  const float dy = y_ - yc;
247  x_ = xc + static_cast<int>(cT*dx - sT*dy);
248  y_ = yc + static_cast<int>(sT*dx + cT*dy);
249  }
250  }
251 }
