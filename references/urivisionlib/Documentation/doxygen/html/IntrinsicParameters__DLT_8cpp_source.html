140 void IntrinsicParameters_DLT::distortPixels(double ux, double uy, double *dx, double *dy) const
141 {
142  // 1. move to the center
143  double cx = ux - Ox_;
144  double cy = uy - Oy_;
145 
146  // 2. calculate scaling factor
147  const double ru2 = cx*cx + cy*cy;
148  const double p = 1.L / (kappa1_ * ru2);
149  const double p2 = p*p;
150  const double bigTerm = 9.L*p + sqrt(81.L*p2 + 12.L*p2*p);
151  const double mu = 1.f + double( bigTerm * pow(18, -1.L/3.L) -
152  pow(2.L/3.L, 1.L/3.L) / bigTerm);
153 
154  // 3. apply scaling
155  *dx = mu * cx;
156  *dy = mu * cy;
157 
158  // 4. apply translation back
159  *dx += Ox_;
160  *dy += Oy_;
161 }
