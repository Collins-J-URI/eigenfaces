289 void RangeMap::updateMesh(void)
290 {
291  // If the range raster is the reference we recalculate the mesh
292  DrawableInformationType dType = getDrawableInformationType();
293  if ((dType == kDense) || (dType == kDenseUncertain))
294  {
295  const ImageRect* vRect = getValidRect();
296  const int iLow = vRect->getTop(),
297  iHigh = vRect->getBottom(),
298  jLow = vRect->getLeft(),
299  jHigh = vRect->getRight();
300 
301  const float *const* range = getShiftedGrayF2D();
302 
303  for (int i=iLow; i<=iHigh; i++)
304  {
305  for (int j=jLow; j<=jHigh; j++)
306  {
307  float r = scaleRange_*range[i][j];
308 
309  // X
310  triMesh_[i][j][0] = r * cosLong_[j] * cosLat_[i];
311 
312  // Y
313  triMesh_[i][j][1] = r * sinLat_[i];
314 
315  // Z (z is in the horizontal direction)
316  triMesh_[i][j][2] = -r * sinLong_[j] * cosLat_[i];

323  }

335 void RangeMap::allocateDenseMesh_(void)
336 {
337  const ImageRect* vRect = getValidRect();
338  const int nbRows = vRect->getHeight(),
339  nbCols = vRect->getWidth();
340  const int iLow = vRect->getTop(),
341  iHigh = vRect->getBottom(),
342  jLow = vRect->getLeft(),
343  jHigh = vRect->getRight();
344 
345  triMesh_ = new float**[nbRows];
346  for (int i=0; i<nbRows; i++)
347  {
348  triMesh_[i] = new float*[nbCols];
349  for (int j=0; j<nbCols; j++)
350  triMesh_[i][j] = new float[3];
351  }
352 
353  float longit = startLongitude_;
354  for (int j=jLow; j<=jHigh; j++, longit += stepLongitude_)
355  {
356  cosLong_[j] = cosf(longit);
357  sinLong_[j] = sinf(longit);
358  }
359 
360  float lat = startLatitude_;
361  for (int i=iLow; i<=iHigh; i++, lat += stepLatitude_)
362  {
363  cosLat_[i] = cosf(lat);
364  sinLat_[i] = sinf(lat);
365  }
366 }
