210 void _MoviePlayer_QT::getFrame_(int n, RasterImage* imgOut)
211 {
212  FAIL_CONDITION( imgOut == NULL,
213  kNullRasterImageError,
214  "null RasterImage passed as destination to a movie player");
215  FAIL_CONDITION( n < 1,
216  kInvalidParameterError,
217  "movie player asked to skip invalid number of frames");
218  setDestination_(imgOut);
219 
220  if (nextTime_ != -1)
221  {
222  OSType mediaType = VideoMediaType;
223 
224  // skip n-1 frames
225  for (int k=1; (k < n) && (nextTime_ != -1); k++)
226  {
227  frameIndex_++;
228 
229  // Get next sample, stop if the time value for next sample is -1 (no more samples)
230  GetMovieNextInterestingTime(movie_, nextTimeMediaSample, 1, &mediaType,
231  currentTime_, 0, &nextTime_, NULL);
232 
233  }
234  if (nextTime_ != -1)
235  {
236  currentTime_ = nextTime_;
237  SetMovieTimeValue(movie_, currentTime_);
238  UpdateMovie(movie_);
239  MoviesTask(movie_, 0);
240  // Get next sample (the end if the time value for next sample is -1,
241  // i.e no more samples)
242  GetMovieNextInterestingTime(movie_, nextTimeMediaSample, 1, &mediaType,
243  currentTime_, 0, &nextTime_, NULL);
244 
245  frameIndex_++;
246 // imgOut->restoreGraphWorld( );
247  }
248  }

250 }
