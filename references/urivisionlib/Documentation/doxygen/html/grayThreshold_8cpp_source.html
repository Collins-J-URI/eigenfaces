391 // the projection mode would be defined. Here I only want to have a 2D

403  glViewport(0, 0, w, h);

498 void handleMouseDrag(int buttonState, int ix, int iy, int winIndex)
499 {
500  ImageRect *myRect;
501  FillingMode localMode;
502 
503  if (buttonState == GLUT_DOWN)
504  {
505  // If the mouse button was not already pressed down, we start recording
506  if (!gMouseDown)
507  {
508  gMouseDown = true;
509  gDragWin = winIndex;
510  gStartDragX = ix;
511  gStartDragY = iy;
512  }
513  }
514  else // state == GLUT_UP
515  {
516  // check that the mouse had been dragged from within the same subwindow and
517  // defines a suffficiently large search region.
518  if (gMouseDown && withinBounds(ix, iy))
519  {
520  int theLeft = MIN(ix, gStartDragX),
521  theTop = MIN(iy, gStartDragY),
522  theWidth = MAX(ix, gStartDragX) - theLeft + 1,
523  theHeight = MAX(iy, gStartDragY) - theTop + 1;
524 
525  if ((theWidth >= kMinSearchDim) && (theHeight >= kMinSearchDim))
526  {
527  myRect = new ImageRect( theLeft, theTop, theWidth, theHeight);
528 
529  // We pick for the selected rectangle a filling mode that difffers from that of
530  // the rest of the subwindow.
531  switch (gModeWin[winIndex])
532  {
533  case kIdentityFill:
534  localMode = kWhiteFill;
535  break;
536  case kOrigColorFill:
537  localMode = kBlackFill;
538  break;
539  case kBlackFill:
540  localMode = kOrigGrayFill;
541  break;
542  case kWhiteFill:
543  localMode = kOrigGrayFill;
544  break;
545  case kMediumGrayFill:
546  localMode = kOrigColorFill;
547  break;
548  case kOrigGrayFill:
549  localMode = kWhiteFill;
550  break;
551  default:
552  break;
553  }
554 
555  gGrayParams->fillMode = localMode;
556  gThresholder->applyInto(gMyImage[UPPER_RIGHT], gGrayParams, myRect,
557  (RasterImage_RGBa *) gMyImage[winIndex]);
558  // Here I *know* that the part of gMyImage[winIndex] that was not selected
559  // still contains valid data, so I force the valid data rectangle to
560  // the full frame
561  gMyImage[winIndex]->setValidRect(gMyImage[winIndex]->getBoundRect());
562  delete myRect;
563  }
564  }
565  gMouseDown = false;
566  gDragWin = -1;
567 
568  myDisplayMain();
569  }
570 }
