15 /*--------------------------------------------------------------------------+
16 | This small demo program illustrates the difference between the bounding |
17 | rectangle and the valid data rectangle, the latter being always a |
18 | subpart (maybe totality) of the former. |
19 | The demo shows how the shift of the bounding and valid data rectangles |
20 | is affected by a scaling operation. |
21 | The '.' and ',' keys (located at '>' and '<') scale down and up the |
22 | rendering of the objects. |
23 | If the user hits the space bar, then the display is switched to the |
24 | "absolute" mode, with the valid data rectangle set at a location set by |
25 | the user. That location is initially (0, 0) and can be modified by |
26 | using the arrow keys. When in that mode, I assigned the same origin to |
27 | raster image and the vector field (not because it has to be so, but |
28 | because I did not ferel it was worth the extra effort to assign them |
29 | different origins) |
30 | |
31 | Author: jean-yves herve', 3D Group for Interactive Visualization, |
32 | Department of Computer Science and Statistics |
33 | University of Rhode Island |
34 +--------------------------------------------------------------------------*/

164 void CreateImageAndField(void)
165 {
166  //-------------------------------------------------------------
167  // the image is created in a rectangle [100, 399]x[50, 249]
168  //-------------------------------------------------------------
169  int nbRows = 200,
170  nbCols = 300;
171  ImageRect myRect(100, 50, nbCols, nbRows);
172 
173  gMyImage = new RasterImage_RGBa(&myRect);
174  // I want the image's bounding rectangle and valid data rectangle to be displayed.
175  gMyImage->setDrawFrame(true);
176  gMyImage->setFrameColor(0.f, 0.f, 1.f);
177  gMyImage->setDrawValidDataFrame(true);
178 
179  // We are not going to initialize the full image (just to show how this is
180  // done and how to set properly the valid data rectangle. Of course, it is
181  // possible to initialize the full image, as we did in Simple IO).
182  int padRow1 = 10,
183  padRow2 = 30,
184  padCol1 = 50,
185  padCol2 = 130;
186 
187  // We initialize the image using the "shifted" base raster
188  unsigned char *const*rgba = gMyImage->getShiftedRaster2D(R_W_ACCESS);
189 
190  // First I fill the entire raster with bogus data
191  int iLow = myRect.getTop(),
192  iHigh = myRect.getBottom(),
193  jLow = myRect.getLeft(),
194  jHigh = myRect.getRight();
195 
196  for (int i=iLow; i<=iHigh; i++)
197  {
198  for (int j=jLow; j<=jHigh; j++)
199  {
200  rgba[i][4*j] = static_cast<unsigned char>(S_N_P_NOISE_SCALE * rand() / 4);
201  rgba[i][4*j+1] = static_cast<unsigned char>(S_N_P_NOISE_SCALE * rand() / 2);
202  rgba[i][4*j+2] = static_cast<unsigned char>(S_N_P_NOISE_SCALE * rand() / 2);
203  //
204  rgba[i][4*j+3] = static_cast<unsigned char>(255);
205  }
206  }
207 
208  // Only part of the raster contains valid data
209  iLow += padRow1;
210  iHigh -= padRow2;
211  jLow += padCol1;
212  jHigh -= padCol2;
213 
214  for (int i=iLow; i<=iHigh; i++)
215  for (int j=jLow; j<=jHigh; j++)
216  {
217  rgba[i][4*j] = static_cast<unsigned char>((i-iLow-nbCols/2.)*(i-iLow-nbCols/2.)/120. + 20);
218  rgba[i][4*j+1] = static_cast<unsigned char>((j-jLow)/4 + 30);
219  rgba[i][4*j+2] = static_cast<unsigned char>(127*(cos(0.06*(i-iLow))*sin(0.1*(j-jLow))+0.5));
220  rgba[i][4*j+3] = static_cast<unsigned char>(255);
221  }
222 
223  // we set the valid data rectangle to reflect the values that were put into
224  // the image.
225  gMyImage->setValidRect(jLow, iLow, jHigh-jLow+1, iHigh-iLow+1);
226 
227 
228 
229  //-------------------------------------------------------------
230  // the vector field is created in a rectangle [150, 449]x[100, 349]
231  //-------------------------------------------------------------
232  nbRows = 250;
233  nbCols = 400;
234  //
235  padRow1 = 10;
236  padRow2 = 30;
237  padCol1 = 30;
238  padCol2 = 70;
239  //
240  myRect.setRect(120, 100, nbCols, nbRows);
241  gMyField = new VectorField_F(&myRect);
242  float **rx = static_cast<float **>(gMyField->getShiftedRasterX2D(R_W_ACCESS));
243  float **ry = static_cast<float **>(gMyField->getShiftedRasterY2D(R_W_ACCESS));
244 
245  // only part of the vector field contains valid data
246  // Only part of the raster contains valid data
247  iLow = myRect.getTop()+3*padRow1;
248  iHigh = myRect.getBottom()-2*padRow2;
249  jLow = myRect.getLeft()+2*padCol1;
250  jHigh = myRect.getRight()-padCol2;
251 
252  for (int i=iLow; i<=iHigh; i++)
253  for (int j=jLow; j<=jHigh; j++)
254  {
255  rx[i][j] = sinf(j/30.f)*cosf(i/20.f);
256  ry[i][j] = sinf(j/50.f);
257  }
258  // Since we initialized the vector field "by hand" we know what the valid data
259  // rectangle is. If we don't set it and the application works in "data safe" mode
260  // the valid data rect will be initialized empty by the constructor and the vector
261  // field would never be rendered.
262  gMyField->setValidRect(jLow, iLow, jHigh-jLow+1, iHigh-iLow+1);
263 
264  // Since we have directly modified the content of the raster, we need to notify
265  // the object that it needs to recalculate its arrows/needles the next time
266  // it renders.
267  gMyField->forceRasterUpdate();
268 
269  // Finally, we want the field to display its bounding frame
270  gMyField->setDrawFrame(true);
271  gMyField->setDrawValidDataFrame(true);
272  gMyField->setFrameColor(0.f, 0.f, 1.f);
273  gMyField->setColor(1.f, 0.f, 0.f);
274 }

336 void myKeyboard(unsigned char c, int x, int y)
337 {
338  float theScale;
339 
340  switch (c)
341  {
342  case 'q':
343  cleanupAndQuit(0);
344  break;
345 
346  // , and . (for < and >) decrease and increase the scale
347  case ',':
348  if (gScaleIndex>1)
349  {
350  gMyImage->setRenderingScale(--gScaleIndex/10.f);
351  gMyField->setRenderingScale(gScaleIndex/10.f);
352  }
353  theScale = gScaleIndex/10.f;
354  break;
355  case '.':
356  if (gScaleIndex<20)
357  {
358  gMyImage->setRenderingScale(++gScaleIndex/10.f);
359  gMyField->setRenderingScale(gScaleIndex/10.f);
360  }
361  theScale = gScaleIndex/10.f;
362  break;
363 
364  case 's':
365  cout << "image index = " << gCurrentImg << "\tscale index = " << gScaleIndex <<
366  "\tscale = " << gMyImage->getRenderingScaleX() << endl;
367  break;
368 
369  // Toggles on-off "display shifted" mode
370  case ' ':
371  gUseOrigin = ! gUseOrigin;
372  break;
373 
374  default:
375  break;
376 
377  }
378 
379  glutPostRedisplay();
380 }
