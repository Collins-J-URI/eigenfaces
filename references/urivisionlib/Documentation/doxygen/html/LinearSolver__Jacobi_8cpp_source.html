423 void LinearSolver_Jacobi::solve_(Matrix* A, Matrix* B, LinearSolverRecord *record,
424  Matrix* currentX, Matrix* previousX, double* invDiag,
425  int incrImprov, bool preserveMatrix)
426 {
427  const int nbRows = B->getNbRows(),
428  nbCols = B->getNbCols();
429 
430  // if one of the diagonal elements is 0 we can't solve the SLE
431  const double* const* a = A->getMatrix();
432  double*const* x = record->sol->getMatrix(R_W_ACCESS);
433  double*const* xk = currentX->getMatrix(R_W_ACCESS);
434  double*const* xkp1 = previousX->getMatrix(R_W_ACCESS);
435  const double*const* b = B->getMatrix(R_W_ACCESS);
436 
437  // The initial state is the one provided by the record passed as parameter
438  for (int i=0; i<nbRows; i++)
439  for (int j=0; j<nbCols; j++)
440  xkp1[i][j] = x[i][j];
441 
442  int nbIterIncrDist = 0;
443  double prevDist = 1.0E10;
444  bool notConverged = true;
445  bool notDiverged = true;
446  do
447  {
448  // swap the "current" and "previous" states
449  {
450  double* const* temp = xk;
451  xk = xkp1;
452  xkp1 = temp;
453  }
454 
455  for (int i=0; i<nbRows; i++)
456  {
457  for (int j=0; j<nbCols; j++)
458  {
459  xkp1[i][j] = b[i][j];
460 
461  for (int l=0; l<nbRows; l++)
462  if (l != i)
463  xkp1[i][j] -= a[i][l]*xk[l][j];
464 
465  xkp1[i][j] *= invDiag[i];
466  }
467  }
468 
469  // compute the change between initial and previous state
470  // We use infinity norm: the distance for all terms must be
471  // less than the threshold
472  double dist = 0.L;
473  for (int i=0; i<nbRows; i++)
474  {
475  for (int j=0; j<nbCols; j++)
476  {
477  double dx = fabs(xkp1[i][j] - xk[i][j]);
478  if (dx > dist)
479  dist = dx;
480  }
481  }
482  if (dist < STOP_ITERATION_THRESHOLD)
483  notConverged = false;
484  else if (dist > prevDist)
485  {
486  nbIterIncrDist++;
487  if (nbIterIncrDist > 6)
488  notDiverged = false;
489  }
490  else
491  {
492  nbIterIncrDist = 0;
493  }
494  }
495  while (notConverged && notDiverged);
496 
497  if (notDiverged)
498  {
499  record->status = LINEAR_SOLVER_SUCCEDED;
500 
501  // after we leave the loop, copy the last computed state into the record to return
502  for (int i=0; i<nbRows; i++)
503  for (int j=0; j<nbCols; j++)
504  x[i][j] = xkp1[i][j];
505 
506  }
507  else
508  {
509  record->status = LINEAR_SOLVER_FAILED;
510  }
511 }
