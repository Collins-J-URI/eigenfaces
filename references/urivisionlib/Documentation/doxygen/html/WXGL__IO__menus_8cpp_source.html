258 void WXGL_IO_Frame::ParamSize(wxCommandEvent& myEvent)
259 {
260  // set up default items for the dialog box
261  wxString strMessage("defMessage");
262  wxString strDefaultValue;
263  wxPoint pointLoc(242,242);
264  char abValue[32];
265 
266  // check event to see if this is looking for rows or columns
267  // set caption string accordingly
268  char *pszMessage = 0;
269  char *pszCaption = "Enter Test Image Size Parameter";
270  if (myEvent.GetId() == MENU_ID_PARAM_COLS)
271  {
272  pszMessage = "Enter # of COLUMNS in test image";
273  sprintf(abValue,"%d",mp_Canvas->GetTestImageCols());
274  }
275  else
276  {
277  pszMessage = "Enter # of ROWS in test image";
278  sprintf(abValue,"%d",mp_Canvas->GetTestImageRows());
279  }
280  strDefaultValue = abValue;
281  wxTextEntryDialog *ptDialog = new wxTextEntryDialog(this,wxString(pszMessage),wxString(pszCaption),strDefaultValue,wxOK|wxCANCEL|wxCENTRE, pointLoc);
282 
283  if (ptDialog != 0)
284  {
285  int nStatus = ptDialog->ShowModal();
286 
287  if (nStatus==wxID_OK)
288  {
289  wxString strAnswer = ptDialog->GetValue();
290  if(strAnswer.IsNumber())
291  {
292  unsigned long ulAnswer;
293  strAnswer.ToULong(&ulAnswer);
294  if (myEvent.GetId() == MENU_ID_PARAM_COLS)
295  {
296  mp_Canvas->SetTestImageCols(ulAnswer);
297  }
298  else if (myEvent.GetId() == MENU_ID_PARAM_ROWS)
299  {
300  mp_Canvas->SetTestImageRows(ulAnswer);
301  }
302  mp_Canvas->CreateTestImage();
303  mp_Canvas->UpdateCanvas();
304  }
305  else
306  {
307  wxMessageBox(
308  _T("Invalid Data Entry\n")+strAnswer,
309  _T("ERROR"),
310  wxOK|wxICON_INFORMATION,
311  this
312  );
313  }
314  }
315  }
316  else
317  {
318  wxMessageBox(
319  _T("Error creating text dialog box\n"),
320  _T("ERROR"),
321  wxOK|wxICON_INFORMATION,
322  this
323  );
324  }
