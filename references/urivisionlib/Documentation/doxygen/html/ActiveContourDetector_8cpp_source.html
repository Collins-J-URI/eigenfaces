256 IterationStatus ActiveContourDetector::iterate(int nbSteps)
257 {
258  // if this balloon detector has been set to operate in "step by step" mode
259  // and has been properly configured (image and input contour sent in),
260  // execute the number of steps specified.
261  if (setupCompleted_){
262  bool converged = false;
263  int count = 0;
264  for(int i=0; (i < nbSteps) && !converged ; i++){
265  IterationStatus stepStatus = oneStep_();
266  converged = stepStatus.hasConverged();
267  count++;
268  }
269  if(converged){
270  cout<< count << " iterations (convergence)" << endl;
271  }
272 
273  }
274 
275  return getStatus();
276 }

279 // v(s) = (x(s), y(s))
280 // elasticity --> .5 * INTEGRAL( alpha(s) |Vs|^2) with respect to ds where Vs = dv(s)/ds
281 // bending --> .5 * INTEGRAL( beta(s) |Vss|^2 with respect to ds
282 void ActiveContourDetector::calculateInternalForce_(void)
283 {
284  const list<Vector2D_F*> *internalForce = internalForce_->getForce();
285  const Contour_F* contour = internalForce_->getContour();
286  const list<ImagePoint_F*> *contourPts = contour->getPoints();
287 
288  list<ImagePoint_F*>::const_iterator itm2, itm1, itp1, itp2, contourEnd = contourPts->end();
289  list<Vector2D_F*>::const_iterator intIter;
290  float w2 = DEFAULT_BETA_2ND_DERIV,
291  w2Times4 = 4.f*w2,
292  w2Times6 = 6.f*w2,
293  w1 = DEFAULT_ALPHA_1ST_DERIV,
294  w1Times2 = 2.f*w1;
295  float a, b, c;
296  float im2X, im1X, ip1X, ip2X, iX;
297  float im2Y, im1Y, ip1Y, ip2Y, iY;
298 
299  // begin by setting all the contour pts to the
300  // correct positions
301  itm2 = contourPts->end();
302  itm2--;
303  itm1 = itm2;
304  itm2--;
305  itp2 = contourPts->begin();
306  itp2++;
307  itp1 = itp2;
308  itp2++;
309  //
310  intIter = internalForce->begin();
311 
312  for(list<ImagePoint_F*>::const_iterator iter = contourPts->begin(); iter != contourEnd; iter++, intIter++){
313  /*
314  The following stems from the equations below.
315  We basically loop through each point call them C, and
316  compute the 2nd and 4th derivatives by the difference of points
317 
318  -(W1*Ci')' + (W2*Ci")" = AV
319  -W1*Ci" + W2*Ci"" = AV
320  Ci" = C(i+1) - 2 * Ci + C(i-1)
321  Ci"" = C(i+2) - 4 * C(i+1) + 6 * Ci - 4C(i-1) + C(i-2)s
322  */
323 
324  a = -w2;
325  b = w1 + w2Times4;
326  c = -w1Times2 - w2Times6;
327 
328  im2X = (*itm2)->getX();
329  im1X = (*itm1)->getX();
330  iX = (*iter)->getX();
331  ip1X = (*itp1)->getX();
332  ip2X = (*itp2)->getX();
333 
334  im2Y = (*itm2)->getY();
335  im1Y = (*itm1)->getY();
336  iY = (*iter)->getY();
337  ip1Y = (*itp1)->getY();
338  ip2Y = (*itp2)->getY();
339 
340  float vectX = a * (im2X + ip2X ) + b * (im1X + ip1X ) + c * iX,
341  vectY = a * (im2Y + ip2Y ) + b * (im1Y + ip1Y ) + c * iY,
342  normVect2 = vectX*vectX + vectY*vectY;
343 
344  if (normVect2 > DEFAULT_MAX_INTERNAL_FORCE) {
345  float scale = static_cast<float>(DEFAULT_MAX_INTERNAL_FORCE / sqrt(normVect2));
346  vectX *= scale;
347  vectY *= scale;
348  }
349 
350  (*intIter)->setX(timeStep_ * vectX);
351  (*intIter)->setY(timeStep_ * vectY);
352 
353  itm2 = itm1;
354  itm1 = iter;
355  itp1 = itp2;
356  itp2++;
357  if (itp2 == contourPts->end()){
358  itp2 = contourPts->begin();
359  }
360  }
361 }
