261 IterationStatus ActiveContourDetector_balloon::oneStep_(void){
262 
263  /* here i calc all the forces then add them to the total force
264  then i apply the total force to the contour
265  after this i set it back to zero for the next iteration.
266 
267  these forces stem from the equation
268  Eint + Eimage = 0
269  where
270  Eint = Internal Force
271  Eimage = Balloon Force -> (-k1(vectorNormal2Curve) + k2(grad/normOfGrad))
272  */
273  IterationStatus status;
274  Contour_F* contour = getContour_(R_W_ACCESS);
275 
276  // pretty smooth.. just calc the forces on each point
277  calculateInternalForce_();
278  calculateEdgeForce_(contour, gradOfNormSqrdOfGrad_, edgeForce_);
279  calculateShrinkForce_(contour, shrinkForce_);
280 
281  // add them up to be one total force
282  totalForce_->setForcesToZero();
283  totalForce_->addForce(internalForce_);
284  totalForce_->addForce(edgeForce_);
285  totalForce_->addForce(shrinkForce_);
286 
287  // apply the force and if it has converged then don't insert pts
288  bool contourHasConverged = totalForce_->applyForceToContour();
289  if (contourHasConverged)
290  {
291  status.setConverged();
292  }
293  else
294  {
295  // check to see if after the move we need to insert a point
296  searchAndInsertPoints_(contour, normSqrdOfGrad_);
297  }
298  nbIterations_++;
299 
300  return status;

669  float Cx = sumX/ptCount,
670  Cy = sumY/ptCount;
671 
672  float distToEnd = sqrtf((Cx - endX)*(Cx - endX) + (Cy - endY)*(Cy - endY)),
673  distToLast = sqrtf((Cx - lastCx)*(Cx - lastCx) + (Cy - lastCy)*(Cy - lastCy)),
674  distToStart = sqrtf((Cx - startX)*(Cx - startX) + (Cy - startY)*(Cy - startY));
675 
676  if (distToEnd > MIN_DIST_BTWN_PTS && distToLast > MIN_DIST_BTWN_PTS && distToStart > MIN_DIST_BTWN_PTS)
677  {
678  if (!swapped0F)
679  {
680  lastCx = static_cast<int>(roundf(Cx));
681  lastCy = static_cast<int>(roundf(Cy));
682 
683  contour->insertPoint(currIndex,Cx,Cy);
684  edgeForce_->insertIntoForceVector(currIndex, 0.f, 0.f);
685  internalForce_->insertIntoForceVector(currIndex, 0.f, 0.f);
686  shrinkForce_->insertIntoForceVector(currIndex, 0.f, 0.f);
687  totalForce_->insertIntoForceVector(currIndex, 0.f, 0.f);
688 
689 
690  if (iter_1 != contourList->begin() || currIndex != 0)
691  {
692  currIndex++;
693  lastInsert = currIndex;
694  }
695 
696  } else {
697 
698  lastCx = static_cast<int>(roundf(Cx));
699  lastCy = static_cast<int>(roundf(Cy));
700 
701  contour->insertPoint(currIndex,Cx,Cy);
702  edgeForce_->insertIntoForceVector(lastInsert, 0.f, 0.f);
703  internalForce_->insertIntoForceVector(lastInsert, 0.f, 0.f);
704  shrinkForce_->insertIntoForceVector(lastInsert, 0.f, 0.f);
705  totalForce_->insertIntoForceVector(lastInsert, 0.f, 0.f);
706  if (iter_1 != contourList->begin() || currIndex != 0)
707  {
708  currIndex++;
709  if (lastInsert != 0)
710  {
711  lastInsert--;
712  } else {
713  lastInsert = 0;
714  }
715  }
716  }
717  }
718  }
719  }
720 
721  bres_d += dy;
722  if ( (bres_d << 1) >= dx )
723  {
724  y0 += BRESENHAM_STEP_SIZE;
725  bres_d -= dx;
726  }
727  }
728 
729  iter_0 = iter_1;
730  iter_1++;
731  currIndex++;
732  }// ; //keep looping until we've visited all the contour points
733
