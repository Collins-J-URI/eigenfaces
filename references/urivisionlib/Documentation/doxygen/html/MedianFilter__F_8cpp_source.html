312 // Computation of the median using N. Wirth's algorithm
313 // Implementation derived from kth_smallest implementation of N Devillard.
314 // Note that if the array is even-sized, the function does not return the average of
315 // the two central values but the lower median.
316 float getMedian_(float* dat, int n)
317 {
318  const int k = (n%2) == 0 ? n/2 - 1 : n/2;
319  // Note for those who might want to reuse that code elsewhere: x should be of the
320  // same type as data_
321  float x;
322  int i, j,
323  l = 0,
324  m = n - 1;
325 
326  while (l<m)
327  {
328  x = dat[k];
329  i = l;
330  j = m;
331  do
332  {
333  while (dat[i] < x)
334  i++;
335  while (dat[j] > x)
336  j--;
337 
338  if (i <= j)
339  {
340  float temp = dat[i];
341  dat[i] = dat[j];
342  dat[j] = temp;
343  i++;
344  j--;
345  }
346 
347  } while (i <= j);
348 
349  if (j < k)
350  l = i;
351  if (i > k)
352  m = j;
353  }
354 
355  return dat[k];
356 }
