831  float theZ, bool shifted, const ImagePoint* origin) const
832 {
833  //--------------------------------------------------------------
834  // Part 1: make sure that what we draw is up to date
835  //--------------------------------------------------------------
836  // Binary images are a special case. OpenGL cannot draw QD binary
837  // raster, so we need to use the gray raster instead.
838  bool binaryRaster = (pattern_->getBaseRasterType() == kBinaryRaster);
839  if (binaryRaster)
840  {
841  // if the gworld is the reference and has been modified, or
842  // the gray raster has been modified, update
843  /// the gray raster (re-binarize it).
844  if (pattern_->graphWorldIsReference() || pattern_->referenceHasChanged())
845  pattern_->requestRasterUpdate( );
846 
847  }
848  else
849  // For other image types, the base raster is the one to draw, so if it
850  // is not up-to-date, update it.
851  if ( !pattern_->graphWorldIsReference() && pattern_->referenceHasChanged() )
852  pattern_->requestRasterUpdate( );
853 
854 
855 
856  //--------------------------------------------------------------
857  // Part 2: render the blob pattern
858  //--------------------------------------------------------------
859  // I apply a -1 scaling vertical scaling to my image so that it gets displayed
860  // with the right side up.
861  GLenum rasterType;
862  int pixelDepth;
863  switch (pattern_->getBaseRasterType())
864  {
865  case kBinaryRaster:
866  case kGrayRaster:
867  rasterType = GL_LUMINANCE;
868  pixelDepth = 1;
869  break;
870 
871  case kRGB24Raster:
872  rasterType = GL_RGB;
873  pixelDepth = 3;
874  break;
875 
876  case kRGBa32Raster:
877  rasterType = GL_RGBA;
878  pixelDepth = 4;
879  break;
880 
881  case kUndefRasterType:
882  case kLongGrayRaster:
883  case kaRGB16Raster:
884  case kaRGB32Raster:
885  FAIL_CONDITION(true, kNoCode, "Unhandled case in a switch statement");
886  }
887  const unsigned char* const* raster = pattern_->getRaster2D();
888 
889  const float SCALE_X = getRenderingScaleX(),
890  SCALE_Y = getRenderingScaleY();
891 
892  glPushMatrix();
893  glLoadIdentity();
894  glPixelZoom(SCALE_X, -SCALE_Y);
895 
896  const LineSegDeque segDeque = blob_->getSegDeque();
897  LineSegDeque::const_iterator lastLine = segDeque.end();
898  for (LineSegDeque::const_iterator lineIter = segDeque.begin(); lineIter != lastLine; lineIter++)
899  {
900  list<HorizontalSegment*>::iterator lastSeg = (*lineIter)->end();
901  for (list<HorizontalSegment*>::iterator segIter = (*lineIter)->begin(); segIter != lastSeg; segIter++)
902  {
903  HorizontalSegment *theSeg;
904  theSeg = (*segIter);
905 
906  glRasterPos3f(scaledLeft + SCALE_X*theSeg->leftX,
907  scaledTop - SCALE_Y*theSeg->y, theZ);
908  glDrawPixels(theSeg->rightX - theSeg->leftX + 1, 1,
909  rasterType, GL_UNSIGNED_BYTE,
910  raster[theSeg->y] + pixelDepth*theSeg->leftX);
911  }
912  }
913  glPopMatrix();
914 
915  //--------------------------------------------------------------
916  // Part 3: if needed, draw the blob's contour
917  //--------------------------------------------------------------
918  if (drawContour_)
919  blob_->renderContour(theZ, shifted, origin);
920 
921 }
