130 void VirtualPart3D::render(int camIndex)
131 {
132  // ask each of the part's points to render themselves
133  vector<VirtualFace3D*>::iterator faceIter = faces_.begin();
134  vector<VirtualPoint3D*>::iterator vertIter = vertices_.begin();
135  vector<NormalVector*>::iterator normIter = normals_.begin();
136  int nFaces = faces_.size();
137  if (renderMode_ == INTERPOSE)
138  {
139  // for each face in the part
140  for (int i=0; i<nFaces; i++)
141  {
142  VirtualFace3D *face = faceIter[i];
143 
144  glBegin(GL_POLYGON);
145  vector<int>::iterator faceVertIter = (face->vertexIndex_).begin();
146  int nbFaceVerts = face->vertexIndex_.size();
147  for (int j=0; j<nbFaceVerts; j++)
148  {
149  int vertIndex = faceVertIter[j];
150  VirtualPoint3D *vert = vertIter[vertIndex];
151  vector<int>::iterator faceVertNormIter = vert->normalIndex_.begin();
152  NormalVector *norm = normIter[faceVertNormIter[i]];
153 
154  glNormal3d(norm->nx, norm->ny, norm->nz);
155 
156  if (vert->getVisibility(camIndex)->code == VISIBLE)
157  glColor4f(.0f, .8f, 0.f, 1.f);
158  else if (vert->getVisibility(camIndex)->code == INVISIBLE)
159  glColor4f(0.f, 0.f, 1.f, 0.f);
160  else if (vert->getVisibility(camIndex)->code == CONTACT)
161  glColor4f(1.f, 0.f, 0.f, 0.f);
162  else // if (vert->getVisibility(camIndex)->code == UNKNOWN)
163  glColor4f(0.f, 1.f, 1.f, 0.f);
164 
165  glVertex3dv(vert->getXYZ());
166  }
167  glEnd();
168  }
169 
170  if (drawFrame_)
171  drawReferenceFrame_();
172 
173  if (drawVertices_)
174  {
175  // virtual points are shaded
176  glPolygonMode(GL_FRONT_AND_BACK, GL_FILL);
177  glDepthMask (GL_TRUE);
178  glDisable (GL_BLEND);
179  vector<VirtualPoint3D *>::iterator ptIter, endVertices = vertices_.end();
180  for (ptIter = vertices_.begin(); ptIter != endVertices; ptIter++)
181  (*ptIter)->render_(camIndex);
182 
183  }
184  }
185 
186  // Rendering mode = GEOMETRY_TEST
187  else
188  {
189  // face edges in wireframe
190  glPolygonMode(GL_FRONT_AND_BACK, GL_LINE);
191 
192  // for each face in the part
193  for (int i=0; i<nFaces; i++)
194  {
195  VirtualFace3D *face = faceIter[i];
196 
197  glColor4f(0.f, .8f, 0.f, 1.f);
198  glBegin(GL_POLYGON);
199  vector<int>::iterator faceVertIter = (face->vertexIndex_).begin();
200  int nbFaceVerts = face->vertexIndex_.size();
201  for (int j=0; j<nbFaceVerts; j++)
202  {
203  int vertIndex = faceVertIter[j];
204  VirtualPoint3D *vert = vertIter[vertIndex];
205 
206  glVertex3dv(vert->getXYZ());
207  }
208  glEnd();
209  }
210 
211  if (drawFrame_)
212  drawReferenceFrame_();
213 
214  // virtual points are shaded
215  glPolygonMode(GL_FRONT_AND_BACK, GL_FILL);
216  glDepthMask (GL_TRUE);
217  glDisable (GL_BLEND);
218  vector<VirtualPoint3D *>::iterator ptIter, endVertices = vertices_.end();
219  for (ptIter = vertices_.begin(); ptIter != endVertices; ptIter++)
220  (*ptIter)->render_(camIndex);
221 
222  }
