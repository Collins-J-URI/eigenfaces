224 void RectifiedStereoMatcher_JJL::computeIntegralImage_(int disp)
225 {
226  int iLow = rect_->getTop(),
227  iHigh = rect_->getBottom(),
228  jLow = rect_->getLeft() + disp,
229  jHigh = rect_->getRight(),
230  newWidth = jHigh - jLow + 1,
231  height = iHigh - iLow + 1,
232  i, j;
233  unsigned char **leftGray = leftImage_->getShiftedGray2D();
234  unsigned char **rightGray = rightImage_->getShiftedGray2D();
235 
236  // For each disparity considered
237  for (int d=0; d<=maxDisp_; d++)
238  {
239  // calculate the error at each pixel
240  calculateError_(leftGray, rightGray, d, iLow, iHigh, jLow, jHigh);
241 
242  // perform the summation along the columns
243  for (i=iLow;i<=iHigh; i++)
244  {
245  prevColSum_[i]= 0;
246  prevColSqSum_[i] = 0;
247  }
248  for (j=jLow;j<=jHigh; j++)
249  {
250  for (i=iLow;i<=iHigh; i++)
251  {
252  integralError_[disp][i][j] += prevColSum_[j];
253  integralSqrError_[disp][i][j] += prevColSqSum_[j];
254  prevColSum_[j] = integralError_[disp][i][j];
255  prevColSqSum_[j] = integralSqrError_[disp][i][j];
256  }
257  }
258 
259  // perform the summation along the rows
260  for (j=jLow;j<=jHigh; j++)
261  {
262  prevRowSum_[j]= 0;
263  prevRowSqSum_[j] = 0;
264  }
265  for (i=iLow;i<=iHigh; i++)
266  {
267  for (j=jLow;j<=jHigh; j++)
268  {
269  integralError_[disp][i][j] += prevRowSum_[j];
270  integralSqrError_[disp][i][j] += prevRowSqSum_[j];
271  prevRowSum_[j] = integralError_[disp][i][j];
272  prevRowSqSum_[j] = integralSqrError_[disp][i][j];
273  }
274  }
275 
276  // set valid data rectangles
277  errorIntegralImage_[d]->setValidRect(jLow, iLow, newWidth, height);
278  sqrErrorIntegralImage_[d]->setValidRect(jLow, iLow, newWidth, height);
279  errorIntegralImage_[d]->setFloatConversionMode(kPositiveFloat);
280  sqrErrorIntegralImage_[d]->setFloatConversionMode(kPositiveFloat);
281  }
282 }

289 void RectifiedStereoMatcher_JJL::calculateError_(unsigned char** leftGray,
290  unsigned char** rightGray, int d, int iLow,
291  int iHigh, int jLow, int jHigh)
292 {
293  int i,j;
294  float a,b,c,errorL,errorR,minError;
295 
296  for (i=iLow; i<=iHigh; i++)
297  {
298 
299  for (j=jLow; j<=jHigh; j++)
300  {
301 
302  a = abs(leftGray[i][j] - rightGray[i][j]);
303  if (j != jLow)
304  b = abs(leftGray[i][j] - (rightGray[i][j-d] + rightGray[i][j-d-1])*0.5);
305  else
306  b = a;
307  if (j != jHigh)
308  c = abs(leftGray[i][j] - (rightGray[i][j-d] + rightGray[i][j-d+1])*0.5);
309  else
310  c = a;
311 
312  errorL = MIN(a,c);
313  if (b <errorL)
314  errorL = b;
315 
316  if (j != jLow)
317  b = abs((leftGray[i][j] + leftGray[i][j-1])*0.5 - rightGray[i][j-d]);
318  else
319  b = a;
320  if ( j != jHigh)
321  c = abs((leftGray[i][j] + leftGray[i][j+1])*0.5 - rightGray[i][j-d]);
322  else
323  c = a;
324  errorR = MIN(a,c);
325  if (b <errorR)
326  errorR = b;
327 
328  minError = MIN(errorL,errorR);
329 
330  integralError_[d][i][j] = minError;
331  integralSqrError_[d][i][j] = minError*minError;
332  }
333  }
