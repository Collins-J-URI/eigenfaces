353 RasterImage* PatternTracker2D::grabImage_(RasterImage* img,
354  int predX, int predY)
355 {
356  RasterImage* grabbedImg = NULL;
357  const ImageRect *imgRect = img->getValidRect(),
358  *blobRect = pattern_->getBlob()->getValidRect();
359  const int blobWidth = blobRect->getWidth();
360  const int blobHeight = blobRect->getHeight();
361 
362  int newSize = static_cast<int>(blobWidth * rectRatio_);
363  int temp = 1, temp2 = 2;
364  // get powers of 2 that bracket the blob's width
365  while (temp2 < newSize)
366  {
367  temp = temp2;
368  temp2 *= 2;
369  }
370  int newWidth = temp;
371 
372  // do same thing for the height
373  newSize = static_cast<int>(blobHeight * rectRatio_);
374  temp = 1, temp2 = 2;
375  while (temp2 < newSize)
376  {
377  temp = temp2;
378  temp2 *= 2;
379  }
380  int newHeight = newSize;
381 
382  // Now check which dimensions the warper advocates for this pattern
383  int loWidth, hiWidth, loHeight, hiHeight;
384  matcher_->getImageSuitableDimensions(newWidth, newHeight, &loWidth, &loHeight,
385  &hiWidth, &hiHeight);
386 
387  int patLeft = predX - (newWidth - blobWidth) / 2;
388  int patTop = predY - (newHeight - blobHeight) / 2;
389  int patRight = patLeft + newWidth;
390  int patBottom = patTop + newHeight;
391 
392  // shift lf & tp (if necessary) to make the bigRect
393  // always inside the image
394  int rtMore = patRight - imgRect->getWidth();
395  int btMore = patBottom - imgRect->getHeight();
396  if (patLeft < 0)
397  patLeft = 0;
398  else if (rtMore > 0)
399  patLeft -= rtMore;
400  if (patTop < 0)
401  patTop = 0;
402  else if (btMore > 0)
403  patTop -= btMore;
404 
405  grabPatternRect_->setRect(patLeft, patTop, newWidth, newHeight);
406  grabPoint_->setCoordinates(0, 0);
407  grabbedImg = RasterImage::grab(img, grabPatternRect_, grabPoint_);
408 
409  // set matchX_ and matchY_ which are inside this newly grabbed image for later matching
410  matchX_ = predX - patLeft;
411  matchY_ = predY - patTop;
412 
413  // update grabbed image dimensions
414  grabbedImgWidth_ = newWidth;
415  grabbedImgHeight_ = newHeight;
416 
417  // update grabbedImgLf_ & grabbedImgTp
418  grabbedImgLf_ = patLeft;
419  grabbedImgTp_ = patTop;
420 
421  return grabbedImg;
