278 | Description: Computes the average normal vector of a polygonal face that is |
279 | specified by the coordinates of its vertices as a 2D array |
280 | The vertices are expected to be listed counterclockwise. |
281 | |
282 | Comment: This function might as well be private, since it will rarely be |
283 | called directly. One of the next 2 methods, which receive |
284 | among their parameters a 1D array of vertex indexes (for a |
285 | polyhedron object), will/should be called instead. |
286 +----------------------------------------------------------------------------------*/
287 void GraphicComponent3D::faceAverageNormal(GLuint n, GLfloat** v, GLfloat* nVect)
288 {
289  GLfloat nv[3],
290  e1[3], e2[3],
291  theta,
292  totalAngle;
293  int i, i1, i2, k;
294 
295  // The difficulty in computing the average normal vector of a polygonal
296  // patch is that we can't just take any 3-tuple of vertices and compute a
297  // cross product because we don't know which way the computed normal vector
298  // will be pointing (i.o.w., do the 3 vertices form a convex or concave corner
299  // of the polygon?). Here I am using the fact that going around the polygon
300  // counterclockwise really corresponds to doing a +2pi rotation around the
301  // normal vector. So here is how I proceed. Start from the first 3 points,
302  // compute the cross product, normalize. This is +/- the desired normal vector
303  // (+ some error), but I don't know which.
304  // For all v_i, v_i+1 v_i+2 3-tuples of vertices, compute the cross product,
305  // if it points in the same direction as the first one, increment the angle,
306  // else decrement it. In any case, add up +/- computed normal vector for the
307  // average. In the end, if the sum of angles is 2pi, we have the right normal
308  // vector, otherwise we have its opposite.
309 
310  //-------------------------------------------------------------------------
311  // Initialize the computation with the first 3 vertices
312  //-------------------------------------------------------------------------
313  // 1. Compute the edge vectors and normalize them
314  for (k=0; k<3; k++)
315  {
316  e1[k] = v[1][k] - v[0][k];
317  e2[k] = v[2][k] - v[0][k];
318  }
319  normalizeVector(e1);
320  normalizeVector(e2);
321 
322  // 2. Compute the cross product. Store its normalized version as the
323  // first normal vector. Compute the dot product and store its arcCos
324  // as the angle of the rotation.
325  commonNormalVector(e1, e2, nVect);
326  totalAngle = acosf(dotProduct(e1, e2));
327 
328  //-------------------------------------------------------------------------
329  // Go through the polygon, counterclockwise, and repeat the following
330  // step for all 3-tuples of consecutive vertices.
331  //-------------------------------------------------------------------------
332  for (i=1; i< static_cast<int>(n); i++)
333  {
334  i1 = (i+1)%n;
335  i2 = (i+2)%n;
336 
337  // 1. Compute the edge vectors and normalize them
338  for (k=0; k<3; k++)
339  {
340  e1[k] = v[i1][k] - v[i][k];
341  e2[k] = v[i2][k] - v[i][k];
342  }
343  normalizeVector(e1);
344  normalizeVector(e2);
345 
346  // 2. compute the cross product and the angle of the rotation
347  commonNormalVector(e1, e2, nv);
348  theta = acosf(dotProduct(e1, e2));
349 
350  // 3. If nv points in the same direction as the initial normal vector
351  // everything is added up.
352  if (dotProduct(nv, nVect) > 0)
353  {
354  for (k=0; k<3; k++)
355  nVect[k] += nv[k];
356 
357  totalAngle += theta;
358  }
359  // if the vectors point in opposite directions, we substract.
360  else
361  {
362  for (k=0; k<3; k++)
363  nVect[k] -= nv[k];
364 
365  totalAngle -= theta;
366  }
367  }
368 
369  //-------------------------------------------------------------------------
370  // Once we have gone through all vertices, we check the sign of the total
371  // rotation we computed. If it is > 0 (should be 2Pi), then we just have
372  // to normalize the normal vector. If it is negative (should be -2Pi),
373  // then we must invert the normal vector before we return it.
374  //-------------------------------------------------------------------------
375  if (totalAngle < 0.f)
376  for (k=0; k<3; k++)
377  nVect[k] *= -1.f;
378 
379  normalizeVector(nVect);
380 }
