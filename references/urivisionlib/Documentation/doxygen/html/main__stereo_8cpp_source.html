281 //---------------------------------------------------------------------------------
282 // Create two range (depth) maps from the disparity map stored in gMyDisparity:
283 // - a dense depth map computed at each image point
284 // - a sparse depth map computed at a number of random points.

286 void computeDepth(void)
287 {
288  //------------------
289  // Step 1: create the IntrinsicParameters object
290  //------------------
291  // I know next to nothing about the cameras that took these images, so
292  // I made up all the data. I create an ideal pinhole camera
293  int nbRows = gMyDisparity->getHeight(),
294  nbCols = gMyDisparity->getWidth();
295  IntrinsicParameters_pinhole *camParams;
296  // Since this is a "pinhole" camera we are dealing with here, not a real one, I need
297  // to compute my focal length in "pixel" units
298  float focPix = 0.5f * gMyLeftImage->getHeight() / tan(DEG_TO_RAD*FOV_Y/2);
299  camParams = new IntrinsicParameters_pinhole(nbCols, nbRows, focPix, focPix);
300 
301  //------------------
302  // Step 2: compute depth at each pixel
303  //------------------
304  // I don't know the geometry of the stereo pair either, only that they are supposed
305  // to form a parallel pair. So I make up a baseline.
306  //float baseline = 1.f;
307 
308  // Here we break the process into a series of baby steps. I create first a
309  // "depth" raster image that will store the depth computed from the disparity
310  // Note that my calculation of "depth" uses completely made up coefficients
311  gMyDepth = new RasterImage_gray_F(nbRows, nbCols);
312  ImageRect *validRect = new ImageRect(0, 0, nbCols, nbRows);
313  float *const* depth = gMyDepth->getShiftedGrayF2D(R_W_ACCESS);
314  const unsigned char *const* disp = gMyDisparity->getShiftedGray2D();
315  for (int i=0; i<nbRows; i++)
316  for (int j=0; j<nbCols; j++)
317  {
318  depth[i][j] = 75.f * (1.f + 1.f / (2.f + disp[i][j]));
319  if (depth[i][j] > MAX_DEPTH)
320  depth[i][j] = MAX_DEPTH;
321  }
322  gMyDepth->setValidRect(validRect);
323  gMyDepth->setFloatConversionMode(kRangeFloat);
324 
325 
326  //------------------
327  // Step 3: create the dense DepthMap object
328  //------------------
329  gMyDenseMap = new DepthMap(gMyDepth, camParams, 1.f);
330  gMyDenseMap->setFloatConversionMode(kRangeFloat);
331  gMyDenseMap->setResolution(gMeshResolution);
332  gMyDenseMap->setWireframe(true);
333  gMyDenseMap->setMaterial(0.8f, 0.1f, 0.1f, .5f, .5f, .5f, 0.3f, .3f, .3f, .5f);
334 }
