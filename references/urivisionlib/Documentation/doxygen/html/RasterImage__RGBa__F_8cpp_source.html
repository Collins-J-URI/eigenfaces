1191 {
1192  const ImageRect* validRect = getValidRect();
1193  const int iLow = validRect->getTop(),
1194  iHigh = validRect->getBottom(),
1195  jLow = validRect->getLeft(),
1196  jHigh = validRect->getRight();
1197  unsigned char *const* rgba;
1198  float *const* fRed;
1199  float *const* fGreen;
1200  float *const* fBlue;
1201 
1202  // Before we call functions of the parent class with side effects on the
1203  // "reference" flags, we copy their values to know what actually needs
1204  // to be updated
1205  bool isCharToFloatUpdate = graphWorldIsReference();
1206 
1207  // Not the prettiest solution, but gets the job done to preserve logical constness
1208  RasterImage_RGBa_F* updateableThis = const_cast<RasterImage_RGBa_F*>(this);
1209 
1210  // if the GWorld has not been allocated (typically
1211  // the first time we render a RasterImage_RGBa_F object) then we must allocate
1212  // it at the same dimensions (bounding rectangle) as the float raster
1213  if (graphWorldIsNull_())
1214  {
1215  updateableThis->allocateGraphWorld_(getHeight(), getWidth());
1216  FAIL_CONDITION( graphWorldIsNull_(),
1217  kGWorldAllocationError,
1218  "GraphWorld allocation failed in RasterImage_RGBa_F::updateRaster_");
1219  updateableThis->RasterImage_RGBa::allocateRaster2D_(getHeight(), getWidth());
1220  updateableThis->RasterImage_RGBa::allocateShiftedRaster2D_(getHeight(), getWidth(), getTop(), getLeft());
1221  }
1222 
1223  // if the bounding rectangle has its origin at (0, 0) then the shifted
1224  // raster was never allocated or it is equal to the unshifted raster
1225  if ((getTop()==0) && (getLeft()==0))
1226  {
1227  rgba = updateableThis->RasterImage_RGBa::getRGBa2D(R_W_ACCESS);
1228  fRed = red2D_;
1229  fGreen = green2D_;
1230  fBlue = blue2D_;
1231  }
1232  // otherwise I use the shifted raster
1233  else
1234  {
1235  rgba = updateableThis->RasterImage_RGBa::getRGBa2D(R_W_ACCESS);
1236  fRed = shiftedRed2D_;
1237  fGreen = shiftedGreen2D_;
1238  fBlue = shiftedBlue2D_;
1239  }
1240 
1241  // First find out which raster is the reference to update from
1242  if (isCharToFloatUpdate)
1243  {
1244  for (int i=iLow; i<=iHigh; i++)
1245  for (int j=jLow, k=4*jLow; j<=jHigh; j++)
1246  {
1247  fRed[i][j] = rgba[i][k++];
1248  fGreen[i][j] = rgba[i][k++];
1249  fBlue[i][j] = rgba[i][k++];
1250  }
1251  }
1252  else
1253  {
1254  ConversionFromFloatMode convMode;
1255  if (hasIndivFloatConvMode_)
1256  convMode = floatConv_;
1257  else
1258  convMode = defaultFloatConv_;
1259 
1260  if ((convMode==kSignedFloat) || (convMode==kPositiveFloat) || (convMode==kRangeFloat))
1261  calculateFloatConversionParams_( );
1262  long nbCols = jHigh - jLow + 1;
1263  for (int i=iLow; i<=iHigh; i++)
1264  updateableThis->convertToRGBa_(fRed[i]+jLow, fGreen[i]+jLow, fBlue[i]+jLow,
1265  rgba[i]+4*jLow, nbCols, valMin_, convScale_, convMode);
1266 
1267  }
1268 
1269  if (isCharToFloatUpdate)
1270  setGraphWorldAsReference_();
1271  else
1272  setFloatRasterAsReference_();
1273

1314  const float* const* greenSource = getShiftedGreenF2D();
1315  const float* const* blueSource = getShiftedBlueF2D();
1316  if ((getTop()==0) && (getLeft()==0))
1317  {
1318  redSource = red2D_;
1319  greenSource = green2D_;
1320  blueSource = blue2D_;
1321  }
1322  else
1323  {
1324  redSource = shiftedRed2D_;
1325  greenSource = shiftedGreen2D_;
1326  blueSource = shiftedBlue2D_;
1327  }
1328 
1329  ConversionFromFloatMode convType;
1330  if (hasIndivFloatConvMode_)
1331  convType = floatConv_;
1332  else
1333  convType = defaultFloatConv_;
1334 
1335  valMin_ = valMax_ = redSource[iLow][jLow];
1336  for (int i=iLow; i<=iHigh; i++)
1337  for (int j=jLow; j<=jHigh; j++)
1338  {

1341  else if (redSource[i][j] < valMin_)
1342  valMin_ = redSource[i][j];
1343  if (greenSource[i][j] > valMax_)
1344  valMax_ = greenSource[i][j];
1345  else if (greenSource[i][j] < valMin_)
1346  valMin_ = greenSource[i][j];
