312  ImageRect* usableRect = new ImageRect(imgIn->getValidRect());
313  usableRect->removeAllAround(1);
314  ImageRect* destRect = ImageRect::intersection(usableRect, theRect);

321 
322  const int iLow = destRect->getTop( ),
323  iHigh = destRect->getBottom( ),
324  jLow = destRect->getLeft( ),
325  jHigh = destRect->getRight( );
326  int im1, ip1, jm1, jp1;
327 
328  if (imgIn->hasFloatRaster() && imgIn->floatRasterIsReference())
329  {
330  const float* const* frIn = (static_cast<const RasterImage_gray_F*>(imgIn))->getShiftedGrayF2D();
331 
332  im1 = iLow-1;
333  ip1 = iLow+1;
334  for (int i=iLow; i<=iHigh; )

338  for (int j=jLow; j<=jHigh; )
339  {
340  if ( (frIn[im1][jm1]*frIn[ip1][jp1] < 0.f) ||
