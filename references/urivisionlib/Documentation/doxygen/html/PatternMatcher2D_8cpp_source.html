148  /* handling of scaleParam to be added here

150 
151 
152  // First determine the effective search rectangle
153  const ImageRect *imgRect = img->getValidRect();
154  ImageRect *usableRect = new ImageRect(imgRect->getLeft(), imgRect->getTop(),
155  imgRect->getWidth() - patternSought->getWidth(),

158  FAIL_CONDITION( effectiveRect == NULL,

160  "Search region does not contain valid data in PatternMatcher2D_corr::match");
161 
162  const RasterImage* patImg = patternSought->getPattern();
163  FAIL_CONDITION( patImg == NULL,

166 
167  const Blob *patBlob = patternSought->getBlob();
168 
169  ///////////

171  bool localRGBa;
172  bool localGray;

174  const RasterImage_gray* grayImg = NULL;
175  switch (patImg->getBaseRasterType())

178  // temp call
179  grayImg = RasterImage_gray::localRasterGray(img, effectiveRect, &localGray);
180  if (patBlob == NULL)

184 
185  if (localGray)

189  case kRGB24Raster:
190  break;
191 
192  case kRGBa32Raster:

201  if (localRGBa)

207  case kLongGrayRaster:
208  case kaRGB16Raster:
209  case kaRGB32Raster:

281 void PatternMatcher2D::allocateRasters_(const RasterImage* patImg, const Blob* patBlob)
282 {
283  int patWidth = patImg->getWidth(),
284  patHeight = patImg->getHeight();
285  BaseRasterType rasterType = patImg->getBaseRasterType();
286 
287  // do we need to delete old rasters?
288  // Either new type or larger dimensions
289  if ( (rasterType!=patRasterType_) || (patWidth>patWidth_) || (patHeight>patHeight_))
290  {
291  if (patRasterType_ != kUndefRasterType)
292  deleteRasters_();
293 
294  switch (rasterType)
295  {
296  case kGrayRaster:
297  grayPat_ = new float*[patHeight];
298  grayImg_ = new float*[patHeight];
299  for (int i=0; i<patHeight; i++)
300  {
301  grayPat_[i] = new float[patWidth];
302  grayImg_[i] = new float[patWidth];
303  }
304  break;
305 
306  case kRGBa32Raster:
307  rgbaPat_ = new float**[patHeight];
308  rgbaImg_ = new float**[patHeight];
309  for (int i=0; i<patHeight; i++)
310  {
311  rgbaPat_[i] = new float*[patWidth];
312  rgbaImg_[i] = new float*[patWidth];
313  for (int j=0; j<patWidth; j++)
314  {
315  rgbaPat_[i][j] = new float[3];
316  rgbaImg_[i][j] = new float[3];
317  }
318  }
319  break;
320 
321  default:
322  break;
323  }
324 
325  // If we did not have a mask and the new pattern requires one, we allocate
326  // a raster for the mask
327  if ((patMask_ == NULL) && (patBlob != NULL))
328  {
329  patMask_ = new unsigned char*[patHeight];
330  for (int i=0; i<patHeight; i++)
331  patMask_[i] = new unsigned char[patWidth];
332 
333  }
334 
335  patRasterType_ = rasterType;
336  patWidth_ = patWidth;
337  patHeight_ = patHeight;
338  }
