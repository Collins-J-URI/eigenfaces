Detailed Description

The base PolyMesh3D class.

A PolyMesh3D is a polygonal mesh defined as a set of polygonal faces.

This code is derived from the one I wrote for my personal "multiple-site" version of the SMART telerobotics project I did while at Ecole Poytechnique de Montreal. I co-wrote (with the interns I supervised) the initial OpenGL C code, then rewrote the whole thing as C++ classes and defined my own 3D format (mostly useful to define models of robots) to match these classes. There is absolutely nothing left of the original stuff, except for the file format and object loader, which were mine to begin with.

Older versions of this code have been used (and will continued to be used) as demos for my computer graphics courses. 

Version history

        - 2010/07/2x [jyh]              massive cleanup of C++ code, with 0-warning objective

391  for (i=0, iT=0, iQ=0, iP=0; i< static_cast<int>(nbFaces_); i++)
392  {

398  j2 = triangle_[iT][2] = theFace[i][2];

402  nF3 = 1 + facesForVertex_[j3][0]++;

404  facesForVertex_[j2][nF2] = iT;
405  facesForVertex_[j3][nF3] = iT;
406  iT++;
407  break;
408  //
409  case 4:
410  j1 = quad_[iQ][1] = theFace[i][1];
411  j2 = quad_[iQ][2] = theFace[i][2];
412  j3 = quad_[iQ][3] = theFace[i][3];
413  j4 = quad_[iQ][4] = theFace[i][4];
414  nF1 = 1 + facesForVertex_[j1][0]++;
415  nF2 = 1 + facesForVertex_[j2][0]++;
416  nF3 = 1 + facesForVertex_[j3][0]++;
417  nF4 = 1 + facesForVertex_[j4][0]++;
418  facesForVertex_[j1][nF1] = iQ + nbTriangles_;
419  facesForVertex_[j2][nF2] = iQ + nbTriangles_;
420  facesForVertex_[j3][nF3] = iQ + nbTriangles_;
421  facesForVertex_[j4][nF4] = iQ + nbTriangles_;
422  iQ++;
423  break;
424  //
425  default:
426  // first allocate the array of vertex indexes for that face_
427  polygon_[iP] = new GLuint[n];
428  polygon_[iP][0] = n;
429  for (k=1; k<=n; k++)
430  {
431  j = polygon_[iP][k] = theFace[i][k];
432  nF = 1 + facesForVertex_[j][0]++;
433  facesForVertex_[j][nF] = iP + nbTriangles_ + nbQuads_;
434  }
435  iP++;
436  break;
437  }
438  }
439

441  // face_ normals (whether the polyhedron is smooth or not).
442  // Triangles first
443  for (i=0; i< static_cast<int>(nbTriangles_); i++)
444  faceExactNormal(vertex_[triangle_[i][1]], vertex_[triangle_[i][2]],
445  vertex_[triangle_[i][3]], normalF_[i]);
446  // Then quads
447  for (iQ=0, i= static_cast<int>(nbTriangles_); i< static_cast<int>(nbQuads_); iQ++,i++)
448  faceAverageNormal(vertex_[quad_[iQ][1]], vertex_[quad_[iQ][2]],
449  vertex_[quad_[iQ][3]], vertex_[quad_[i][4]], normalF_[i]);
450  // Finally generic polygons
451  for (iP=0, i= static_cast<int>(nbTriangles_+nbQuads_); iP< static_cast<int>(nbPolygons_); iP++,i++)
452  faceAverageNormal(polygon_[iP], vertex_, normalF_[i]);
453  //
454  // If we want a smooth polyhedron, then we must compute the average normal
455  // for each vertex.
456  if (smoothPoly_)
457  for (j=0; j< static_cast<int>(nbVert_); j++)
458  for (k=0; k< static_cast<int>(facesForVertex_[j][0]); k++)
459  {
460  i = facesForVertex_[j][k];
461  normalV_[j][0] += normalF_[i][0];
462  normalV_[j][1] += normalF_[i][1];
463  normalV_[j][2] += normalF_[i][2];
464  normalizeVector(normalV_[j]);
465  }
466

468 }
