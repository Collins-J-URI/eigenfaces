284 void Prism3D::init_(GLfloat** topFace, GLfloat** bottomFace)
285 {
286  // Allocate the arrays of vertices and normal vectors
287  for (int i=0; i<=kVeryHighDefSlabs; i++)
288  vertex_[i] = GlUtils::glfMatrix(0, nbPtsPerSlice_-1, 0, 2);
289 
290 
291  // if we have a smooth prism, then there is one normal vector per vertex
292  if (smoothPrism_)
293  {
294  normalV_ = new GLfloat** [kVeryHighDefSlabs+1];
295  FAIL_CONDITION( normalV_==NULL,
296  kOpenGLNumAllocError,
297  "Error while allocating array of GL numbers");
298  for (int i=0; i<=kVeryHighDefSlabs; i++)
299  normalV_[i] = GlUtils::glfMatrix(0, nbPtsPerSlice_-1, 0, 2);
300  }
301  // if the prism is not smooth, then there is one normal vector per face.
302  else
303  normalF_ = GlUtils::glfMatrix(0, nbPtsPerSlice_-1, 0, 2);
304 
305  GLfloat dirStep = 1.f/kVeryHighDefSlabs,
306  d[3];
307 
308 
309  for (int j=0; j< nbPtsPerSlice_; j++)
310  {
311  d[0] = dirStep*(topFace[j][0] - bottomFace[j][0]);
312  d[1] = dirStep*(topFace[j][1] - bottomFace[j][1]);
313  d[2] = dirStep*(topFace[j][2] - bottomFace[j][2]);
314 
315  vertex_[0][j][0] = bottomFace[j][0];
316  vertex_[0][j][1] = bottomFace[j][1];
317  vertex_[0][j][2] = bottomFace[j][2];
318 
319  for (int i=1; i<=kVeryHighDefSlabs; i++)
320  {
321  vertex_[i][j][0] = vertex_[i-1][j][0] + d[0];
322  vertex_[i][j][1] = vertex_[i-1][j][1] + d[1];
323  vertex_[i][j][2] = vertex_[i-1][j][2] + d[2];
324  }
325  }
326 
327  if (smoothPrism_)
328  {
329  for (int j=0; j< nbPtsPerSlice_; j++)
330  for (int i=0; i<=kVeryHighDefSlabs; i++)
331  vertexAverageNormal(vertex_[i][j], vertex_[i][MOD(j-1, nbPtsPerSlice_)],
332  vertex_[i][MOD(j+1, nbPtsPerSlice_)], d, normalV_[i][j]);
333  }
334  else
335  {
336  for (int j=0; j< nbPtsPerSlice_; j++)
337  faceAverageNormal(vertex_[0][j], vertex_[0][MOD(j+1, nbPtsPerSlice_)],
338  vertex_[kVeryHighDefSlabs][MOD(j+1, nbPtsPerSlice_)],
339  vertex_[kVeryHighDefSlabs][j], normalF_[j]);
340  }
341 
342  // if this is a closed cylinder, then we must also compute the normal
343  // vector for the top and bottom faces.
344  if (closedPrism_)
345  {
346  faceAverageNormal(nbPtsPerSlice_, vertex_[0], normalBottom_);
347  // don't forget that the bottom points are listed clockwise.
348  normalBottom_[0] = -normalBottom_[0];
349  normalBottom_[1] = -normalBottom_[1];
350  normalBottom_[2] = -normalBottom_[2];
351 
352  faceAverageNormal(nbPtsPerSlice_, vertex_[kVeryHighDefSlabs], normalTop_);
353  }
354 }
