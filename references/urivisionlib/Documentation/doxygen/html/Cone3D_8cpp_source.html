393 | Function: void init(GLfloat* origin, GLfloat** a) |
394 | |
395 | Status: Private |
396 | |
397 | Parameters: origin coordinates of the origin (bottom' center) |
398 | a coordinates of the major axis, minor axis, and |
399 | direction axis in a 3x3 0-based array. |
400 | |
401 | Returns: Nothing |
402 | |
403 | Description: Initializes the array of vertex and normal coordinates |
404 | |
405 | Comments: This constructor implements the "pointed" cone, where the point |
406 | is at origin + axes[2] |
407 | |
408 +----------------------------------------------------------------------------------*/
409 void Cone3D::init_(GLfloat* origin, GLfloat** a)
410 {
411  GLfloat dirStep = 1.f/kVeryHighDefSlabs,
412  d[3],
413  // The summit of the cone is = origin + axes[2]
414  summit[3] = { origin[0] + a[2][0],
415  origin[1] + a[2][1],
416  origin[2] + a[2][2]},
417 
418  // X, Y, and Z coordinates of the radial and normal directions.
419  radial_j[3],
420  majCross[3], minCross[3];
421 
422  // majCross = a[0] x a[2] (major axis x direction)
423  crossProduct(a[0], a[2], majCross);
424  // minCross = a[1] x a[2] (minor axis x direction)
425  crossProduct(a[1], a[2], minCross);
426 
427  for (int j=0; j< kVeryHighDefCircPts; j++)
428  {
429  for (int k=0; k<3; k++)
430  {
431  // coordinates of the radial vector for that angle
432  radial_j[k] = ct_[j]*a[0][k] + st_[j]*a[1][k];
433 
434  // coordinates of the bottom point for that angle
435  vertex_[0][j][k] = origin[k] + radial_j[k];
436 
437  // the "top vertex" is at the summit of the cone
438  vertex_[kVeryHighDefSlabs][j][k] = summit[k];
439 
440  // coordinates of the "step displacement towards the summit.
441  d[k] = dirStep*(a[2][k] - radial_j[k]);
442 
443  // coordinates of the normal vector for that angle
444  normal_[j][k] = ct_[j]*minCross[k] - st_[j]*majCross[k];
445 
446  }
447  normalizeVector(normal_[j]);
448 
449  for (int i=1; i<kVeryHighDefSlabs; i++)
450  {
451  vertex_[i][j][0] = vertex_[i-1][j][0] + d[0];
452  vertex_[i][j][1] = vertex_[i-1][j][1] + d[1];
453  vertex_[i][j][2] = vertex_[i-1][j][2] + d[2];
454  }
455  }
456 
457  // if this is a closed cone, then we must also compute the normal
458  // vector for the bottom face.
459  // normalBottom = -major x minor / ||major x minor||
460  //
461  if (closedBottom_)
462  commonNormalVector(a[1], a[0], normalBottom_);
463 
464 }

481 | |
482 +----------------------------------------------------------------------------------*/
483 void Cone3D::init_(GLfloat* origin, GLfloat** botAxes, GLfloat* dirAxis,
484  GLfloat** topAxes)
485 {
486  GLfloat dirStep = 1.f/kVeryHighDefSlabs,
487  // X, Y, and Z coordinates of the radial and normal directions.
488  radialTop_j[3], radialBot_j[3],
489  d[3],
490  majCross[3], minCross[3];
491  GLfloat origTop[3] = { origin[0] + dirAxis[0],
492  origin[1] + dirAxis[1],
493  origin[2] + dirAxis[2]};
494 
495  for (int i=0; i<=kVeryHighDefSlabs; i++)
496  vertex_[i] = GlUtils::glfMatrix(0, kVeryHighDefCircPts-1, 0, 2);
497 
498  for (int j=0; j< kVeryHighDefCircPts; j++)
499  {
500  for (int k=0; k<3; k++)
501  {
502  radialTop_j[k] = ct_[j]*topAxes[0][k] + st_[j]*topAxes[1][k];
503  radialBot_j[k] = ct_[j]*botAxes[0][k] + st_[j]*botAxes[1][k];
504 
505 
506  vertex_[0][j][k] = origin[k] + radialBot_j[k];
507  vertex_[kVeryHighDefSlabs][j][k] = origTop[k] + radialTop_j[k];
508  d[k] = dirStep*(vertex_[kVeryHighDefSlabs][j][k] - vertex_[0][j][k]);
509  }
510 
511  // Ready to compute the coordinates of the cones "inner" vertices
512  for (int i=1; i<kVeryHighDefSlabs; i++)
513  {
514  vertex_[i][j][0] = vertex_[i-1][j][0] + d[0];
515  vertex_[i][j][1] = vertex_[i-1][j][1] + d[1];
516  vertex_[i][j][2] = vertex_[i-1][j][2] + d[2];
517  }
518 
519  // We can also compute the normal vector for each angular section of the cone
520  crossProduct(botAxes[0], d, majCross);
521  crossProduct(botAxes[1], d, minCross);
522  for (int k=0; k<3; k++)
523  normal_[j][k] = ct_[j]*minCross[k] - st_[j]*majCross[k];
524  //
525  normalizeVector(normal_[j]);
526  }
527 
528  // if this is a closed cone, then we must also compute the normal
529  // vector for the top and bottom faces.
530  if (closedTop_)
531  // normalTop = major x minor / ||major x minor||
532  commonNormalVector(topAxes[0], topAxes[1], normalTop_);
533  //
534  if (closedBottom_)
535  // normalBottom = -major x minor / ||major x minor||
536  commonNormalVector(botAxes[1], botAxes[0], normalBottom_);
537 
538 }
