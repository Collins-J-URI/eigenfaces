149  // get the bounding box of the stitched image
150  leftBound = coordX[0];
151  rightBound = coordX[0] + image[0]->getNbCols() - 1;

159  rightBound = coordX[k] + image[k]->getNbCols() - 1;
160  if (topBound > coordY[k])

163  bottomBound = coordY[k] + image[k]->getNbRows() - 1;
164  }
165

167  nbCols = rightBound - leftBound + 1;
168

170  dest = outImage->getRGB2D();
171 
172  for (short k=0; k<nbImages; k++)

175  shiftH = 3*(coordX[k] - leftBound);

396 void ManualStitcher::computeIntersections(RasterImage** image, short i)
397 {
398  short li, ri, lj, rj, // left, right, top, and botttom bounds of the 2 images
399  ti, tj, bi, bj;
400  //
401  bool condHi, condHj, condVi, condVj;
402 
403  // Compute absolute left, right, top & bottom coordinates of both images
404  li = coordX[i];
405  ti = coordY[i];
406  ri = li + image[i]->getNbCols() - 1;
407  bi = ti + image[i]->getNbRows() - 1;
408 
409  // For each image Ij other than the reference image
410  for (short j=0, l=0; j<nbImages; j++)
411  {
412  if (i != j)
413  {
414  lj = coordX[j];
415  tj = coordY[j];
416  rj = lj + image[j]->getNbCols() - 1;
417  bj = tj + image[j]->getNbRows() - 1;
418 
419  //-----------------------------------------------------------------
420  // check if images i and j (in their current position) intersect
421  //-----------------------------------------------------------------
422  condHi = ((li >= lj) && (li < rj)); // i's L between's j's L & R
423  condHj = ((lj >= li) && (lj < ri)); // j's L between's i's L & R
424  condVi = ((ti >= tj) && (ti < bj)); // i's T between's j's T & B
425  condVj = ((tj >= ti) && (tj < bi)); // j's T between's i's T & B
426 
427  if ( (condHi || condHj) && (condVi || condVj) )
428  {
429  intersect[j] = true;
430 
431  xMin[j] = MAX(li, lj);
432  xMax[j] = MIN(ri, rj);
433  yMin[j] = MAX(ti, tj);
434  yMax[j] = MIN(bi, bj);
435 
436  }
437  else
438  intersect[j] = false;
439 
440  }
441  else
442  intersect[j] = false;
443 
444  }
