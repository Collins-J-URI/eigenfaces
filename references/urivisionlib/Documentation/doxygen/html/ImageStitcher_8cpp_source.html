439  k_j = yMin[j]-coordY[j];

447  V_j = MIN(k_j, b_j);
448

450  h_j = xMin[j]-coordX[j];

453  {

455  l_j = 3*h_j;

461

463  w_i = MIN(H_i, V_i);
464  w_j = MIN(H_j, V_j);
465  //
466  if (w_i + w_j != 0)
467  {
468  s = 1.f / (w_i + w_j);
469

471  g = (unsigned char) (s *(w_i*rgb_i[k_i][l_i+1] + w_j*rgb_j[k_j][l_j+1]));

474  rgb_i[k_i][l_i] = rgb_j[k_j][l_j] = r;
475  rgb_i[k_i][l_i+1] = rgb_j[k_j][l_j+1] = g;

479  }
