642 float WarpMatchScoreFunctionND::func(const float* x)
643 {
644  int k = 0;
645  // copy data from x to grid_
646  switch(outerMode_)
647  {
648  case BORDER_FREE:
649  case BORDER_INWARD:
650  for (int i=0; i<gridRows_; i++)
651  for (int j=0; j<gridCols_; j++)
652  {
653  grid_[i][j][0] = x[k++];
654  grid_[i][j][1] = x[k++];
655  }
656  break;
657 
658  case BORDER_SLIDE:
659  // on top row only x coordinates change (slide)
660  for (int j=1; j<gridCols_-1; j++)
661  grid_[0][j][0] = x[k++];
662 
663  for (int i=1; i<gridRows_-1; i++)
664  {
665  // points on the left border can only slide in y direction
666  grid_[i][0][1] = x[k++];
667  // inner points can move freely
668  for (int j=0; j<gridCols_; j++)
669  {
670  grid_[i][j][0] = x[k++];
671  grid_[i][j][1] = x[k++];
672  }
673  // points on the right border can only slide in y direction
674  grid_[i][gridCols_-1][1] = x[k++];
675  }
676 
677  // on bottom row only x coordinates change (slide)
678  for (int j=1; j<gridCols_-1; j++)
679  grid_[gridRows_-1][j][0] = x[k++];
680 
681  break;
682 
683  case BORDER_FIXED:
684  for (int i=1; i<gridRows_-1; i++)
685  for (int j=1; j<gridCols_-1; j++)
686  {
687  grid_[i][j][0] = x[k++];
688  grid_[i][j][1] = x[k++];
689  }
690  break;
691 
692  default:
693  FAIL_CONDITION( true,
694  kInvalidParameterError,
695  "Unknown base pattern matcher type");
696  break;
697 
698  }
699 
700  warper_->warp(grid_);
701 
702  // Finally calculate the matching score
703  float score = 0.f;
704  switch(matcher_->getBaseMatcherType())
705  {
706  case PATTERN_MATCHER_2D_CORR:
707  if (patRasterType_ == kGrayRaster)
708  {
709  ((PatternMatcher2D_corr *)matcher_)->covariance((RasterImage_gray* )imgOut_,
710  matchPointX_,
711  matchPointY_,
712  imgScore1D_,
713  imgScale1D_);
714  score = -imgScore1D_[0] * imgScale1D_[0] * patScale1D_[0];
715  }
716  else
717  {
