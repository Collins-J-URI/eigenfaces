212 {
213  //----------------------------------------------------------------
214  // STEP 0: Test for null or invalid type parameters
215  //----------------------------------------------------------------
216  FAIL_CONDITION( imgIn == NULL,
217  kNullRasterImageError,
218  "NULL input raster image passed as parameter");
219 
220  FAIL_CONDITION( theRect == NULL,
221  kNullRectangleError,
222  "NULL ImageRect parameter in GaussianFilter_F::applyInto");
223 
224  FAIL_CONDITION( imgOut == NULL,
225  kNullRasterImageError,
226  "NULL output raster image passed as parameter");
227 
228  // so far I only accept a RasterImage_gray_F as a destination
229 
230  FAIL_CONDITION( (imgOut->getBaseRasterType() != kGrayRaster) ||
231  (!imgOut->hasFloatRaster()),
232  kInvalidRasterType,
233  "GaussianFilter_F::applyInto only accepts RasterImage_gray_F images as destination");
234 
235 
236  //----------------------------------------------------------------
237  // STEP 1: Test for rectangle intersection
238  //----------------------------------------------------------------
239  // determine the output rectangle that the input image can produce.
240  // Here, because we apply a filter of width&height (2m+1), we
241  // "lose m all around".
242  ImageRect* usableRect = new ImageRect(imgIn->getValidRect());
243  const int m = getWidth() / 2;
244  usableRect->removeAllAround(m);
245  // ... and determine its intersection with the user-specified output rectangle
246  ImageRect* destRect = ImageRect::intersection(usableRect, theRect);
247  delete usableRect;
248  FAIL_CONDITION( destRect == NULL,
249  kNullRectangleError,
250  "Destination rectangle does not intersect valid data rectangle");
251  // if the output image does not contain the destination rect, it must be resized.
252  if (!imgOut->contains(destRect))
253  imgOut->setBoundRect(destRect);
254 
255 
256  //----------------------------------------------------------------
257  // STEP 2: Operate only on a gray-level image
258  //----------------------------------------------------------------
259  bool localGray;
260  const RasterImage_gray* grayImg = RasterImage_gray::localRasterGray(imgIn, &localGray);
261 
262 
263  //----------------------------------------------------------------
264  // STEP 3: Perform the calculations
265  //----------------------------------------------------------------
266  // allocate a 2D raster for intermediate calculations if needed
267  bool newTempRaster = allocateTempRaster_(imgIn->getValidRect());
268 
269  switch (getWidth())
270  {
271  case 3:
272  applyInto_3x3_(grayImg, destRect, imgOut);
273  break;
274 
275  case 5:
276  applyInto_5x5_(grayImg, destRect, imgOut);
277  break;
278 
279  case 7:
280  applyInto_7x7_(grayImg, destRect, imgOut);
281  break;
282 
283  case 9:
284  applyInto_9x9_(grayImg, destRect, imgOut);
285  break;
286 
287  case 11:
288  applyInto_11x11_(grayImg, destRect, imgOut);
289  break;
290 
291  case 13:
292  applyInto_13x13_(grayImg, destRect, imgOut);
293  break;
294 
295  default:
296  applyInto_(grayImg, destRect, imgOut);
297  break;
298  }
299 
300  //----------------------------------------------------------------
301  // STEP 4: Set rectangle
302  //----------------------------------------------------------------
303  // Very important: At the end of the computation, don't forget to set the
304  // output's valid data rectangle. In "safe" mode, the library has no way to guess
305  // what the valid data rectangle should be and sets it to its minimum value... empty
306  imgOut->setValidRect(destRect);
307  (static_cast<RasterImage_gray_F*>(imgOut))->setFloatConversionMode(getFloatConversionMode());
308 
309  //----------------------------------------------------------------
310  // STEP 5: Free temporary storage (if needed)
311  //----------------------------------------------------------------
312  if (localGray)
313  delete grayImg;
314  delete destRect;
315  if (newTempRaster)
316  freeTempRaster_();
317
