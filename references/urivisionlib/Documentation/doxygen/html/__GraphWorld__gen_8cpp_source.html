39 _GraphWorld_gen::_GraphWorld_gen(int pixelDepth, const ImageRect* boundsRect)
40  try : baseType_(kUndefRasterType),
41  nbRows_(boundsRect->getHeight()),
42  nbCols_(boundsRect->getWidth()),
43  rowBytes_(0),
44  pixelDepth_(pixelDepth),
45  raster_(NULL)
46 {
47  long nbBytes;
48 
49  raster_ = NULL;
50  switch (pixelDepth)
51  {
52  // binary raster
53  case 1:
54  baseType_ = kBinaryRaster;
55  nbBytes = (nbCols_+7)/8;
56  rowBytes_ = (nbCols_+7)/8;
57  nbBytes = (1L*nbRows_)*rowBytes_;
58  raster_ = new unsigned char[nbBytes];
59  break;
60 
61  // gray-level raster
62  case 8:
63  baseType_ = kGrayRaster;
64  rowBytes_ = ((nbCols_ + 3)/4)*4;
65  nbBytes = (1L*nbRows_)*rowBytes_;
66  raster_ = new unsigned char[nbBytes];
67  break;
68 
69  // 16-bit color raster
70  case 16:
71  baseType_ = kaRGB16Raster;
72  rowBytes_ = ((2*nbCols_ + 3)/4)*4;
73  nbBytes = (1L*nbRows_)*rowBytes_;
74  raster_ = new unsigned char[nbBytes];
75  break;
76 
77  // RGB raster
78  case 24:
79  baseType_ = kRGB24Raster;
80  rowBytes_ = ((3*nbCols_ + 3)/4)*4;
81  nbBytes = (1L*nbRows_)*rowBytes_;
82  raster_ = new unsigned char[nbBytes];
83  break;
84 
85  // ARGB raster
86  case 32:
87  baseType_ = kRGBa32Raster;
88  rowBytes_ = 4*nbCols_;
89  nbBytes = (1L*nbRows_)*rowBytes_;
90  raster_ = new unsigned char[nbBytes];
91  break;
92 
93  default:
94  FAIL_CONDITION( true, kGWorldInvalidPixelDepth,
95  "_GraphWorld_gen pixel depth must be 1, 8, 16, 24, or 32");
96  break;
97 
98  }
99 
100  FAIL_CONDITION( raster_==NULL,
101  kGWorldAllocationError,
102  "GWorld allocation failed in _GraphWorld_gen constructor");
103 
104  pixelDepth_ = pixelDepth;
105 }
106 catch(...) {
107  FAIL_CONDITION( boundsRect == NULL,
108  kNullParameterError,
109  "null ImageRect pointer passed to _GraphWorld_QT constructor.");
110  FAIL_CONDITION( true,
111  kInvalidParameterError,
112  "invalid ImageRect pointer passed to _GraphWorld_QT constructor.");
