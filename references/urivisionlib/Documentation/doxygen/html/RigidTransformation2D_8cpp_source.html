95 void RigidTransformation2D::orthogonalizeMatrix_(void)
96 {
97  float *const* mat = getMatrix(R_W_ACCESS);
98 
99  FAIL_CONDITION( mat[2][2] == 0.f,
100  kMatrixCalculationError,
101  "Cannot orthogonalize a matrix when scale element is 0");
102  if (mat[2][2] != 1.f)
103  {
104  for (int i=0; i<2; i++)
105  for (int j=0; j<3; j++)
106  mat[i][j] /= mat[2][2];
107 
108  }
109  // force the bottom row to be {0, 0, 1}
110  mat[2][0] = mat[2][1] = 0.f; mat[2][2] = 1.f;
111 
112  // If the determinant is negative this is not a rigid transformation matrix
113  FAIL_CONDITION( (mat[0][0]*mat[1][1] - mat[1][0] * mat[0][1]) <= 0,
114  kMatrixCalculationError,
115  "Matrix determinant <= 0: it cannot be a rigid transformation");
116 
117  // Now I compute the 2x2 orthogonal matrix closest to the current matrix
118  // We expect the matrix to be very close to orthogonal, so basically we can
119  // look for the angle theta s.t. the upper-left part 2x2 submatrix looks like
120  // {{cos(theta), -sin(theta)}, {sin(theta), cos(theta)}}
121  const float v0 = mat[0][0] + mat[1][1];
122  const float v1 = mat[1][0] - mat[0][1];
123  const float nv = sqrtf(v0*v0+v1*v1);
124  const float cTheta = v0/nv;
125  const float sTheta = v1/nv;
126  mat[0][0] = mat[1][1] = cTheta;
127  mat[0][1] = - (mat[1][0] = sTheta);
128 
129  angle_ = atan2f(sTheta, cTheta);
130 }
