280  for (int k=0; k<gColorParams->nbModes; k++)

459  //-----------------------------------------------
460  float oldAmb[4], oldDif[4], oldSpec[4], oldShiny;
461  glGetMaterialfv(GL_FRONT, GL_AMBIENT, oldAmb);
462  glGetMaterialfv(GL_FRONT, GL_DIFFUSE, oldDif);

464  glGetMaterialfv(GL_FRONT, GL_SHININESS, &oldShiny);
465 
466  glPushMatrix();
467 
468  //-----------------------------------------------
469  // 1. Build the string to display

473  int infoLn = strlen(infoStr);
474 
475  //-----------------------------------------------

477  //-----------------------------------------------
478  int textWidth = 0;
479  for (int k=0; k<infoLn; k++)

481  // add a few pixels of padding

488  glColor3f(kTextBackgroundColor[0], kTextBackgroundColor[1], kTextBackgroundColor[2]);

490  glVertex2i(0, gImageHeight);

492  glVertex2i(gImageWidth, gImageHeight - FONT_HEIGHT - 4);
493  glVertex2i(0, gImageHeight - FONT_HEIGHT - 4);
494  glEnd();
495  glTranslatef(0.f, 0.f, 1.f);
