188  //

204  mat_ = new Matrix(3, 3);
205  double* const* matData = mat_->getMatrix(R_W_ACCESS);
206  for (int j=0, l=0; j<3; j++)

208  matData[i][j] = rotData[l++];
209 
210  }
211  else

213  mat_ = Matrix::identityMatrix(3);
214  }
215

222  quaternionIsUpToDate_ = true;

224  break;
