179 void ImageWarper_GL::getImageSuitableDimensions(int imgWidth, int imgHeight,
180  int* lowImgWidth, int* lowImgHeight,
181  int* highImgWidth, int* highImgHeight) const
182 {
183  int tempDim, lowWidth, lowHeight;
184 
185  // We find the largest power of two lower than or equal to the input width
186  tempDim = imgWidth;
187  lowWidth = 1;
188  tempDim--;
189  while (tempDim != 0)
190  {
191  tempDim /= 2;
192  lowWidth *= 2;
193  }
194 
195  // We do the same thing for the height
196  tempDim = imgHeight,
197  lowHeight = 1;
198  tempDim--;
199  while (tempDim != 0)
200  {
201  tempDim /= 2;
202  lowHeight *= 2;
203  }
204 
205  // We determine the low and high height dimensions
206  *lowImgHeight = lowHeight;
207  if (imgHeight == lowHeight)
208  *highImgHeight = lowHeight;
209  else
210  *highImgHeight = 2*lowHeight;
211 
212  // Now, depending on the platform on which we run, we will have different
213  // lower and upper width
214  //
215  // QuickTime gworlds are padded on the Mac side
216  #if URIVL_USE_QUICKTIME && URIVL_OS_MACOS
217  *lowImgWidth = lowWidth - 4;
218  *highImgWidth = 2*lowWidth - 4;
219  #else
220  *lowImgWidth = lowWidth;
221  if (imgHeight == lowWidth)
222  *highImgWidth = lowWidth;
223  else
224  *highImgWidth = 2*lowWidth;
225  #endif
226 }

236 void ImageWarper_GL::warp_(float*** xy)
237 {
238  glMatrixMode(GL_MODELVIEW);
239  glLoadIdentity();
240  glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT); // Clear Screen And Depth Buffer
241 
242  glEnable(GL_TEXTURE_2D);
243  glBindTexture(GL_TEXTURE_2D, textureID_);
244 
245  glColor4f(1.f, 1.f, 1.f, 1.f);
246 
247  // map the texture to each square
248  for (int i=0; i<gridRows_ - 1; i++)
249  {
250  for (int j=0; j<gridCols_ - 1; j++)
251  {
252  glBegin(GL_POLYGON);
253  glTexCoord2fv(texturePoint_[i][j]);
254  glVertex2fv(xy[i][j]);
255  glTexCoord2fv(texturePoint_[i+1][j]);
256  glVertex2fv(xy[i+1][j]);
257  glTexCoord2fv(texturePoint_[i+1][j+1]);
258  glVertex2fv(xy[i+1][j+1]);
259  glTexCoord2fv(texturePoint_[i][j+1]);
260  glVertex2fv(xy[i][j+1]);
261  glEnd();
262  }
263  }
264 
265  // read from front buffer
266  glReadBuffer(GL_BACK);
267 
268  switch (rasterType_)
269  {
270  case kGrayRaster:
271  // read pixel data
272  glReadPixels(0, 0, imgHeight_, imgHeight_, GL_LUMINANCE, GL_UNSIGNED_BYTE, imgOut1D_);
273  break;
274 
275  case kRGBa32Raster:
276  // read pixel data
277  glReadPixels(0, 0, imgHeight_, imgHeight_, GL_RGBA, GL_UNSIGNED_BYTE, imgOut1D_);
278  break;
279 
280  default:
281  break;
282  }
283 
284  glDisable(GL_TEXTURE_2D);
