235 void DepthMap::updateMesh(void)
236 {
237  // If the depth raster is the reference we recalculate the mesh
238  DrawableInformationType dType = getDrawableInformationType();
239  if ((dType == kDense) || (dType == kDenseUncertain))
240  {
241  const ImageRect* vRect = getValidRect();
242  const int iLow = vRect->getTop(),
243  iHigh = vRect->getBottom(),
244  jLow = vRect->getLeft(),
245  jHigh = vRect->getRight();
246  const float *const* depth = getShiftedGrayF2D();
247  const double fx = cameraParams_->getFocalLengthX();
248  const double fy = cameraParams_->getFocalLengthY();
249  double x, y;
250  float Z;
251 
252  // First pass: determine the origin of the depth map.
253  // We place it a the middle of the map's "bottom"
254  float bottomZ = 0.f;
255  for (int i=iLow; i<=iHigh; i++)
256  for (int j=jLow; j<=jHigh; j++)
257  {
258  // If there is a distortion function, it is applied here
259  cameraParams_->pixelToMetric(j, i, &x, &y);
260 
261  Z = -scaleZ_ * depth[i][j];
262  if (Z < bottomZ)
263  bottomZ = Z;
264 
265  triMesh_[i][j][0] = static_cast<float>(-x * Z / fx);
266  triMesh_[i][j][1] = static_cast<float>(y * Z / fy);
267  triMesh_[i][j][2] = static_cast<float>(Z);
268  }
269 
270  // We can now produce the origin's coordinates
271  // For this we need the inverse projection of the image's center
272  cameraParams_->pixelToMetric((jLow + jHigh)/2, (iLow + iHigh)/2, &x, &y);
273  float origX = static_cast<float>(-x * bottomZ / fx),
274  origY = static_cast<float>(-y * bottomZ / fy);
275  if (origin_ != NULL)
276  delete origin_;
277  origin_ = new Point3D(origX, origY, bottomZ);
278 
279  // Second pass, we transform the mesh's points' cooordinates to be
280  // relative to the mesh's origin
281  for (int i=iLow; i<=iHigh; i++)
282  for (int j=jLow; j<=jHigh; j++)
283  {
284  triMesh_[i][j][0] -= origX;
285  triMesh_[i][j][1] -= origY;
286  triMesh_[i][j][2] -= bottomZ;
287  }
288 
289  }
290  // else I don't know yet what to do
291  else
292  {
293  FAIL_CONDITION( true,
294  kFunctionNotImplemented,
295  "Not implemented yet");
296  }
297 }
