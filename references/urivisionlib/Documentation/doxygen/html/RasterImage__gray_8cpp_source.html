461 void RasterImage_gray::writeInto_(RasterImage* imgOut, const ImageRect* copyRect,
462  const ImagePoint* origin) const
463 {
464  const unsigned char* const* graySource;
465 
466  // The GWorld only works with unshifted rasters so we need to compute
467  // the "true" copy rectangles first
468  ImageRect* sourceRect = new ImageRect(copyRect->getLeft() - getLeft(),
469  copyRect->getTop() - getTop(),
470  copyRect->getWidth(), copyRect->getHeight());
471  ImageRect* destRect = origin == NULL ?
472  new ImageRect(copyRect->getLeft() - imgOut->getLeft(),
473  copyRect->getTop() - imgOut->getTop(),
474  copyRect->getWidth(), copyRect->getHeight())
475  :
476  new ImageRect(origin->getX() - imgOut->getLeft(),
477  origin->getY() - imgOut->getTop(),
478  copyRect->getWidth(), copyRect->getHeight());
479 
480  const int iLowSource = sourceRect->getTop(),
481  iLowDest = destRect->getTop(),
482  iHighSource = sourceRect->getBottom(),
483  jLowSource = sourceRect->getLeft(),
484  jHighSource = sourceRect->getRight(),
485  jLowDest = destRect->getLeft();
486  const int nbRows = sourceRect->getHeight(),
487  nbCols = sourceRect->getWidth();
488 
489  // Subclasses should override this function, but in case they don't, we need to
490  // do the conversion through the base gray-level raster.
491  if (!gworldIsReference_ && referenceHasChanged_)
492  updateRaster_( );
493 
494  switch( imgOut->getBaseRasterType() )
495  {
496  case kBinaryRaster:
497  // At this point we are ready to perform the copy/conversion and we use
498  // the CopyBits function of QuickDraw/QuickTime
499  gworld_->copyInto(imgOut->gworld_, sourceRect, destRect, false);
500 
501  imgOut->gworldIsReference_ = true;
502  imgOut->referenceHasChanged_ = true;
503  break;
504 
505  case kGrayRaster:
506  // if we are dealing with a plain gray raster image, just copy the bytes
507  if (!imgOut->hasFloatRaster())
508  {
509  // At this point we are ready to perform the copy/conversion and we use
510  // the CopyBits function of QuickDraw/QuickTime
511  gworld_->copyInto(imgOut->gworld_, sourceRect, destRect, false);
512 
513  imgOut->gworldIsReference_ = true;
514  imgOut->referenceHasChanged_ = true;
515  }
516  //
517  // if imgOut has a float raster, we had better use it
518  else
519  {
520  graySource = getShiftedRaster2D();
521  float* const* fGrayDest = (static_cast<RasterImage_gray_F*>(imgOut))->getShiftedGrayF2D(R_W_ACCESS);
522 
523  // This call may have been made from within the RasterImage_gray_F constructor of imgOut,
524  // in which case grayDest has not been allocated yet
525  if (fGrayDest == NULL)
526  {
527  imgOut->allocateRaster2D_(nbRows, nbCols);
528  imgOut->allocateShiftedRaster2D_(nbRows, nbCols, iLowDest, jLowDest);
529  fGrayDest = (static_cast<RasterImage_gray_F*>(imgOut))->getShiftedGrayF2D(R_W_ACCESS);
530  }
531 
532  for (int i=iLowSource, iDest=jLowDest; i<=iHighSource; i++, iDest++)
533  for (int j=jLowSource, jDest=jLowDest; j<=jHighSource; j++, jDest++)
534  fGrayDest[iDest][jDest] = graySource[i][j];
535 
536  imgOut->gworldIsReference_ = false;
537  imgOut->referenceHasChanged_ = true;
538  (static_cast<RasterImage_gray_F*>(imgOut))->setFloatConversionMode(kSaturatedPositive);
539  }
540 
541  break;
542 
543  case kRGB24Raster:
544  // At this point we are ready to perform the copy/conversion and we use
545  // the CopyBits function of QuickDraw/QuickTime
546  gworld_->copyInto(imgOut->gworld_, sourceRect, destRect, true);
547 
548  imgOut->gworldIsReference_ = true;
549  imgOut->referenceHasChanged_ = true;
550  break;
551 
552  case kaRGB16Raster:
553  // At this point we are ready to perform the copy/conversion and we use

560

564  {
565  // At this point we are ready to perform the copy/conversion and we use

568 
569  imgOut->gworldIsReference_ = true;

592 
593  for (int i=iLowSource, iDest=jLowDest; i<=iHighSource; i++, iDest++)
594  for (int j=jLowSource, jDest=jLowDest; j<=jHighSource; j++, jDest++)
595  {
596  fRedOut[iDest][jDest] = graySource[i][j];

613  }

631 void RasterImage_gray::convertToBinary(const unsigned char* theSource,
632  unsigned char* theDestination,
633  long nbPixels)
634 {
635  const long nbBytes = (nbPixels+7)/8;
636  const int remainder = static_cast<int>(nbPixels % 8);
637  const unsigned char* source = theSource;
638  unsigned char* dest = theDestination;
639  const unsigned char BIN_0 = static_cast<unsigned char>(0),
640  BIN_HALF = static_cast<unsigned char>(127);
641  unsigned char d;
642 
643  // do the conversion for the full bytes
644  for (long l=0; l<nbBytes-1; l++)
645  {
646  d = BIN_0;
647 
648  for (int k=7; k>=0; k--)
649  {
650  if (*(source++) > BIN_HALF)
651  d |= (1 << k);
652  }
653  *(dest++) = d;
654 
655  }
656 
657  // do the conversion for the remainder bits (incomplete byte)
658  d = BIN_0;
659  if (remainder > 0)
660  {
661  for (int k=7; k>7-remainder; k--)
662  {
663  if (*(source++) > BIN_HALF)
664  d |= (1 << k);
665  }
666  *(dest++) = d;
667  }
668 }

909 void RasterImage_gray::setBoundRect_(int x, int y, int theWidth, int theHeight)
910 {
911  FAIL_CONDITION( propertiesAreLocked_,
912  kAccessToLockedData,
913  "Attempt to resize a RasterImage with locked properties");
914 
915  // if width and height have not changed, then the operation is just a translation.
916  // The ImageRect member has been updated in DrawableObject2D::setBoundRect, so we only need
917  // to update the "shifted" array (no re-allocation needed)
918  if ((theWidth == getWidth()) && (theHeight == getHeight()))
919  {
920  if ((x != getLeft()) || (y != getTop()))
921  {
922  // Since the rasters have the same dimensions, there is no need
923  // to delete and reallocate the shifted rasters. Just apply a
924  // relative shift to them
925  const int iLow = getTop(),
926  iHigh = getBottom(),
927  jLow = getLeft(),
928  rowShift = y - iLow,
929  colShift = x - jLow;
930 
931  if (shiftedGray2D_)
932  {
933  for (int i=iLow; i<=iHigh; i++)
934  shiftedGray2D_[i] += colShift;
935  shiftedGray2D_ += rowShift;
936  }
937  }
938  // if nothing has changed, there is nothing to do
939  }
940  // if the dimensions of the rectangles are different, then we must delete
941  // and reallocate the GraphWorld as well as the 0-based 2D raster.
942 
943  else
944  {
945  // I am not sure if this is the right thing to do: should subclasses completely
946  // reimplement this or call this function? This is opting for the latter.
947  if (gworld_ != NULL)
948  {
949  delete gworld_;
950  deleteShiftedRaster2D_(y);
951  deleteRaster2D_();
952 
953  gworld_ = new GraphWorld(1, theHeight, theWidth);
954 
955 // Now done in DrawableObject2D
956 // setBoundRect(x, y, theWidth, theHeight);
957 // setValidRect(x, y, theWidth, theHeight);
958 
959  allocateRaster2D_(theHeight, theWidth);
960  allocateShiftedRaster2D_(theHeight, theWidth, y, x);
961  }
962  }
963 }
