398 double Registration2D_ICP::llsSolutionTransformation(int nbPts, const double* ptX, const double* ptY,
399  const bool *match, const double* refPtX,
400  const double* refPtY, Transformation2D *transf)
401 {
402  //------------------------------------------------------------------------
403  // Do the basic summations of x & y coordinates of the reference and data
404  // points, dot products, and determinants
405  //------------------------------------------------------------------------
406  int matchCount = 0;
407  double sumPtX = 0, sumPtY = 0,
408  sumRefX = 0, sumRefY = 0,
409  sumDet = 0, sumDotProd = 0;
410  for (int i=0; i<nbPts; i++)
411  {
412  if (match[i])
413  {
414  sumPtX += ptX[i];
415  sumPtY += ptY[i];
416  sumRefX += refPtX[i];
417  sumRefY += refPtY[i];
418  sumDet += ptX[i]*refPtY[i] - ptY[i]*refPtX[i];
419  sumDotProd += ptX[i]*refPtX[i] + ptY[i]*refPtY[i];
420  //
421  matchCount++;
422  }
423  }
424 
425  //------------------------------------------------------------------------
426  // Compute the angle and translation of the rigid transformation
427  //------------------------------------------------------------------------
428  const double meanPtX = sumPtX / matchCount,
429  meanPtY = sumPtY / matchCount,
430  meanRefX = sumRefX / matchCount,
431  meanRefY = sumRefY / matchCount,
432  meanDet = sumDet / matchCount,
433  meanDotProd = sumDotProd / matchCount;
434  const double dotProdMean = meanPtX*meanRefX + meanPtY*meanRefY,
435  detMean = meanPtX*meanRefY - meanPtY*meanRefX;
436 
437  const double A = detMean - meanDet,
438  B = dotProdMean - meanDotProd,
439  normAB = sqrt(A*A + B*B);
440 
441  const double cosTheta = -B / normAB,
442  sinTheta = A / normAB;
443 
444  const double tx = meanPtX - cosTheta*meanRefX + sinTheta*meanRefY,
445  ty = meanPtY - sinTheta*meanRefX - cosTheta*meanRefY;
446 
447  //------------------------------------------------------------------------
448  // Update the transformation's matrix
449  //------------------------------------------------------------------------
450  float* const* const mat = transf->getMatrix(R_W_ACCESS);
451  mat[0][0] = mat[1][1] = static_cast<float>(cosTheta);
452  mat[0][1] = - (mat[1][0] = static_cast<float>(sinTheta));
453  //
454  mat[0][2] = static_cast<float>(tx);
455  mat[1][2] = static_cast<float>(ty);
456  //
457  mat[2][0] = mat[2][1] = 0.f;
458  mat[2][2] = 1.f;
459 
460  //------------------------------------------------------------------------
461  // And finally compute the squared norm of the residual
462  //------------------------------------------------------------------------
463  double normSqRes = 0.L;
464  for (int i=0; i<nbPts; i++)
465  {
466  if (match[i])
467  {
468  const double dx = ptX[i] - (cosTheta*refPtX[i] - sinTheta*refPtY[i] + tx),
469  dy = ptY[i] - (sinTheta*refPtX[i] + cosTheta*refPtY[i] + ty);
470 
471  normSqRes += dx*dx + dy*dy;
472  }
473  }
474 
475  return normSqRes;
476 }
