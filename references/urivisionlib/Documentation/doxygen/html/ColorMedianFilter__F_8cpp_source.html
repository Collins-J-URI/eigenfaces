162 {
163  //----------------------------------------------------------------
164  // STEP 0: Test for null or invalid type parameters
165  //----------------------------------------------------------------
166  FAIL_CONDITION( imgIn == NULL,
167  kNullRasterImageError,
168  "NULL input raster image passed as parameter");
169 
170  FAIL_CONDITION( theRect == NULL,
171  kNullRectangleError,
172  "NULL ImageRect parameter in VectorField_F::copyInto");
173 
174  FAIL_CONDITION( imgOut == NULL,
175  kNullRasterImageError,
176  "NULL output raster image passed as parameter");
177 
178  // so far I only accept a RasterImage_RGBa_F as a destination
179  FAIL_CONDITION( imgOut->getBaseRasterType() != kRGBa32Raster ||
180  (!imgOut->hasFloatRaster()),
181  kInvalidRasterType,
182  "GaussianFilter_F::applyInto only accepts RasterImage_RGBa_F images as destination");
183 
184  //----------------------------------------------------------------
185  // STEP 1: Test for rectangle intersection
186  //----------------------------------------------------------------
187  // determine the output rectangle that the input image can produce.
188  // Here, it is the entire valid data rectangle.
189  const ImageRect *outRect = imgIn->getValidRect();
190  // ... and determine its intersection with the user-specified output rectangle
191  ImageRect *destRect = ImageRect::intersection(theRect, outRect);
192  FAIL_CONDITION( destRect == NULL,
193  kNullRectangleError,
194  "Destination rectangle does not intersect valid data rectangle");
195  // if the output image does not contain the destination rect, it must be resized.
196  if (!outRect->contains(destRect))
197  imgOut->setBoundRect(destRect);
198 
199  //----------------------------------------------------------------
200  // STEP 2: Get an RGBa input image
201  //----------------------------------------------------------------
202  bool localRGBa = false;
203  const RasterImage_RGBa* rgbaImg = RasterImage_RGBa::localRasterRGBa(imgIn, &localRGBa);
204 
205  //----------------------------------------------------------------
206  // STEP 3: Perform the calculations
207  //----------------------------------------------------------------
208  const int iLow = destRect->getTop(),
209  iHigh = destRect->getBottom(),
210  jLow = destRect->getLeft(),
211  jHigh = destRect->getRight();
212  const ImageRect* operRect = getRect();
213  const int recTop = operRect->getTop(),
214  recBottom = operRect->getBottom(),
215  recLeft = operRect->getLeft(),
216  recRight = operRect->getRight();
217 
218  if (rgbaImg->hasFloatRaster() && rgbaImg->floatRasterIsReference())
219  {
220  const float* const* redIn = (static_cast<const RasterImage_RGBa_F*>(rgbaImg))->getShiftedRedF2D();
221  const float* const* greenIn = (static_cast<const RasterImage_RGBa_F*>(rgbaImg))->getShiftedGreenF2D();
222  const float* const* blueIn = (static_cast<const RasterImage_RGBa_F*>(rgbaImg))->getShiftedBlueF2D();
223  //
224  float* const* redOut = (static_cast<RasterImage_RGBa_F*>(imgOut))->getShiftedRedF2D(R_W_ACCESS);
225  float* const* greenOut = (static_cast<RasterImage_RGBa_F*>(imgOut))->getShiftedGreenF2D(R_W_ACCESS);
226  float* const* blueOut = (static_cast<RasterImage_RGBa_F*>(imgOut))->getShiftedBlueF2D(R_W_ACCESS);
227  //
228  for (int i=iLow; i<= iHigh; i++)
229  {
230  const int kLow = MAX(i+recTop, iLow);
231  const int kHigh = MIN(i+recBottom, iHigh);
232  //
233  for (int j=jLow; j<=jHigh; j++)
234  {
235  // collect the data around the pixel.
236  const int lLow = MAX(j+recLeft, jLow);
237  const int lHigh = MIN(j+recRight, jHigh);
238  int n = 0;
239  //
240  for (int k=kLow; k<=kHigh; k++)
241  for (int l=lLow; l<=lHigh; l++)
242  {
243  redData_[n] = redIn[k][l];
244  greenData_[n] = greenIn[k][l];
245  blueData_[n++] = blueIn[k][l];
246  }
247 
248  redOut[i][j] = getColorMedian_(redData_, n);
249  greenOut[i][j] = getColorMedian_(greenData_, n);
250  blueOut[i][j] = getColorMedian_(blueData_, n);
251  }
252  }
253  }
254  else
255  {
256  const unsigned char* const* rgbaIn = rgbaImg->getShiftedRGBa2D();
257  //
258  float* const* redOut = (static_cast<RasterImage_RGBa_F*>(imgOut))->getShiftedRedF2D(R_W_ACCESS);
259  float* const* greenOut = (static_cast<RasterImage_RGBa_F*>(imgOut))->getShiftedGreenF2D(R_W_ACCESS);
260  float* const* blueOut = (static_cast<RasterImage_RGBa_F*>(imgOut))->getShiftedBlueF2D(R_W_ACCESS);
261  //
262  for (int i=iLow; i<= iHigh; i++)
263  {
264  const int kLow = MAX(i+recTop, iLow);
265  const int kHigh = MIN(i+recBottom, iHigh);
266 // int fourj = 4*jLow; // unused var. -CJC
267  //
268  for (int j=jLow; j<=jHigh; j++)
269  {
270  // collect the data around the pixel.
271  const int lLow = MAX(j+recLeft, jLow);
272  const int lHigh = MIN(j+recRight, jHigh);
273  const int fourlMin = 4*lLow;
274  int n = 0;
275  //
276  for (int k=kLow; k<=kHigh; k++)
277  {
278  int fourl = fourlMin;
279  for (int l=lLow; l<=lHigh; l++)
280  {
281  redData_[n] = rgbaIn[k][fourl++];
282  greenData_[n] = rgbaIn[k][fourl++];
283  blueData_[n++] = rgbaIn[k][fourl++];
284  fourl++;
285  }
286  }
287 
288  redOut[i][j] = getColorMedian_(redData_, n);
289  greenOut[i][j] = getColorMedian_(greenData_, n);
290  blueOut[i][j] = getColorMedian_(blueData_, n);
291  }
292  }
293  }

296  // output's valid data rectangle. In "safe" mode, the library has no way to guess
297  // what the valid data rectangle should be and sets it to its minimum value... empty
298  imgOut->setValidRect(outRect);
299  imgOut->requestRasterUpdate();

301  // cleanup
302  if (localRGBa)

312 // Computation of the median using N. Wirth's algorithm
313 // Implementation derived from kth_smallest implementation of N Devillard.
314 // Note that if the array is even-sized, the function does not return the average of
315 // the two central values but the lower median.
316 float getColorMedian_(float* dat, int n)
317 {
318  const int k = (n%2) == 0 ? n/2 - 1 : n/2;
319  float x;
320  int i, j,
321  l = 0,
322  m = n - 1;
323 
324  while (l<m)
325  {
326  x = dat[k];
327  i = l;
328  j = m;
329  do
330  {
331  while (dat[i] < x)
332  i++;
333  while (dat[j] > x)
334  j--;
335 
336  if (i <= j)
337  {
338  float temp = dat[i];
339  dat[i] = dat[j];
340  dat[j] = temp;
341  i++;
342  j--;
343  }
344 
345  } while (i <= j);
346 
347  if (j < k)
348  l = i;
349  if (i > k)
350  m = j;
351  }
352 
353  return dat[k];
354 }
