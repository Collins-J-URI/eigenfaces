26  /** Possible configurations for two cameras that can lead to a stereo
27  * disparity. Generally in stereo systems one camera is the "left camera" and
28  * the other is the "right camera." In that case I may want to compute the disparity
29  * "left to right or right to left." But the cameras could also be disposed in a
30  * vertical configuration with a "top camera" and a "bottom camera," in which case
31  * the disparity could be computed "top to bottom" or "botom to top." Finally,
32  * for those other "Generic" configurations the disparity can be computed
33  " from image 1 to image 2 or from image 2 to image 1.
34  */

36  HORIZONTAL_L2R = 0,
37  HORIZONTAL_R2L,
38  VERTICAL_T2B,
39  VERTICAL_B2T,
40  // In the following 2 configs, one camera is artitrarily
41  // identified as "left camera" and the other as "right camera"
42  GENERIC_1_2,
43  GENERIC_2_1
44  } StereoDisparityType;
