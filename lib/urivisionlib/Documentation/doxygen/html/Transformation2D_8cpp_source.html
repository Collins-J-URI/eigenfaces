183 bool Transformation2D::checkOrthogonality_(void) const
184 {
185  // last row must be 0 0 1
186  if ((mat_[2][0] != 0.f) || (mat_[2][1] != 0.f) || (mat_[2][2] != 1.f))
187  return false;
188 
189  const float SMALL_VAL = (float) 1.0E-5;
190 
191  const float dotProd01 = mat_[0][0]*mat_[0][1] + mat_[1][0]*mat_[1][1];
192  if (fabsf(dotProd01) > SMALL_VAL)
193  return false;
194 
195  const float norm0 = mat_[0][0]*mat_[0][0] + mat_[1][0]*mat_[1][0];
196  if (fabsf(norm0 - 1.f) > SMALL_VAL)
197  return false;
198 
199  const float norm1 = mat_[0][1]*mat_[0][1] + mat_[1][1]*mat_[1][1];
200  if (fabsf(norm1 - 1.f) > SMALL_VAL)
201  return false;
202 
203  return true;
204 }
