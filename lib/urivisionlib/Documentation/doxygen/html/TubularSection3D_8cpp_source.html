169 
170  GLfloat** axes = GlUtils::glfMatrix(0, 2, 0, 2);
171 
172  try
173  {
174  // dirAxis is the direction axis of the cylinder
175  axes[2][0] = dirAxis[0];
176  axes[2][1] = dirAxis[1];
177  axes[2][2] = dirAxis[2];
178  //
179  // For the major axis, we need to build a vector orthogonal to dirAxis.
180  if ((dirAxis[0]!= 0.f) || (dirAxis[2]!= 0.f)) // this is not the local Y axis
181  {
182  GLfloat w[3] = {0.f, 1.f, 0.f};
183  commonNormalVector(w, dirAxis, axes[0]); // "X" = "Y" x "Z"
184  commonNormalVector(dirAxis, axes[0], axes[1]); // "Y" = "Z" x "X"
185  //
186  // and now we can multiply by the radius
187  for (int i=0; i<3; i++)
188  {
189  axes[0][i] *= radius;
190  axes[1][i] *= radius;
191  }
192  }
193  // if dirAxis was the local Y axis, then I can take Z and X as major and minor
194  else
195  {
196  axes[0][0] = axes[0][1] = axes[1][1] = axes[1][2] = 0.f;
197  axes[0][2] = axes[1][0] = radius;
198  }
199  }
200  catch (...)
201  {
202  deleteArrays_();
203  GlUtils::free_glfMatrix(axes, 0, 2, 0, 2);
204  FAIL_CONDITION( true,
205  kInvalidParameterError,
206  "Invalid array passed to TubularSection3D constructor");
207  }
208

400 | Description: Initializes the array of vertex and normal coordinates |
401 | |
402 +----------------------------------------------------------------------------------*/
403 void TubularSection3D::init_(GLfloat* origin, GLfloat** a, GLfloat theta0, GLfloat dTheta)
404 {
405  GLfloat dirStep = 1.f/kVeryHighDefSlabs,
406  dir[kVeryHighDefSlabs+1][3],
407  // X, Y, and Z coordinates of the radial and normal directions.
408  radial_j[3],
409  majCross[3], minCross[3];
410 
411  // Allocate the arrays of vertices and normal vectors
412  vertex_ = new GLfloat** [kVeryHighDefSlabs+1];
413  normal_ = GlUtils::glfMatrix(0, nbCircSegs_, 0, 2);
414 
415  for (int i=0; i<=kVeryHighDefSlabs; i++)
416  vertex_[i] = GlUtils::glfMatrix(0, nbCircSegs_, 0, 2);
417 
418  // majCross = a[0] x a[2] (major axis x direction)
419  crossProduct(a[0], a[2], majCross);
420  // minCross = a[1] x a[2] (minor axis x direction)
421  crossProduct(a[1], a[2], minCross);
422 
423  // First, we precompute the graduation steps for the slabs along
424  // the "direction" axis (a[2]).
425  for (int i=0; i<=kVeryHighDefSlabs; i++)
426  for (int k=0; k<3; k++)
427  dir[i][k] = i*dirStep*a[2][k];
428 
429  GLfloat theta_j = theta0;
430  for (int j=0; j<= nbCircSegs_; j++)
431  {
432  GLfloat cos_tj = cosf(theta_j);
433  GLfloat sin_tj = sinf(theta_j);
434 
435  radial_j[0] = cos_tj*a[0][0] + sin_tj*a[1][0];
436  normal_[j][0] = cos_tj*minCross[0] - sin_tj*majCross[0];
437  //
438  radial_j[1] = cos_tj*a[0][1] + sin_tj*a[1][1];
439  normal_[j][1] = cos_tj*minCross[1] - sin_tj*majCross[1];
440  //
441  radial_j[2] = cos_tj*a[0][2] + sin_tj*a[1][2];
442  normal_[j][2] = cos_tj*minCross[2] - sin_tj*majCross[2];
443 
444  normalizeVector(normal_[j]);
445 
446  for (int i=0; i<=kVeryHighDefSlabs; i++)
447  {
448  // M = origin + cos[u] major + sin[u] minor + v direction
449  vertex_[i][j][0] = origin[0] + radial_j[0] + dir[i][0];
450  vertex_[i][j][1] = origin[1] + radial_j[1] + dir[i][1];
451  vertex_[i][j][2] = origin[2] + radial_j[2] + dir[i][2];
452  }
453 
454  theta_j += dTheta;
455  }
456 }
