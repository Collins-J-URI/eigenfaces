41 _GraphWorld_QT::_GraphWorld_QT(int pixelDepth, const ImageRect* theBoundsRect)
42  try :
43  gworld_(NULL),
44  nbRows_(theBoundsRect->getHeight()),
45  pixelDepth_(pixelDepth),
46  baseType_(kUndefRasterType),
47  pixmap_(NULL),
48  device_(NULL),
49  currentGWorld_(NULL),
50  previousGWorld_(NULL),

52  previousGDevice_(NULL)
53 {
54  CTabHandle clut;
55  QDErr myError;
56  Rect myRect = theBoundsRect->rect_;
57 
58  // The _GraphWorld_QT's rectangle always has its origin at (0, 0)
59  MacOffsetRect(&myRect, static_cast<short>(-myRect.left), static_cast<short>(-myRect.top));
60 
61  switch (pixelDepth)
62  {
63  // binary raster
64  case 1:
65  baseType_ = kBinaryRaster;
66  clut = (ColorTable **) NULL;
67  break;
68 
69  // gray-level raster
70  case 8:
71  baseType_ = kGrayRaster;
72  clut = GetCTable (40);
73  break;
74 
75  // 16-bit color raster
76  case 16:
77  baseType_ = kaRGB16Raster;
78  clut = (ColorTable **) NULL;
79  break;
80 
81  // RGB raster
82  case 24:
83  baseType_ = kRGB24Raster;
84  clut = (ColorTable **) NULL;
85  break;
86 
87  // ARGB raster
88  case 32:
89  baseType_ = kRGBa32Raster;
90  clut = (ColorTable **) NULL;
91  myRect.bottom++; // extra row for ARGB --> RGBA shift
92  break;
93 
94  default:
95  FAIL_CONDITION( true, kGWorldInvalidPixelDepth,
96  "_GraphWorld_QT pixel depth must be 8, 16, 24, or 32");
97  break;
98 
99  }
100 
101  // Note that the boundRect_ member of an ImageRect object uses the pixel-as-rectangle
102  // convention, which is the one that the GWorld uses too.
103  gworld_ = NULL;
104  myError = NewGWorld(&gworld_,
105  (short) pixelDepth,
106  &myRect,
107  clut,
108  (GDevice **) NULL,
109  0L);
110  FAIL_CONDITION ((myError!= noErr) || (gworld_==NULL), kGWorldAllocationError,
111  "GWorld allocation failed in _GraphWorld_QT constructor");
112 
113  pixmap_ = GetGWorldPixMap(gworld_);
114  FAIL_CONDITION (pixmap_==NULL, kGWorldNullPixMapError,
115  "GWorld has a NULL PixMap in _GraphWorld_QT constructor");
116 
117  // lock the pixmap
118  lock_();
119 }
120 catch(...) {
121  FAIL_CONDITION( theBoundsRect == NULL,
122  kNullParameterError,
123  "null ImageRect pointer passed to _GraphWorld_QT constructor.");
124  FAIL_CONDITION( true,
125  kInvalidParameterError,
126  "invalid ImageRect pointer passed to _GraphWorld_QT constructor.");

359 // Does a CopyBits to copy one _GraphWorld_QT into another, possibly modifying dimensions
360 // and pixel depth on the way
361 void _GraphWorld_QT::copyInto_(_GraphWorld_QT* destination, const ImageRect* sourceRect,
362  const ImageRect* destRect, bool inverseGray)
363 {
364  // First, store the current graphic port and device
365  CGrafPtr origPort;
366  GDHandle origDev;
367  GetGWorld(&origPort, &origDev);
368 
369  // Extract the PixMap (handle) of the source and destination GWorlds
370  PixMapHandle sourceMap = GetGWorldPixMap(this->gworld_),
371  destMap = GetGWorldPixMap(destination->gworld_);
372 
373  // Lock the PixMap of the source GWorld
374  FAIL_CONDITION( !(LockPixels(sourceMap)),
375  kPixMapLockFailed,
376  "failed to lock source PixMap prior to CopyBits");
377 
378  // Prepare the copy
379  ForeColor(blackColor);
380  BackColor(whiteColor);
381 // SetGWorld(gworld_, NULL);
382 //jyh May 08 2008
383  SetGWorld(destination->gworld_, NULL);
384 
385  if (inverseGray)
386  CopyBits((BitMap *) *sourceMap, (BitMap *) *destMap,
387  sourceRect->getRectQT_(), destRect->getRectQT_(), notSrcCopy, NULL);
388  else
389  CopyBits((BitMap *) *sourceMap, (BitMap *) *destMap,
390  sourceRect->getRectQT_(), destRect->getRectQT_(), srcCopy, NULL);
391 
392  // Unlock the PixMap and restore initial port and device
393  UnlockPixels(sourceMap);
394  SetGWorld(origPort, origDev);
395 }

410 // OpenGL must use the effective number of columns (i.e., including the row padding
411 // of the GWorld) for its rendering of the raster image.
412 void _GraphWorld_QT::draw_(void)
413 {
414  RGBColor blackColour = { 0x0000, 0x0000, 0x0000 };
415  RGBColor whiteColour = { 0xFFFF, 0xFFFF, 0xFFFF };
416 
417  setGraphWorld_();
418  RGBBackColor(&whiteColour);
419  RGBForeColor(&blackColour);
420 
421  switch (baseType_)
422  {
423  case kBinaryRaster:
424  glDrawPixels( getEffectiveNbCols_(), nbRows_,
425  GL_LUMINANCE,
426  GL_UNSIGNED_BYTE,
427  getRaster_());
428  break;
429 
430  case kGrayRaster:
431  glDrawPixels( getEffectiveNbCols_(), nbRows_,
432  GL_LUMINANCE,
433  GL_UNSIGNED_BYTE,
434  getRaster_());
435  break;
436 
437  case kRGB24Raster:
438  glDrawPixels( getEffectiveNbCols_(), nbRows_,
439  GL_RGB,
440  GL_UNSIGNED_BYTE,
441  getRaster_());
442  break;
443

445  glDrawPixels( getEffectiveNbCols_(), nbRows_,
446  GL_RGBA,

452  break;
