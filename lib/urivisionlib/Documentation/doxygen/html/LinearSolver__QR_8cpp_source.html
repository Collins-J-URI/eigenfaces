476 void LinearSolver_QR::solveBS_(int nbRows, int nbCols, const double* const*const r,
477  const double* const* const b, double* const*const x)
478 {
479  // We want to solve R . X = B
480  // The outer loop deals with the columns of the right-side term
481  for (int j=0; j<nbCols; j++)
482  {
483  if (r[nbRows-1][nbRows-1] != 0)
484  x[nbRows-1][j] = b[nbRows-1][j] / r[nbRows-1][nbRows-1];
485  else
486  x[nbRows-1][j] = 0.L;
487 
488  for (int i=nbRows-2; i>=0; i--)
489  {
490  x[i][j] = b[i][j];
491  for (int k=i+1; k<nbRows; k++)
492  x[i][j] -= r[i][k] * x[k][j];
493 
494  if (r[i][i] != 0)
495  x[i][j] /= r[i][i];
496 
497  }
498  }
499 }

502 void LinearSolver_QR::applyQt_(int nbRows, int nbColsR, int nbColsB, const double* const*const v,
503  double* const*const b)
504 {
505  // We apply the Hv matrices (under the form of their v vector) that were computed
506  // for the different columns (left to right) of the SLE's original matrix
507  for (int j=0; j<nbColsR; j++)
508  {
509  // Vector v is applied to all columns of the right-side term
510  for (int l=0; l<nbColsB; l++)
511  {
512  double bv = 0.L;
513  // compute the dot product of the (partial) column and v
514  for (int i=j, k=0; i<nbRows; i++, k++)
515  bv += b[i][l]*v[j][k];
516 
517  // and subtract 2 v^T a v from a
518  bv *= 2.L;
519  for (int i=j, k=0; i<nbRows; i++, k++)
520  b[i][l] -= bv * v[j][k];
521 
522  }
523  }

530 LinearSolverRecordStatus LinearSolver_QR::factorize_(int nbRows, int nbCols, double* const*const r,
531  double* const colR, double* const*const v)
532 {
533  LinearSolverRecordStatus status = REGULAR_MATRIX, colStatus;
534 
535  // We proceed column by column, from the left to the right
536  for (int j=0, lv=nbRows; j<nbCols; j++, lv--)
537  {
538  // Step 0: Copy the data from column j of the matrix
539  for (int i=j; i<nbRows; i++)
540  colR[i] = r[i][j];
541 
542  // Step 1: Determine the v vector that constructs the orthogonal Householder Hv matrix
543  // Hv = I - 2 (v v^T) / (v^T v)
544  /// v is computed from elements from rows j to m-1 of column j
545  double alphaV;
546  colStatus = computeHouseholderV_(lv, colR+j, v[j], &alphaV);
547 
548  if (colStatus == SINGULAR_MATRIX)
549  status = SINGULAR_MATRIX;
550  else if ((colStatus == ILL_CONDITIONED_MATRIX) && (status != SINGULAR_MATRIX))
551  status = ILL_CONDITIONED_MATRIX;
552 
553  if (status != SINGULAR_MATRIX)
554  {
555  // column j is "compressed"
556  r[j][j] = alphaV;
557  // This is not really needed. I do it so far for aesthetic reasons but as soon
558  // as the code has been thoroughly tested, I will remove this for better
559  // performance.
560  for (int i=j+1; i<nbRows; i++)
561  r[i][j] = 0;
562 
563  // Vector v is applied to all other columns right of j. This only afffects
564  // rows j and below
565  for (int l=j+1; l<nbCols; l++)
566  {
567  double av = 0;
568  // compute the dot product of the (partial) column and v
569  for (int i=j, k=0; i<nbRows; i++, k++)
570  av += r[i][l]*v[j][k];
571 
572  // and subtract 2 v^T a v from a
573  av *= 2.L;
574  for (int i=j, k=0; i<nbRows; i++, k++)
575  r[i][l] -= av * v[j][k];
576 
577  }
578  }
579  }
580 
581  return status;
582 }

585 LinearSolverRecordStatus LinearSolver_QR::computeHouseholderV_(int n, double* const a, double* const v,
586  double* alphaV)
587 {
588  LinearSolverRecordStatus status = REGULAR_MATRIX;
589  double invNormV, normA, normA1 = 0.L;
590 
591 
592  for (int i=1; i<n; i++)
593  {
594  v[i] = a[i];
595  normA1 += a[i]*a[i];
596  }
597  normA = sqrt(normA1 + a[0]*a[0]);
598 
599  if (normA == 0.L)
600  status = SINGULAR_MATRIX;
601  else
602  {
603  if (normA < ILL_COND_THRESHOLD)
604  status = ILL_CONDITIONED_MATRIX;
605 
606  if (a[0] > 0.L)
607  {
608  v[0] = a[0] + normA;
609  *alphaV = -normA;
610  invNormV = 1.L / sqrt(normA1 + v[0]*v[0]);
611  }
612  else
613  {
614  v[0] = a[0] - normA;
615  *alphaV = normA;
616  invNormV = 1.L / sqrt(normA1 + v[0]*v[0]);
617  }
618 
619  // we normalize v to speedup later calculations
620  for (int i=0; i<n; i++)
621  v[i] *= invNormV;
622 
623  }
624 
625  return status;
