294 void RectifiedStereoMatcher_corr::computeSearchRects_(ImagePoint* pt, ImageRect* imgRect,
295  const int theWidth, const int theHeight,
296  const int verticalPadding,
297  const StereoImageID l_or_r, ImageRect* rect)
298 {
299  const int iLow = imgRect->getTop(),
300  iHigh = imgRect->getBottom(),
301  jLow = imgRect->getLeft(),
302  jHigh = imgRect->getRight(),
303  ptX = pt->getX(),
304  ptY = pt->getY();
305  // because pattern matching uses the upper-left corner of the pattern as
306  // origin, we need to translate our point by the pattern's half dimensions
307 
308  int iMin, iMax, jMin, jMax,
309  shiftedPtX,
310  shiftedPtY;
311 
312  //---------------------------------------------------
313  // Small matcher
314  //---------------------------------------------------
315  shiftedPtX = ptX-theWidth/2,
316  shiftedPtY = ptY-theHeight/2;
317  iMin = shiftedPtY - verticalPadding;
318  iMax = shiftedPtY + verticalPadding;
319  if (iMin < iLow)
320  iMin = iLow;
321  if (iMax > iHigh)
322  iMax = iHigh;
323 
324  jMin = shiftedPtX - maxDisparity_;
325  if (jMin < jLow)
326  jMin = jLow;
327  jMax = pt->getX()+theWidth/2 + maxDisparity_;
328  if (jMax > jHigh)
329  jMax= jHigh;
330 
331  // find a match in the left image
332  if (l_or_r == LEFT_IMAGE)
333  {
334  // the search rectangle on the right horizontally
335  // o starts from the image's left border
336  // o ends to the left pt's x (infinite disp.), shifted by pat's half width
337  rect->setRect(jMin, iMin, shiftedPtX-jMin+1, iMax-iMin + 1);
338  }
339 
340  // find a match in the right image
341  else
342  {
343  // the search rectangle on the left horizontally
344  // o starts from the right pt's x (infinite disp.), shifted by pat's half width
345  // o ends at the image's right border - width
346  rect->setRect(shiftedPtX, iMin, jMax-theWidth-shiftedPtX+1, iMax-iMin + 1);
347  }
348 }
