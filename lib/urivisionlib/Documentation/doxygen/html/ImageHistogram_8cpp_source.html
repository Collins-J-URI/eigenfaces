257 void ImageHistogram::computeHistogram_(RasterImage_gray* img, const ImageRect* rect)
258 {
259  const int iLow = rect->getTop(),
260  iHigh = rect->getBottom(),
261  jLow = rect->getLeft(),
262  jHigh = rect->getRight();
263  const float scale = (1.f * histVals_) / (ucRangeMax_ - ucRangeMin_);
264  const float shift = -scale*ucRangeMin_;
265  const unsigned char* const* grayIn = img->getShiftedGray2D();
266 
267  // clear the histogram
268  for (int k=0; k<histVals_; k++)
269  hist_[k] = 0L;
270 
271  // If we have to compte a histogram for a large number of pixels (I arbitrarily put the bar
272  // at 4 times 256), then we are better off precomputing a lookup table
273  const long nbPixels = (iHigh-iLow+1)*(jHigh-jLow+1);
274  if (nbPixels > 1024)
275  {
276  int lut[256];
277  for (int l=0; l<256; l++)
278  lut[l] = static_cast<int>(shift + scale*l);
279 
280  // and do the counting
281  for (int i=iLow; i<=iHigh; i++)
282  for (int j=jLow; j<=jHigh; j++)
283  hist_[lut[grayIn[i][j]]]++;
284 
285  }
286  // for the histogram of a small region, we perform the conversion for each pixel
287  else
288  {
289  // and do the counting
290  for (int i=iLow; i<=iHigh; i++)
291  for (int j=jLow; j<=jHigh; j++)
292  hist_[static_cast<int>(shift + scale*grayIn[i][j])]++;
293  }
294 }
