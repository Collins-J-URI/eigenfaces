66 //jyh I am not sure to what extent what I do degrades the quality of the generator.
67 //Because RAND_MAX can be very low (32767) the values stored in the two buffers might
68 //be very small as well. So I make sure that all 32 bits of the buffers's elements
69 //are used.
70 void RandomGenerator_R250_R521::setSeed(long theSeed)
71 {
72  //jyh
73  //I use a seed to initialize the C random generator
74  unsigned int mySeed;
75  if (INT_IS_LONG)
76  mySeed = static_cast<unsigned int>(theSeed);
77  else
78  {
79  // keep the most significant bits of the time stamp to be the seed
80  mySeed = static_cast<unsigned int>(theSeed >> (sizeof(long)-sizeof(int)));
81  }
82  srand(mySeed);
83 
84  int i = R521_LEN;
85  unsigned long mask1 = 1;
86  unsigned long mask2 = 0xFFFFFFFF;
87 
88  //jyh
89  while (i-- > R250_LEN) {
90  r521_buffer_[i] = lrand();
91  }
92  while (i-- > 31) {
93  r250_buffer_[i] = lrand();
94  r521_buffer_[i] = lrand();
95  }
96 
97  /*Michael Brundage
98  Establish linear independence of the bit columns
99  by setting the diagonal bits and clearing all bits above
100  */
101  while (i-- > 0) {
102  r250_buffer_[i] = (lrand() | mask1) & mask2;
103  r521_buffer_[i] = (lrand() | mask1) & mask2;
104  mask2 ^= mask1;
105  mask1 >>= 1;
106  }
107  r250_buffer_[0] = mask1;
108  r521_buffer_[0] = mask2;
109  r250_index_ = 0;
110  r521_index_ = 0;
111 
112  //jyh
113  //I call nextULong to get rid of the first number in the generator, which is
114  //always an extremely large one (would make a nextDouble of 0.99x)
115  nextULong();
116 }

118 unsigned long RandomGenerator_R250_R521::nextULong(void)
119 {
120  /*Michael Brundage
121  I prescale the indices by sizeof(unsigned long) to eliminate
122  four shlwi instructions in the compiled code. This minor optimization
123  increased perf by about 12%.
124 
125  I also carefully arrange index increments and comparisons to minimize
126  instructions. gcc 3.3 seems a bit weak on instruction reordering. The
127  j1/j2 branches are mispredicted, but nevertheless these optimizations
128  increased perf by another 10%.
129  */
130 
131  int i1 = r250_index_;
132  int i2 = r521_index_;
133  unsigned char* b1 = (unsigned char*)r250_buffer_;
134  unsigned char* b2 = (unsigned char*)r521_buffer_;
135  unsigned long * tmp1, * tmp2;
136  unsigned long r, s;
137  int j1, j2;
138 
139  j1 = i1 - R250_IB;
140  if (j1 < 0)
141  j1 = i1 + R250_IA;
142  j2 = i2 - R521_IB;
143  if (j2 < 0)
144  j2 = i2 + R521_IA;
145 
146  tmp1 = (unsigned long *)(b1 + i1);
147  r = (*(unsigned long *)(b1 + j1)) ^ (*tmp1);
148  *tmp1 = r;
149  tmp2 = (unsigned long *)(b2 + i2);
150  s = (*(unsigned long *)(b2 + j2)) ^ (*tmp2);
151  *tmp2 = s;
152 
153  i1 = (i1 != sizeof(unsigned long)*(R250_LEN-1)) ? (i1 + sizeof(unsigned long)) : 0;
154  r250_index_ = i1;
155  i2 = (i2 != sizeof(unsigned long)*(R521_LEN-1)) ? (i2 + sizeof(unsigned long)) : 0;
156  r521_index_ = i2;
157 
158  return r ^ s;
159 }

227 unsigned long lrand(void)
228 {
229  unsigned long rl = rand();
230  for (int i=1; i<RandomGenerator_R250_R521::FULL_IN_ULONG; i++)
231  rl = (rl << MAX_BITS_IN_RAND) | rand();
232 
233  if (RandomGenerator_R250_R521::RES_TO_ULONG)

236  (MAX_BITS_IN_RAND - RandomGenerator_R250_R521::RES_TO_ULONG);
237  rl = (rl << RandomGenerator_R250_R521::RES_TO_ULONG) | lsb;
238  }
