334 void ImagePoint_F::rotate(float angle, const ImagePoint* rotCenter)
335 {
336  const float cT = cosf(angle),
337  sT = sinf(angle);
338 
339  // if the rotation center is null we rotate around the origin
340  if (rotCenter == NULL)
341  {
342  const float x = cT*xf_ - sT*yf_;
343  yf_ = sT*xf_ + cT*yf_;
344  xf_ = x;
345  }
346  else
347  {
348  float xcf, ycf;
349  if (typeid(*rotCenter) == typeid(uriVL::ImagePoint_F))
350  {
351  const ImagePoint_F *rotCenterF = (ImagePoint_F*) rotCenter;
352  xcf = rotCenterF->xf_;
353  ycf = rotCenterF->yf_;
354  }
355  else
356  {
357  xcf = rotCenter->getX();
358  ycf = rotCenter->getY();
359  }
360  const float dx = x_ - xcf;
361  const float dy = y_ - ycf;
362  xf_ = xcf + cT*dx - sT*dy;
363  yf_ = ycf + sT*dx + cT*dy;
364  }
365 
366  x_ = static_cast<int>(round(xf_));
367  y_ = static_cast<int>(round(yf_));
368 }
