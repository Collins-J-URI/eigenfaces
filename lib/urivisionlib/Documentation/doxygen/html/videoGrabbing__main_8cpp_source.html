215 //---------------------------------------------------------------------------------
216 // setupOrthoCam
217 //
218 // This function maps world coordinates to screen coordinates. This is where
219 // the projection mode would be defined. Here I only want to have a 2D
220 // rectangle where to draw 2D stuff on top of my image, so I define my camera
221 // as an orthographic camera, and I set the world coordinates to coincide with
222 // the pixel coordinates (this is what the gluOrtho2D call does).
223 //---------------------------------------------------------------------------------
224 void setupOrthoCam(int w, int h)
225 {
226  glMatrixMode(GL_PROJECTION);
227  glLoadIdentity();
228  gluOrtho2D( 0.0f, (GLfloat) w, 0.0f, (GLfloat) h);
229 
230  glMatrixMode(GL_MODELVIEW);
231  glLoadIdentity();
232  glViewport(0, 0, w, h);
233 }
