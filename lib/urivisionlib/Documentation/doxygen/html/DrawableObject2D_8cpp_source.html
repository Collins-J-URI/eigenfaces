579 void DrawableObject2D::translate(int dx, int dy)
580 {
581  // Needed to accomodate the child classes that don't maintain their bounding
582  // rectangle at all times: forces recomputation for bounding rect if needed
583  const ImageRect* theRect = getBoundRect();
584  setBoundRect_(theRect->getLeft() + dx, theRect->getTop() + dy,
585  theRect->getWidth(), theRect->getHeight());
586  boundRect_.translate(dx, dy);
587 
588  // Same thing with valid rectangle
589  theRect = getValidRect();
590  setValidRect(theRect->getLeft() + dx, theRect->getTop() + dy,

777  int& scaledLeft, int& scaledTop,
778  int& scaledRight, int& scaledBottom) const
779 {
780  const int renderWinHeight = getWindowHeight();
781  const int renderWinWidth = getWindowWidth();
782 
783  if (origin != NULL)
784  {
785  FAIL_CONDITION( (renderWinHeight < 0) || (renderWinWidth < 0),
786  kInvalidRectangleError,
787  "The dimensions of the window must be known in the shifted mode");
788 
789  // if we display at a specified location, then the global (0, 0) corner should be
790  // at the upper left corner of our window.
791  scaledLeft = static_cast<int>(floorf(origin->getX()*scaleX_));
792  scaledRight = static_cast<int>(floorf((origin->getX() + (theRight - theLeft))*scaleX_));
793  scaledTop = renderWinHeight - static_cast<int>(floorf(origin->getY()*scaleY_));
794  scaledBottom = renderWinHeight -
795  static_cast<int>(floorf(origin->getY() + (theBottom - theTop)*scaleY_));
796  }
797  else if (shifted)
798  {
799  FAIL_CONDITION( (renderWinHeight < 0) || (renderWinWidth < 0),
800  kInvalidRectangleError,
801  "The dimensions of the window must be known in the shifted mode");
802 
803  // if we display shifted, then the global (0, 0) corner should be in
804  // the upper left corner of our window.
805  scaledLeft = static_cast<int>(floorf(theLeft*scaleX_));
806  scaledRight = static_cast<int>(floorf(theRight*scaleX_));
807  scaledTop = renderWinHeight - static_cast<int>(floorf(theTop*scaleY_));
808  scaledBottom = renderWinHeight - static_cast<int>(floorf(theBottom*scaleY_));
809  }
810  else
811  {
812  // If we display unshifted, then we need to move to the upper-left
813  // corner of the image, and draw right and down from there.
814  scaledLeft = 0;
815  scaledRight = static_cast<int>(floorf(getWidth()*scaleX_));
816  scaledTop = static_cast<int>(floorf(getHeight()*scaleY_));
817  scaledBottom = 0;
818  }
819 }

853 void DrawableObject2D::render(float theZ, bool shifted, const ImagePoint* origin) const
854 {
855  // If needed, forces a recalculation of the bounding rectangle
856  // From this point until the end of the function, the bounding rect will not change,
857  // so, e.g., trueBoundRect->getLeft(), getLeft(), and getBoundRect()->getLeft() all return
858  // the same value.
859  const ImageRect* trueBoundRect = getBoundRect();
860 
861  //--------------------------------------------------------------
862  // Part 1: update rendering location if needed
863  //--------------------------------------------------------------
864  if ((hasIndivWindowDimensions_ && dimensionsHaveChangedSinceLastRender_) ||
865  defaultDimensionsHaveChangedSinceLastRender_)
866  {
867  computeScaledCorners_(getLeft(), getTop(), getRight(), getBottom(), shifted, origin,
868  scaledLeft_, scaledTop_, scaledRight_, scaledBottom_);
869  dimensionsHaveChangedSinceLastRender_ = false;
870  }
871 
872  //--------------------------------------------------------------
873  // Part 2: do subclass-specific rendering stuff
874  //--------------------------------------------------------------
875  render_(scaledLeft_, scaledTop_, scaledRight_, scaledBottom_, theZ, shifted, origin);
876 
877 
878  //--------------------------------------------------------------
879  // Part 3: if required, indicate the bounding box by a colored frame
880  //--------------------------------------------------------------
881  if (mustDrawFrame_)
882  {
883  float pRed, pGreen, pBlue;
884  if (frameColor_ != NULL)
885  {
886  pRed = frameColor_[0];
887  pGreen = frameColor_[1];
888  pBlue = frameColor_[2];
889  }
890  else
891  pRed = pGreen = pBlue = 0.f;
892 
893  // The actual drawing
894  glPushMatrix();
895  glLoadIdentity();
896  glTranslatef(scaledLeft_, scaledTop_, theZ);
897  glScalef(scaleX_, -scaleY_, 1.f);
898  glColor3f(pRed, pGreen, pBlue);
899  glBegin(GL_LINE_LOOP);
900  glVertex3f(0, 0, 0);
901  glVertex3f(trueBoundRect->getWidth(), 0, 0);
902  glVertex3f(trueBoundRect->getWidth(), trueBoundRect->getHeight(), 0);
903  glVertex3f(0, trueBoundRect->getHeight(), 0);
904  glEnd();
905  glPopMatrix();
906  }
907 
908  //--------------------------------------------------------------
909  // Part 4: if required and necessary (not all data are valid),
910  // indicate the valid data rectangle by a colored frame
911  //--------------------------------------------------------------
912  if (mustDrawValidDataFrame_ && (!validRect_.coincidesWith(&boundRect_) || !mustDrawFrame_))
913  {
914  const ImageRect* trueValidRect = getValidRect();
915  int shiftX = trueValidRect->getLeft() - getLeft();
916  int shiftY = trueValidRect->getTop() - getTop();
917 
918  float pRed, pGreen, pBlue;
919  if (frameColor_ != NULL)
920  {
921  pRed = 1.f - frameColor_[0];
922  pGreen = 1.f - frameColor_[1];
923  pBlue = 1.f - frameColor_[2];
924  }
925  else
926  pRed = pGreen = pBlue = 0.f;
927 
928  // The actual drawing
929  glPushMatrix();
930 
931  // moved to the location of the bounding rect
932  glTranslatef(scaledLeft_, scaledTop_, theZ);
933  // apply the rendering scale
934  glScalef(scaleX_, -scaleY_, 1.f);
935  // then apply relative shift to the origin of the valid rect
936  glTranslatef(shiftX, shiftY, 0.f);
937 
938  glColor3f(pRed, pGreen, pBlue);
939  glBegin(GL_LINE_LOOP);
940  glVertex3f(0, 0, 0);
941  glVertex3f(trueValidRect->getWidth(), 0, 0);
942  glVertex3f(trueValidRect->getWidth(), trueValidRect->getHeight(), 0);
943  glVertex3f(0, trueValidRect->getHeight(), 0);
944  glEnd();
945 
946  glPopMatrix();
947  }
948 }
