583 // Should update the non-reference raster, using the values stored in the
584 // reference raster.
585 void RasterImage_gray_F::updateRaster_(void) const
586 {
587  const ImageRect* validRect = getValidRect();
588  const int iLow = validRect->getTop(),
589  iHigh = validRect->getBottom(),
590  jLow = validRect->getLeft(),
591  jHigh = validRect->getRight();
592  float *const* fltGray;
593  unsigned char *const* gray;
594 
595  // Before we call functions of the parent class with side effects on the
596  // "reference" flags, we copy their values to know what actually needs
597  // to be updated
598  bool isCharToFloatUpdate = graphWorldIsReference();
599 
600  // Not the prettiest solution, but gets the job done to preserve logical constness
601  RasterImage_gray_F* updateableThis = const_cast<RasterImage_gray_F*>(this);
602 
603  // if the GWorld has not been allocated (typically
604  // the first time we render a RasterImage_gray_F object) then we must allocate
605  // it at the same dimensions (bounding rectangle) as the float raster
606  if (graphWorldIsNull_())
607  {
608  updateableThis->allocateGraphWorld_(getHeight(), getWidth());
609  FAIL_CONDITION( graphWorldIsNull_(),
610  kGWorldAllocationError,
611  "GraphWorld allocation failed in RasterImage_gray_F::updateRaster_");
612  updateableThis->RasterImage_gray::allocateRaster2D_(getHeight(), getWidth());
613  updateableThis->RasterImage_gray::allocateShiftedRaster2D_(getHeight(), getWidth(), getTop(), getLeft());
614  }
615 
616  // if the bounding rectangle has its origin at (0, 0) then the shifted
617  // raster was never allocated or it is equal to the unshifted raster
618  if ((getTop()==0) && (getLeft()==0))
619  {
620  gray = updateableThis->RasterImage_gray::getGray2D(R_W_ACCESS);
621  fltGray = fltGray2D_;
622  }
623  // otherwise I use the shifted raster
624  else
625  {
626  gray = updateableThis->RasterImage_gray::getShiftedGray2D(R_W_ACCESS);
627  fltGray = shiftedFltGray2D_;
628  }
629 
630  // First find out which raster is the reference to update from
631  if (isCharToFloatUpdate)
632  {
633  for (int i=iLow; i<=iHigh; i++)
634  for (int j=jLow; j<=jHigh; j++)
635  fltGray[i][j] = gray[i][j];
636  }
637  else
638  {
639  ConversionFromFloatMode convMode;
640  if (hasIndivFloatConvMode_)
641  convMode = floatConv_;
642  else
643  convMode = defaultFloatConv_;
644 
645  if ((convMode==kSignedFloat) || (convMode==kPositiveFloat) || (convMode==kRangeFloat))
646  calculateFloatConversionParams_( );
647  long nbCols = jHigh - jLow + 1;
648  for (int i=iLow; i<=iHigh; i++)
649  updateableThis->convertToGray_(fltGray[i]+jLow, gray[i]+jLow, nbCols, valMin_, convScale_, convMode);
650 
651  }
652 
653  if (isCharToFloatUpdate)
654  setGraphWorldAsReference_();
655  else
656  setFloatRasterAsReference_();
657 
658 }

688 void RasterImage_gray_F::copyInto(RasterImage* imgOut, const ImageRect* theRect,
689  const ImagePoint* origin) const
690 {

921 void RasterImage_gray_F::calculateFloatConversionParams_(void) const
922 {
923  const float* const* theGray;
924  const ImageRect* validRect = getValidRect();
925  int iLow = validRect->getTop(),
926  iHigh = validRect->getBottom(),
927  jLow = validRect->getLeft(),
928  jHigh = validRect->getRight();
929 
930  if ((getTop()==0) && (getLeft()==0))
931  theGray = fltGray2D_;
932  else
933  theGray = shiftedFltGray2D_;
934 
935 
936  ConversionFromFloatMode convType;
937  if (hasIndivFloatConvMode_)
938  convType = floatConv_;
939  else
940  convType = defaultFloatConv_;
941 
942  valMin_ = valMax_ = fltGray2D_[iLow][jLow];
943  for (int i=iLow; i<=iHigh; i++)
944  for (int j=jLow; j<=jHigh; j++)
945  if (theGray[i][j] > valMax_)
946  valMax_ = theGray[i][j];
947  else if (theGray[i][j] < valMin_)
948  valMin_ = theGray[i][j];
949 
950  switch (convType)
951  {
952  // We don't care about the MIN, simply rescale the MAX to 255
953  case kPositiveFloat:
954  valMin_ = 0.f;
955  convScale_ = 255.0f/valMax_;
956  break;
957 
958  // We rescale to 0 255, with 0 at 128
959  case kSignedFloat:
960  if (valMax_ > -valMin_)
961  valMin_ = -valMax_;
962  else
963  valMax_ = -valMin_;
964 
965  convScale_ = 127.5f/valMax_;
966  break;
967 
968  case kRangeFloat:
969  convScale_ = 255.f/(valMax_ - valMin_);
970  break;
971 
972  default:
973  break;
974  }
975 }

978 void RasterImage_gray_F::convertToGray_(const float* theSource,
979  unsigned char* theDestination,
980  long nbPixels,
981  float valMin, float convScale,
982  ConversionFromFloatMode myFloatConv)
983 {
984  const float *grayIn = theSource;
985  unsigned char *grayOut = theDestination;
986 
987  switch (myFloatConv)
988  {
989  case kSignedFloat:
990  case kPositiveFloat:
991  case kRangeFloat:
992  for (long l=0; l<nbPixels; l++)
993  *(grayOut++) = static_cast<unsigned char>(convScale*(*(grayIn++)-valMin));
994  break;
995 
996 
997  case kSaturatedPositive:
998  for (long l=0; l<nbPixels; l++)
999  {
1000  float val = *(grayIn++);
1001  if (val >= 0.f)
1002  if (val <= 255.f)
1003  *(grayOut++) = static_cast<unsigned char>(val);
1004  else
1005  *(grayOut++) = static_cast<unsigned char>(255);
1006  else
1007  *(grayOut++) = static_cast<unsigned char>(0);
1008  }
1009  break;
1010 
1011 
1012  case kSaturatedSigned:
1013  for (long l=0; l<nbPixels; l++)
1014  {
1015  float val = *(grayIn++);
1016 
1017  if (val >= -128.f)
1018  if (val <= 127.f)
1019  *(grayOut++) = static_cast<unsigned char>(val + 128);
1020  else
1021  *(grayOut++) = static_cast<unsigned char>(255);
1022  else
1023  *(grayOut++) = static_cast<unsigned char>(0);
1024  }
1025  break;
1026 
1027  default:
1028  FAIL_CONDITION( true,
1029  kRasterImageError,
1030  "Unknown float to gray conversion mode");
1031  break;
1032  }
1033 }
