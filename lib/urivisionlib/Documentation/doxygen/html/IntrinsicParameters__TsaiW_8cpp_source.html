193 void IntrinsicParameters_TsaiW::distortPixels(double ux, double uy, double* dx, double* dy) const
194 {
195  // 1. move to the center
196  double cx = ux - Ox_;
197  double cy = uy - Oy_;
198 
199  // 2. calculate scaling factor
200  const double ru2 = cx*cx + cy*cy;
201  const double p = 1.L / (kappa1_ * ru2);
202  const double p2 = p*p;
203  const double bigTerm = 9.L*p + sqrt(81.L*p2 + 12.L*p2*p);
204  const double mu = 1.f + double( bigTerm * pow(18, -1.L/3.L) -
205  pow(2.L/3.L, 1.L/3.L) / bigTerm);
206 
207  // 3. apply scaling
208  *dx = mu * cx;
209  *dy = mu * cy;
210 
211  // 4. apply translation back
212  *dx += Ox_;
213  *dy += Oy_;
214 }
