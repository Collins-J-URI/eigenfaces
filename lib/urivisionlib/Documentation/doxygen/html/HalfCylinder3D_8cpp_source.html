70 HalfCylinder3D::HalfCylinder3D(GLfloat* origin, GLfloat* dirAxis, GLfloat radius,
71  bool lengthIsClosed, bool sidesAreClosed)
72  : TubularSection3D(origin, dirAxis, radius, 0.f, PI_F),
73  //
74  closedLength_(lengthIsClosed),
75  closedSides_(sidesAreClosed)
76 {
77  GLfloat **axes = GlUtils::glfMatrix(0, 2, 0, 2);
78  //
79  // dirAxis is the direction axis of the cylinder
80  axes[2][0] = dirAxis[0];
81  axes[2][1] = dirAxis[1];
82  axes[2][2] = dirAxis[2];
83  //
84  // For the major axis, we need to build a vector orthogonal to dirAxis.
85  if ((dirAxis[0]!= 0.f) || (dirAxis[2]!= 0.f)) // this is not the local Y axis
86  {
87  GLfloat w[3] = {0.f, 1.f, 0.f};
88  commonNormalVector(w, dirAxis, axes[0]); // "X" = "Y" x "Z"
89  commonNormalVector(dirAxis, axes[0], axes[1]); // "Y" = "Z" x "X"
90  //
91  // and now we can multiply by the radius
92  axes[0][0] *= radius;
93  axes[1][0] *= radius;
94  //
95  axes[0][1] *= radius;
96  axes[1][1] *= radius;
97  //
98  axes[0][2] *= radius;
99  axes[1][2] *= radius;
100  }
101  // if dirAxis was the local Y axis, then I can take Z and X as major and minor
102  else
103  {
104  axes[0][0] = axes[0][1] = axes[1][1] = axes[1][2] = 0.f;
105  axes[0][2] = axes[1][0] = radius;
106  }
107 
108  init_(axes);
109 
110  GlUtils::free_glfMatrix(axes, 0, 2, 0, 2);
111 }

223 /*----------------------------------------------------------------------------------+
224 | Function: void init(GLfloat** a) |
225 | |
226 | Status: Public |
227 | |
228 | Parameters: a coordinates of the major axis, minor axis, and |
229 | direction axis in a 3x3 0-based array. |
230 | |
231 | Returns: Nothing |
232 | |
233 | Description: Initializes the array of vertex and normal coordinates |
234 | |
235 +----------------------------------------------------------------------------------*/
236 void HalfCylinder3D::init_(GLfloat** a)
237 {
238  GLfloat dirStep = 1.f/kVeryHighDefSlabs,
239  dir[kVeryHighDefSlabs+1][3],
240  majCross[3], minCross[3];
241 
242  // majCross = a[0] x a[2] (major axis x direction)
243  crossProduct(a[0], a[2], majCross);
244  // minCross = a[1] x a[2] (minor axis x direction)
245  crossProduct(a[1], a[2], minCross);
246 
247  // First, we precompute the graduation steps for the slabs along
248  // the "direction" axis (a[2]).
249  for (int i=0; i<=kVeryHighDefSlabs; i++)
250  for (int k=0; k<3; k++)
251  dir[i][k] = i*dirStep*a[2][k];
252 
253 
254  // if the top and bottom sides are filled, normal vectors must be computed
255  // for these sides
256  if (closedSides_)
257  {
258  // topNormal = major x minor / ||major x minor||
259  commonNormalVector(a[0], a[1], normalTop_);
260  normalBottom_[0] = -normalTop_[0];
261  normalBottom_[1] = -normalTop_[1];
262  normalBottom_[2] = -normalTop_[2];
263  }
264  // Same thing for the length side
265  if (closedLength_)
266  // normalLength = major x direction / ||major x direction||
267  commonNormalVector(a[0], a[2], normalLength_);
268 }
