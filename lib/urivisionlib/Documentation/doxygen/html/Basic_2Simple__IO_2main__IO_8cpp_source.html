15 /*--------------------------------------------------------------------------+
16 | This small demo program is recycled from code I wrote for the computer |
17 | graphics course to show how to do texture mapping. |
18 | In the original program, the images were read from files or simply |
19 | generated to be mapped as textures on a graphic object. |
20 | In this revised version, they are now displayed in a glut window. |
21 | |
22 | You can cycle through the windows with the '[' and ']' keys. |
23 | |
24 | Note: If you built this demo using QuickTime (rather than FreeImage |
25 | of the "hack reader/writer," a file selector window will pop up. It |
26 | waits for you to open an image file (jpeg, gif, etc.). This is only |
27 | the case with the QuickTime version, because QuickTime for windows |
28 | include a very primitive GUI library, so I exploit this in the demo |
29 | to show how to use a file selecter. All the other images are read |
30 | using a file path (the way it's done with the non-QuickTime builds of |
31 | this demo). |
32 | |
33 | The important point to keep in mind is that in C/C++ our image is |
34 | a *1D* array of unsigned char. If you want to manipulate your image |
35 | as a 2D array, you will have to create arrays for your row pointers. |
36 | |

173  //-------------------------------------------------------------
174  // Image 0 is created by computation --> fill the array with RGB values
175  //-------------------------------------------------------------
176  n = 256;
177  //
178  unsigned char *tempBuf = new unsigned char[n * n * 3];
179 
180  if (tempBuf !=NULL)
181  {
182  for (int i=0; i<n; i++)
183  {
184  idx = 3 * n * i;
185 
186  for (int j=0; j<n; j++)
187  {
188  tempBuf[idx++] = static_cast<unsigned char>(i);
189  tempBuf[idx++] = static_cast<unsigned char>(255-j);
190  tempBuf[idx++] = static_cast<unsigned char>(0);
191  }
192  }
193  gMyImage[0] = new RasterImage_RGB(n, n, tempBuf);
194  delete tempBuf;
195  }
196  else
197  {
198  cout << "Generation of image 0 failed.";
199  cleanupAndQuit(-1);
200  }
201 
202 
203  //-------------------------------------------------------------
204  // Image 1 is created by computation --> fill the array with RGBA values
205  // This code fragment demonstrate how to get direct access to the image's
206  // raster for initialization and other IP calculations
207  //-------------------------------------------------------------
208  n = 512;
209  //
210  // the raster image is created at the desired dimensions
211  gMyImage[1] = new RasterImage_RGBa(n, n);
212  if (gMyImage[1] != NULL)
213  {
214  // we get a reference to the image's base raster
215  // Here, we use the version of the getRaster family of functions that return a
216  // read-write reference
217  unsigned char *const* rgba = gMyImage[1]->getRaster2D(R_W_ACCESS);
218 
219  // just to play nice, we get the image's dimensions, although
220  // they should be both equal to n
221  int nbRows = gMyImage[1]->getHeight(),
222  nbCols = gMyImage[1]->getWidth();
223  // from this point on
224  for (int i=0; i<nbRows; i++)
225  for (int j=0, l=0; j<nbCols; j++, l+=4)
226  {
227  rgba[i][l] = static_cast<unsigned char>(i/4);
228  rgba[i][l+1] = static_cast<unsigned char>(j/4);
229  rgba[i][l+2] = static_cast<unsigned char>(floor(127*(cos(0.06*i)*sin(0.1*j)+0.5)));
230  rgba[i][l+3] = static_cast<unsigned char>(255);
231  }
232 
233  // By default, the RasterImage class adopts the "safe" mode for the definition
234  // of the valid data rectangle. Since image1 was just allocated not initialized,
235  // its valid data rectangle was located at the upper left corner of the bounding
236  // rectangle, with width and height 0. We must now set the correct values.
237  gMyImage[1]->setValidRect(0, 0, nbCols, nbRows);
238  }
239  else
240  {
241  cout << "Generation of image 1 failed.";
242  cleanupAndQuit(-1);
243  }
244 
245 
246  //-------------------------------------------------------------
247  // Image 2 is created by computation --> fill the array with Gray values
248  //-------------------------------------------------------------
249  n = 256;
250  //
251  gMyImage[2] = new RasterImage_gray(n, n);
252  if (gMyImage[2] !=NULL)
253  {
254  // Here again, we use the version of the getRaster family of functions that return a
255  // read-write reference
256  unsigned char *const* gray = gMyImage[2]->getRaster2D(R_W_ACCESS);
257 
258  for (int i=0; i<n; i++)
259  for (int j=0; j<n; j++)
260  gray[i][j] = static_cast<unsigned char>(i>j ? i : j);
261 
262  }
263  else
264  {
265  cout << "Generation of image 3 failed.";
266  cleanupAndQuit(-1);
267  }

271  //-------------------------------------------------------------
