80 void ImageLaplacian_Sobel::applyInto(RasterImage* imgIn, ImageRect* theRect, RasterImage* imgOut)
81 {
82  //----------------------------------------------------------------
83  // STEP 0: Test for null or invalid type parameters
84  //----------------------------------------------------------------
85  FAIL_CONDITION( imgIn == NULL,
86  kNullRasterImageError,
87  "NULL input raster image passed as parameter");
88 
89  FAIL_CONDITION( theRect == NULL,
90  kNullRectangleError,
91  "NULL ImageRect parameter in ImageLaplacian_Sobel::applyInto");
92 
93  FAIL_CONDITION( imgOut == NULL,
94  kNullRasterImageError,
95  "NULL output raster image passed as parameter");
96 
97  // so far I only accept a RasterImage_gray_F as a destination
98  FAIL_CONDITION( (imgOut->getBaseRasterType() != kGrayRaster) ||
99  !(imgOut->hasFloatRaster()),
100  kInvalidRasterType,
101  "ImageLaplacian_Sobel::applyInto only accepts RasterImage_gray_F images as destination");
102 
103 
104  //----------------------------------------------------------------
105  // STEP 1: Test for rectangle intersection
106  //----------------------------------------------------------------
107  // determine the output rectangle that the input image can produce.
108  // Here, because we apply a filter of width&height 3, we
109  // "lose m all around".
110  ImageRect *usableRect = new ImageRect(imgIn->getValidRect());
111  usableRect->removeAllAround(1);
112  // ... and determine its intersection with the user-specified output rectangle
113  ImageRect *destRect = ImageRect::intersection(theRect, usableRect);
114  delete usableRect;
115  FAIL_CONDITION( destRect == NULL,
116  kNullRectangleError,
117  "Destination rectangle does not intersect valid data rectangle");
118  // if the output image does not contain the destination rect, it must be resized.
119  if (!imgOut->contains(destRect))
120  imgOut->setBoundRect(destRect);
121 
122 
123  //----------------------------------------------------------------
124  // STEP 2: Operate only on a gray-level image
125  //----------------------------------------------------------------
126  bool localGray;
127  RasterImage_gray *grayImg = RasterImage_gray::localRasterGray(imgIn, &localGray);
128 
129 
130  //----------------------------------------------------------------
131  // STEP 3: Perform the calculations
132  //----------------------------------------------------------------
133  switch (form_)
134  {
135  case LAPLACIAN_SOBEL_FORM_1:
136  applyInto_form1_(grayImg, destRect, imgOut);
137  break;
138 
139  case LAPLACIAN_SOBEL_FORM_2:
140  applyInto_form2_(grayImg, destRect, imgOut);
141  break;
142 
143  case LAPLACIAN_SOBEL_FORM_3:
144  applyInto_form3_(grayImg, destRect, imgOut);
145  break;
146 
147  default:
148  FAIL_CONDITION( true,
149  kImageProcessingError,
150  "invalid Laplacian Sobel form");
151  break;
152  }
153 
154  //----------------------------------------------------------------
155  // STEP 4: Set rectangle
156  //----------------------------------------------------------------
157  // Very important: At the end of the computation, don't forget to set the
158  // output's valid data rectangle. In "safe" mode, the library has no way to guess
159  // what the valid data rectangle should be and sets it to its minimum value... empty
160  imgOut->setValidRect(destRect);
161  ((RasterImage_gray_F *) imgOut)->setFloatConversionMode(floatConversionMode_);
162  imgOut->setFloatRasterAsReference();
163 
164  //----------------------------------------------------------------
165  // STEP 5: Free temporary storage (if needed)
166  //----------------------------------------------------------------
167  if (localGray)
168  delete grayImg;
169  delete destRect;
170 }

173 void ImageLaplacian_Sobel::applyInto_form1_(RasterImage* imgIn, ImageRect* theRect,
174  RasterImage* imgOut)
175 {
176  int iLow = theRect->getTop(),
177  iHigh = theRect->getBottom(),
178  jLow = theRect->getLeft(),
179  jHigh = theRect->getRight();
180 
181  /// 2 cases
182  if (imgIn->hasFloatRaster())
183  {
184  float **inGray = ((RasterImage_gray_F *) imgIn)->getShiftedGrayF2D();
185  float **outGray = ((RasterImage_gray_F *) imgOut)->getShiftedGrayF2D();
186 
187  // compute convolution over the processing rectangle
188  int im1 = iLow-1, ip1 = iLow+1;
189  for (int i=iLow; i<=iHigh; i++)
190  {
191  int jm1 = jLow-1, jp1 = jLow+1;
192 
193  for (int j=jLow; j<=jHigh; j++)
194  {
195  outGray[i][j] = inGray[i][jp1] + inGray[i][jm1] +
196  inGray[im1][j] + inGray[ip1][j] -
197  4.f*inGray[i][j];
198 
199  // shift the j indexes (j itself is taken care of in the loop
200  // statement)
201  jm1 = j;
202  jp1++;
203  }
204 
205  // shift the i indexes (i itself is taken care of in the loop
206  // statement)
207  im1 = i;
208  ip1++;
209  }
210  }
211  else
212  {
213  unsigned char **inGray = imgIn->getShiftedRaster2D();
214  float **outGray = ((RasterImage_gray_F *) imgOut)->getShiftedGrayF2D();
215 
216  // compute convolution over the processing rectangle
217  int im1 = iLow-1, ip1 = iLow+1;
218  for (int i=iLow; i<=iHigh; i++)
219  {
220  int jm1 = jLow-1, jp1 = jLow+1;
221 
222  for (int j=jLow; j<=jHigh; j++)
223  {
224  outGray[i][j] = inGray[i][jp1] + inGray[i][jm1] +
225  inGray[im1][j] + inGray[ip1][j] -
226  4.f*inGray[i][j];
227 
228  // shift the j indexes (j itself is taken care of in the loop
229  // statement)
230  jm1 = j;
231  jp1++;
232  }
233 
234  // shift the i indexes (i itself is taken care of in the loop
235  // statement)

237  ip1++;
238  }
239  }
