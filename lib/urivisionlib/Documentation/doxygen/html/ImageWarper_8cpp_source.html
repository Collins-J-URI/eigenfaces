106 void ImageWarper::setImages(const RasterImage* imgIn, RasterImage* imgOut)
107 {
108  FAIL_CONDITION( imgIn == NULL,
109  kNullRasterImageError,
110  "null input image assigned to an ImageWarper");
111  FAIL_CONDITION( imgOut == NULL,
112  kNullRasterImageError,
113  "null output image assigned to an ImageWarper");
114 
115  // So far we only warp RGBa32 and gray rasters
116  BaseRasterType rasterType = imgIn->getBaseRasterType();
117  FAIL_CONDITION((rasterType != kGrayRaster) && (rasterType != kRGBa32Raster),
118  kInvalidParameterError,
119  "only RGBa32 and gray rasters supported yet");
120  FAIL_CONDITION( imgOut->getBaseRasterType() != rasterType,
121  kInvalidParameterError,
122  "input and output raster types must match (so far)");
123  rasterType_ = rasterType;
124 
125  // Verify that the input image specified has dimensions that are
126  // suitable to the warper
127  const int imgHeight = imgIn->getHeight(),
128  imgWidth = imgIn->getWidth();
129  int loWidth, hiWidth, loHeight, hiHeight;
130  getImageSuitableDimensions(imgWidth, imgHeight, &loWidth, &loHeight, &hiWidth, &hiHeight);
131  FAIL_CONDITION(((loWidth != imgWidth) && (hiWidth != imgWidth)) ||
132  ((loHeight != imgHeight) && (hiHeight != imgHeight)),
133  kInvalidParameterError,
134  "input image has improper dimensions for warping");
135  imgWidth_ = imgWidth;
136  imgHeight_ = imgHeight;
137  effImgWidth_ = getEffectivePixelsPerRow(imgWidth);
138  imgIn_ = imgIn;
139  imgIn1D_ = imgIn->getRaster();
140 
141  // make sure that the output image is large enough to store the result
142  // of the warping
143  if (!imgOut->contains(imgIn))
144  imgOut->setBoundRect(imgIn->getBoundRect());
145  imgOut_ = imgOut;
146  imgOut1D_ = imgOut->getRaster(R_W_ACCESS);
147 // imgOut_->setValidRect(imgIn->getValidRect());
148 
149  // if grid dimensions were previously set, we can now initialize the grids
150  if (xy_ != NULL)
151  initializeGrids_();
152 }

155 void ImageWarper::setGrid(int gridRows, int gridCols)
156 {
157  FAIL_CONDITION( gridRows < 3 || gridCols < 3,
158  kInvalidParameterError,
159  "gridRows and gridCols must >= 3");
160 
161  // if the user changed the grid's dimensions, we must delete the
162  // old grids first
163  if ((gridRows != gridRows_) || (gridCols != gridCols_))
164  deleteGrids_();
165 
166  // set the grid's dimensions
167  gridRows_ = gridRows;
168  gridCols_ = gridCols;
169  allocateGrids_();
170 
171  // if an image was previously set, we can also initialize the grid
172  if (imgIn1D_ != NULL)
173  initializeGrids_();
174 }

203  float*** grid = new float** [gridRows_];
204  for (int i=0; i<gridRows_; i++)
