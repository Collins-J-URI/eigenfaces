15 /*--------------------------------------------------------------------------+
16 | This simple application demonstrates the use of ImageWarper classes to |
17 | apply geometric transformations to an image. |
18 | This demo uses the OpenGL implementation of warping (ImageWarper_GL), |
19 | which in fact at the time of this writing is the sole implementation in |
20 | the library, but any other future implementation could be used instead, |
21 | the only difference that OpenGL places restrictions (on some platforms) |
22 | on the dimensions of its texture maps. |
23 | |
24 | The input image is displayed in the left subwindow, with the original |
25 | warping grid overlaid on it. |
26 | The right subwindow displays the warped image with the current warping |
27 | grid overlaid on it. The user can warp the image by clicking one of |
28 | the warping grid's nodes. The active node changes from blue to red. |
29 | The user can then click to where this point should be moved (within |
30 | the image). |

210 void initImagesAndWarper(void)
211 {
212  //-------------------------------------------------------------
213  // We read an image from a file
214  //-------------------------------------------------------------
215  const string filePath = color_dir + "bottles_256x256.tga";
216 // const string filePath = color_dir + "mandrill_256x256.tga";
217 // const string filePath = color_dir + "warpedMandrill_256x256.tga";
218  ImageReader myReader(const_cast<char *>(filePath.c_str()));
219  RasterImage *tempImage = myReader.getFrame(true);
220  if (tempImage ==NULL)
221  {
222  cout << "Error encountered while reading image file " << filePath << endl;
223  cleanupAndQuit(-1);
224  }
225  const BaseRasterType imgType = tempImage->getBaseRasterType();
226  if (imgType != kRGBa32Raster)
227  {
228  cout << "This demo only takes RGBA32 color images" << endl;
229  cleanupAndQuit(-1);
230  }
231 
232  //-------------------------------------------------------------
233  // Create the warper
234  //-------------------------------------------------------------
235  // Initially the warper has fixed border nodes and free-moving
236  // inner nodes
237  gInnerMode = INNER_FREE;
238  gBorderMode = BORDER_FIXED;
239  gWarper = new ImageWarper_GL(gInnerMode, gBorderMode);
240 
241  // we find out which image dimensions are recommended by the warper (closest
242  // to the input image's dimensions)
243  int imgHeight = tempImage->getHeight(),
244  imgWidth = tempImage->getWidth();
245  int loWidth, hiWidth, loHeight, hiHeight;
246  gWarper->getImageSuitableDimensions(imgWidth, imgHeight, &loWidth, &loHeight,
247  &hiWidth, &hiHeight);
248 
249  // for this demo I don't mind chopping a few pixels off the right and bottom
250  // edges of my image, so if the "low" values are at least 90% of the input
251  // I keep them
252  if (10*loWidth >= 9*imgWidth)
253  imgWidth = loWidth;
254  else
255  imgWidth = hiWidth;
256  if (10*loHeight >= 9*imgHeight)
257  imgHeight = loHeight;
258  else
259  imgHeight = loHeight;
260 
261  // Now we can copy (part of) the temporary image into the input image
262  ImageRect myRect(0, 0, imgWidth, imgHeight),
263  *trueRect = ImageRect::intersection(&myRect, tempImage->getValidRect());
264  gImgIn = new RasterImage_RGBa(imgHeight, imgWidth);
265  gImgOut = new RasterImage_RGBa(imgHeight, imgWidth);
266  gImgIn->setValidRect(&myRect);
267  gImgOut->setValidRect(&myRect);
268  tempImage->writeInto(gImgIn, trueRect);
269  delete trueRect;
270 }

516 // we only recourd mouse clicks in the right subwindow
517 void myMouse(int button, int buttonState, int ix ,int iy)
518 {
519  if ((button == GLUT_LEFT_BUTTON) && (buttonState == GLUT_UP))
520  {
521  // if no point was previously selected, this mouse click is used to
522  // select a new point
523  if ((gSelectedPtRow == -1) && (gSelectedPtCol == -1))
524  {
525  // did the user click on one of the active points?
526  bool pointClicked = false;
527  gSelectedPtRow = gSelectedPtCol = -1;
528  for (int i=1; (i<gGridRows-1) && !pointClicked; i++)
529  {
530  for (int j=1; (j<gGridCols-1) && !pointClicked; j++)
531  {
532  float dx = ix - gGridPoint[i][j][0],
533  dy = iy - gGridPoint[i][j][1],
534  dist2 = dx*dx + dy*dy;
535  if (dist2 < 25)
536  {
537  pointClicked = true;
538  gSelectedPtRow = i;
539  gSelectedPtCol = j;
540  }
541  }
542  }
543  }
544  // otherwise, the location of the mouse click gives the new coordinates of
545  // the grid point
546  else
547  {
548  float tempX = gGridPoint[gSelectedPtRow][gSelectedPtCol][0],
549  tempY = gGridPoint[gSelectedPtRow][gSelectedPtCol][1];
550  gGridPoint[gSelectedPtRow][gSelectedPtCol][0] = ix;
551  gGridPoint[gSelectedPtRow][gSelectedPtCol][1] = iy;
552 
553  // if this is a valid displacement, proceed
554  if (gWarper->isValidDisplacement(gGridPoint))
555  {
556  // Note that the warper performs again the above validity test. We only do it
557  // it so that we can restore the grid to a correct state if the displacement
558  // was not valid
559  applyWarp();
560  }
561  // otherwise restore grid point values and ignore
562  else
563  {
564  gGridPoint[gSelectedPtRow][gSelectedPtCol][0] = tempX;
565  gGridPoint[gSelectedPtRow][gSelectedPtCol][1] = tempY;
566  }
567 
568  // in any case, deselect the point
569  gSelectedPtRow = gSelectedPtCol = -1;
570  }
571  }

574 }
