329 /* Should update the non-reference raster, using the values stored in the reference
330  * raster.
331  */
332 void RasterImage_binary::updateRaster_(void) const
333 {
334  if (referenceHasChanged_)
335  {
336  // Change this flag now to avoid an infinite loop in copyInto
337  setReferenceHasChanged_(false);
338 
339  if (gworldIsReference_)
340  {
341  // If we have never allocated the gray raster, now would be a good time
342  // to do so.
343  if (grayForm_ == NULL)
344  grayForm_ = new RasterImage_gray(getBoundRect());
345 
346  // Copy the binary raster into the gray one
347  copyInto(grayForm_);
348  }
349  else
350  {
351  // To copy the gray raster into the binary one, we first need to
352  // "saturate the gray raster
353  const ImageRect* validRect = getValidRect();
354  const int iLow = validRect->getTop(),
355  iHigh = validRect->getBottom(),
356  jLow = validRect->getLeft(),
357  jHigh = validRect->getRight();
358  unsigned char *const* gray = grayForm_->getShiftedRaster2D(R_W_ACCESS);
359 
360  for (int i=iLow; i<=iHigh; i++)
361  for (int j=jLow; j<=jHigh; j++)
362  if (gray[i][j] != 0)
363  gray[i][j] = static_cast<unsigned char>(255);
364 
365  // then we do the CopyBits
366  // OK, the const_cast is ugly, but this is really the only way to preserve
367  // logical constness
368  grayForm_->copyInto(const_cast<RasterImage_binary*>(this));
369  }
370  }

373 RasterImage_gray* RasterImage_binary::getGrayRasterImage(void) const
374 {
375  // if the gworld is the reference
376  if (gworldIsReference_)
377  {
378  // If we have never allocated the gray raster, now would be a good time
379  // to do so.
380  if (grayForm_ == NULL)
381  grayForm_ = new RasterImage_gray(const_cast<RasterImage_binary*>(this));
382  // otherwise, udpdate the gray raster only if the binary raster has changed
383  else if (referenceHasChanged_)
384  {
385  // sets the flag to false to avoid loop repeats
386  setReferenceHasChanged_(false);
387 
388  // Copy the binary raster into the gray one
389  // I cast away this object's constness because the changes that may occur
390  // are only of type update. The "true" content is not changed
391  (const_cast<RasterImage_binary*>(this))->copyInto(grayForm_);
392  }
393  }
394 
395  // the gray raster is now the reference
396  setOtherRasterAsReference_();
397  unpackedRasterIsReference_ = true;
398 
399  return grayForm_;
400 }

446 void RasterImage_binary::writeInto_(RasterImage* imgOut, const ImageRect* copyRect,
447  const ImagePoint* origin) const
448 {
449  const unsigned char* const* graySource = NULL;
450  // The GWorld only works with unshifted rasters so we need to compute
451  // the "true" copy rectangles first
452  ImageRect* sourceRect = new ImageRect(copyRect->getLeft() - getLeft(),
453  copyRect->getTop() - getTop(),
454  copyRect->getWidth(), copyRect->getHeight());
455  ImageRect* destRect = origin == NULL ?
456  new ImageRect(copyRect->getLeft() - imgOut->getLeft(),
457  copyRect->getTop() - imgOut->getTop(),
458  copyRect->getWidth(), copyRect->getHeight())
459  :
460  new ImageRect(origin->getX() - imgOut->getLeft(),
461  origin->getY() - imgOut->getTop(),
462  copyRect->getWidth(), copyRect->getHeight());
463 
464  const int //iLowSource = sourceRect->getTop(),
465  iLowDest = destRect->getTop(),
466  //iHighSource = sourceRect->getBottom(),
467  //jLowSource = sourceRect->getLeft(),
468  jLowDest = destRect->getLeft();
469  const int nbRows = sourceRect->getHeight(),
470  nbCols = sourceRect->getWidth();
471 
472  // Subclasses should override this function, but in case they don't, we need to
473  // do the conversion through the base gray-level raster.
474  if (!gworldIsReference_ && referenceHasChanged_)
475  updateRaster_( );
476 
477  switch( imgOut->getBaseRasterType() )
478  {
479  case kBinaryRaster:
480 
481  // At this point we are ready to perform the copy/conversion and we use
482  // the CopyBits function of QuickDraw/QuickTime
483  gworld_->copyInto(imgOut->gworld_, sourceRect, destRect, false);
484 
485  imgOut->setGraphWorldAsReference_();
486  imgOut->setReferenceHasChanged_(true);
487  break;
488 
489  case kGrayRaster:
490  // if we are dealing with a plain gray raster image, just copy the bytes
491  if (!imgOut->hasFloatRaster()) {
492 
493  // At this point we are ready to perform the copy/conversion and we use
494  // the CopyBits function of QuickDraw/QuickTime
495  gworld_->copyInto(imgOut->gworld_, sourceRect, destRect, false);
496 
497  imgOut->setGraphWorldAsReference_();
498  imgOut->setReferenceHasChanged_(true);
499  }
500  //
501  // if imgOut has a float, use this image's gray raster to do the copy.
502  else
503  {
504  // temporary
505  const int iLow = copyRect->getTop(),
506  iHigh = copyRect->getBottom(),
507  jLow = copyRect->getLeft(),
508  jHigh = copyRect->getRight();
509  //
510  graySource = getShiftedRaster2D();
511  float* const* fGrayDest = (static_cast<RasterImage_gray_F*>(imgOut))->getShiftedGrayF2D(R_W_ACCESS);
512  // This call may have been made from within the RasterImage_gray_F constructor of imgOut,
513  // in which case grayDest has not been allocated yet
514  if (fGrayDest == NULL)
515  {
516  imgOut->allocateRaster2D_(nbRows, nbCols);
517  imgOut->allocateShiftedRaster2D_(nbRows, nbCols, iLowDest, jLowDest);
518  fGrayDest = (static_cast<RasterImage_gray_F*>(imgOut))->getShiftedGrayF2D(R_W_ACCESS);
519  }
520 
521  for (int i=iLow; i<=iHigh; i++)
522  for (int j=jLow; j<=jHigh; j++)
523  fGrayDest[i][j] = graySource[i][j];
524 
525  imgOut->setFloatRasterAsReference_();
526  imgOut->setReferenceHasChanged_(true);
527  (static_cast<RasterImage_gray_F*>(imgOut))->setFloatConversionMode(kSaturatedPositive);
528  }
529  break;
530 
531  case kRGB24Raster:
532  // At this point we are ready to perform the copy/conversion and we use
533  // the CopyBits function of QuickDraw/QuickTime
534  gworld_->copyInto(imgOut->gworld_, sourceRect, destRect, true);
535 
536  imgOut->setGraphWorldAsReference_();
537  imgOut->setReferenceHasChanged_(true);
538  break;
539 
540  case kaRGB16Raster:
541 
542  // At this point we are ready to perform the copy/conversion and we use
543  // the CopyBits function of QuickDraw/QuickTime
544  gworld_->copyInto(imgOut->gworld_, sourceRect, destRect, false);
545 
546  imgOut->setGraphWorldAsReference_();
547  imgOut->setReferenceHasChanged_(true);
548  break;
549 
550  case kRGBa32Raster:
551  // if we are dealing with a plain gray raster image, just copy the bytes
552  if (!imgOut->hasFloatRaster())
553  {
554  // At this point we are ready to perform the copy/conversion and we use
555  // the CopyBits function of QuickDraw/QuickTime
556  gworld_->copyInto(imgOut->gworld_, sourceRect, destRect, true);
557 
558  imgOut->setGraphWorldAsReference_();
559  imgOut->setReferenceHasChanged_(true);
560  }
561  //
562  // if imgOut has a float raster, we had better use it
563  else
564  {
565  // temporary
566  const int iLow = copyRect->getTop(),
567  iHigh = copyRect->getBottom(),
568  jLow = copyRect->getLeft(),
569  jHigh = copyRect->getRight();
570  //
571  graySource = getShiftedRaster2D();
572  float* const* fRedDest = (static_cast<RasterImage_RGBa_F*>(imgOut))->getShiftedRedF2D(R_W_ACCESS);
573  float* const* fGreenDest = (static_cast<RasterImage_RGBa_F*>(imgOut))->getShiftedGreenF2D(R_W_ACCESS);
574  float* const* fBlueDest = (static_cast<RasterImage_RGBa_F*>(imgOut))->getShiftedBlueF2D(R_W_ACCESS);
575  for (int i=iLow; i<=iHigh; i++)
576  for (int j=jLow; j<=jHigh; j++)
577  fRedDest[i][j] = fGreenDest[i][j] = fBlueDest[i][j] = graySource[i][j];
578 
579  imgOut->setFloatRasterAsReference_();
580  imgOut->setReferenceHasChanged_(true);
581  (static_cast<RasterImage_RGBa_F*>(imgOut))->setFloatConversionMode(kSaturatedPositive);
582  }
583  break;
584 
585  default:
586  FAIL_CONDITION( true,
587  kRasterImageError,
588  "Invalid image type in RasterImage_gray::writeInto_");
589  break;
590 
591  }
592 
593  if (sourceRect != NULL)
594  delete sourceRect;
595  if (destRect != NULL)
596  delete destRect;
597 
598 }

610 void RasterImage_binary::convertToGray(const unsigned char* theSource,
611  unsigned char* theDestination,
612  long nbPixels)
613 {
614  const long nbBytes = (nbPixels+7)/8;
615  const int remainder = static_cast<int>(nbPixels % 8);
616  const unsigned char* source = theSource;
617  unsigned char* dest = theDestination;
618  const unsigned char BIN_0 = static_cast<unsigned char>(0),
619  BIN_255 = static_cast<unsigned char>(255);
620 
621  // do the conversion for the full bytes
622  for (long l=0; l<nbBytes-1; l++)
623  {
624  unsigned char c = *(source++);
625 
626  for (int k=7; k>=0; k--)
627  *(dest++) = (c & (1 << k)) ? BIN_255 : BIN_0;
628 
629  }
630 
631  // do the conversion for the remainder bits (incomplete byte)
632  unsigned char c = *(source++);
633  for (int k=7; k>7-remainder; k--)
634  *(dest++) = (c & (1 << k)) ? BIN_255 : BIN_0;
635 
636 }
