182 void NonlinearOptimizerND_conjDir::minimize_(const FunctionND* f,const double* startX,
183  const double* xMin, const double* xMax,
184  double tol, OptimizerRecordND* optRec)
185 {
186  // If the function had not been pre-set we need to perform some allocations
187  if (!isOptimizationFunction(f))
188  {
189  setFunction(f);
190  }
191 
192  // if USER_SPECIFIED_DIRECTIONS was selected as direction update mode but the user has
193  // not set update directions, the mode is changed to CANONICAL_DIRECTIONS
194  if ((resetMethod_ == USER_SPECIFIED_DIRECTIONS) && (userDirections_ == NULL))
195  resetMethod_ = CANONICAL_DIRECTIONS;
196 
197  updateConjugateDirections_();
198 
199 
200  const int nDim = getDimension();
201 
202  // set the starting position and the conjugate directions
203  for (int i=0; i<nDim; i++)
204  {
205  Pi_[i] = P0_[i] = startX[i];
206 
207  // the conjugate directions are initially the cardinal directions
208  for (int j=0; j<nDim; j++)
209  u_[i][j] = 0.L;
210  u_[i][i] = 1.0L;
211  }
212 
213 //jyh 2013-08-30
214 // // inform the section function of the bounds for this search
215 // fdSec1D_->setBounds(xMin, xMax);
216 
217  // the best value found so far is the one at our starting point
218  double optVal = f->func(startX);
219 
220  cycleCounter_ = 0;
221  optRec->nbOuterIters = 0;
222  optRec->nbInnerIters = 0;
223  optRec->isSuccess = true; // the power of positive thinking
224 
225  bool ongoingSearch = true;;
226  // repeat the cycle until a minimum is found or one of the
227  // line searches fail
228  while (ongoingSearch)
229  {
230  // One cycle over the conjugate directions
231  cycleCounter_++;
232 
233  // best direction so far
234  bestDirIndex_ = -1;
235  double bestReduction = -1;
236  double inCycleOptVal = optVal;
237 
238  for (int i=0; (i<nDim) && ongoingSearch; i++)
239  {
240  // perform line minimization along direction u[i]
241 //jyh 2013-08-30
242 // fdSec1D_->setSection(Pi_, u_[i]);
243  double sMin = fdSec1D_->getLowerBound(),
244  sMax = fdSec1D_->getUpperBound();
245  lineMinimizer_->minimize(fdSec1D_, 0, sMin, sMax, tol, lineRecord_);
246 
247  // if the line minimization succeeded, we need to update the state to prepare
248  // for the next direction or cycle
249  if (lineRecord_->isSuccess)
250  {
251  float valReduction = inCycleOptVal - lineRecord_->optVal;
252  if (valReduction > bestReduction)
253  {
254  bestReduction = valReduction;
255  bestDirIndex_ = i;
256  }
257  inCycleOptVal = lineRecord_->optVal;
258  optRec->nbInnerIters += lineRecord_->nbIters;
259 
260  /// move to the new point before changing direction
261  double s = lineRecord_->optX;
262  for (int j=0; j<nDim; j++)
263  Pi_[j] += s * u_[i][j];
264 
265  }
266  // if the line minimization failed, there is no need to search any further
267  else
268  {
269  optRec->isSuccess = false;
270  ongoingSearch = false;
271  }
272 
273  optRec->nbOuterIters++;
274  }
275 
276  // are we done yet? What fraction of its value did the function vary over a full
277  // cycle?
278  if (tol*(fabs(inCycleOptVal) + fabs(optVal)) > 2.0*(optVal - inCycleOptVal))
279  ongoingSearch = false;
280  else
281  updateConjugateDirections_();
282 
283  optVal = inCycleOptVal;
284  }
285 
286  // we store the results of our iteration in the optimizer record.
287  // The numbers of iterations are already up-to-date
288  for (int j=0; j<nDim; j++)
289  optRec->optX[j] = Pi_[j];
290  optRec->optVal = optVal;
291 }

330 void NonlinearOptimizerND_conjDir::updateConjugateDirections_(void)
331 {
332  const int nDim = getDimension();
333 
334  // First, if we have regular reset, check if it's time to do one
335  if ((updateMethod_ != NO_UPDATE) && ((cycleCounter_ % (reset_ * nDim)) == 0))
336  {
337  switch (resetMethod_)
338  {
339  case USER_SPECIFIED_DIRECTIONS:
340  FAIL_CONDITION( true,
341  kCaseNotImplemented,
342  "case not implemented yet in NonlinearOptimizerND_conjDir::updateConjugateDirections_");
343  break;
344 
345  RANDOM_ORTHOGONAL_DIRECTIONS:
346  FAIL_CONDITION( true,
347  kCaseNotImplemented,
348  "case not implemented yet in NonlinearOptimizerND_conjDir::updateConjugateDirections_");
349  break;
350 
351  CANONICAL_DIRECTIONS:
352  default:
353  for (int i=0; i<nDim; i++)
354  {
355  for (int j=0; j<nDim; j++)
356  u_[i][j] = 0.L;
357  u_[i][i] = 1.0L;
358  }
359  break;
360  }
361  }
362  // else, go for regular update
363  else
364  {
365  switch (updateMethod_)
366  {
367  case NO_UPDATE:
368  //
369  // do nothing
370  break;
371 
372  case POWELL_UN_IS_XN_X0:
373  // simply permutate the pointers
374  for (int i=0; i<nDim-1; i++)
375  u_[i] = u_[i+1];
376 
377  // new direction is given by P0-PN
378  for (int j=0; j<nDim-1; j++)
379  u_[nDim-1][j] = Pi_[j] - P0_[j];
380  break;
381 
382 
383  case POWELL_DISCARD_LARGEST_DECREASE_DIR:
384  FAIL_CONDITION( true,
385  kCaseNotImplemented,
386  "case not implemented yet in NonlinearOptimizerND_conjDir::updateConjugateDirections_");
387  break;
388 
389  default:
390  FAIL_CONDITION( true,
391  kCaseNotImplemented,
392  "case not implemented yet in NonlinearOptimizerND_conjDir::updateConjugateDirections_");
393  }
394  }
395 }
