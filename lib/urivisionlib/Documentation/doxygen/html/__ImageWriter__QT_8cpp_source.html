276 OSType getWriteExtensionType_(const char* ext)
277 {
278  char lcExt[5];
279  int k;
280  for (k=0; ext[k] != 0; k++)
281  {
282  lcExt[k] = tolower(ext[k]);
283  }
284  lcExt[k] = 0;
285 
286  if (strncmp(lcExt, "jpg", 3) == 0)
287  return kQTFileTypeJPEG;
288  else if (strncmp(lcExt, "jpeg", 4) == 0)
289  return kQTFileTypeJPEG;
290  else if (strncmp(lcExt, "bmp", 3) == 0)
291  return kQTFileTypeBMP;
292  else if (strncmp(lcExt, "tga", 3) == 0)
293  return kQTFileTypeTargaImage;
294  else if (strncmp(lcExt, "tif", 3) == 0)
295  return kQTFileTypeTIFF;
296  else if (strncmp(lcExt, "tiff", 4) == 0)
297  return kQTFileTypeTIFF;
298  else if (strncmp(lcExt, "png", 3) == 0)
299  return kQTFileTypePNG;
300  else if (strncmp(lcExt, "gif", 3) == 0)
301  return kQTFileTypeGIF;
302  else if (strncmp(lcExt, "psd", 3) == 0)
303  return kQTFileTypePhotoShop;
304  else if (strncmp(lcExt, "sgi", 3) == 0)
305  return kQTFileTypeSGIImage;
306  else if (strncmp(lcExt, "rgb", 3) == 0)
307  return kQTFileTypeSGIImage;
308  else if (strncmp(lcExt, "rgba", 4) == 0)
309  return kQTFileTypeSGIImage;
310 
311  return (-1);
312 }
