290 void ElevationMap::updateMesh(void)
291 {
292  // If the elevation raster is the reference we recalculate the mesh
293  DrawableInformationType dType = getDrawableInformationType();
294  if ((dType == kDense) || (dType == kDenseUncertain))
295  {
296  const ImageRect* vRect = getValidRect();
297  const int iLow = vRect->getTop(),
298  jLow = vRect->getLeft(),
299  nbRows = getHeight(),
300  nbCols = getWidth();
301 
302  const float* elevation = getGrayF();
303 
304  // Then we compute the XYZ coordinates and store them in the triMesh array
305  long ptIndex = 0L;
306  for (int i=0, iSh = iLow; i<nbRows; i++, iSh++)
307  for (int j=0, jSh=jLow; j<nbCols; j++, jSh++)
308  {
309  triMesh_[i][j][0] = scaleX_ * jSh;
310  triMesh_[i][j][1] = scaleY_ * iSh;
311  triMesh_[i][j][2] = scaleZ_ * elevation[ptIndex++];
312  }
313  }
314  // else I don't know yet what to do
315  else
316  {
317  }
318 }
