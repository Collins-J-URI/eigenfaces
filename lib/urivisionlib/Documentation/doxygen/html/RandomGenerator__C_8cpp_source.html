68 unsigned long RandomGenerator_C::nextULong(void)
69 {
70  unsigned long rl = rand();
71  for (int i=1; i<FULL_IN_ULONG; i++)
72  rl = (rl << MAX_BITS_IN_RAND) | rand();

74  if (RES_TO_ULONG)

77  (MAX_BITS_IN_RAND - RES_TO_ULONG);
