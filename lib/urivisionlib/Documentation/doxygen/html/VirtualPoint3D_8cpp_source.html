207 void VirtualPoint3D::addFaceInfo(int faceIndex, int uvIndex, int normalIndex)
208 {
209  // if the face index vector is too small, resize it to be able to add an element
210  // at the new faceIndex
211  if (faceIndex >= static_cast<int>(normalIndex_.size())) {
212  for (int i=normalIndex_.size(); i<=faceIndex; i++)
213  {
214  normalIndex_.push_back(-1);
215  uvIndex_.push_back(-1);
216  }
217  }
218 
219  vector<int>::iterator normal = normalIndex_.begin();
220  vector<int>::iterator uv = uvIndex_.begin();
221  normal[faceIndex] = normalIndex;
222  uv[faceIndex] = uvIndex;
223 }

225 void VirtualPoint3D::setSelected(bool selected)
226 {
227  if (selected == true){
228 
229  // if a point was selected before, deselect it
230  if (currentSelected_ != NULL)
231  currentSelected_->setSelected(false);
232 
233  // this point is selected
234  isSelected_ = true;

236 
237  // all its neighbors are selected neighbors
238  vector <VirtualPoint3D*>:: iterator nb, endNeighbor = neighbors_.end();
239  for (nb = neighbors_.begin(); nb != endNeighbor; nb++)
240  (*nb)->setSelectedNeighbor(true);
241 
242  }
243  else{
244  // deselect this point
245  isSelected_ = false;
246 
247  // deselect all neighbors as selected neighbors
248  // all its neighbors are selected neighbors
249  vector <VirtualPoint3D*>:: iterator nb, endNeighbor = neighbors_.end();
250  for (nb = neighbors_.begin(); nb != endNeighbor; nb++)
251  (*nb)->setSelectedNeighbor(false);

298 void VirtualPoint3D:: render_(int camIndex) const
299 {
300  // drawing a small square
301  glPushMatrix();
302 
303  // Move from the reference frame to this point
304  glTranslatef(static_cast<float>(point_->getX()),
305  static_cast<float>(point_->getY()),
306  static_cast<float>(point_->getZ()));
307 
308 
309  if (renderMode_ == INTERPOSE)
310  if (color_ != NULL)
311  glColor4fv(color_);
312  else
313  glColor4fv(defaultColor_);
314  else if (isSelected_)
315  glColor4f(1.f, 0.f, 0.f, 1.f);
316  else if (isSelectedNeighbor_)
317  glColor4f(1.f, 1.f, 0.f, 1.f);
318  else
319  glColor4fv(defaultColor_);
320 
321  if (hasIndividualRenderScale_)
322  glScalef(renderScale_, renderScale_, renderScale_);
323  else
324  glScalef(defaultRenderScale_, defaultRenderScale_, defaultRenderScale_);
325 
326  glBegin(GL_QUAD_STRIP);
327  glVertex3f(-1.f, -1.f, 1.f);
328  glVertex3f(-1.f, -1.f, -1.f);
329  glVertex3f(1.f, -1.f, 1.f);
330  glVertex3f(1.f, -1.f, -1.f);
331  glVertex3f(1.f, 1.f, 1.f);
332  glVertex3f(1.f, 1.f, -1.f);
333  glVertex3f(-1.f, 1.f, 1.f);
334  glVertex3f(-1.f, 1.f, -1.f);

337  glEnd();
