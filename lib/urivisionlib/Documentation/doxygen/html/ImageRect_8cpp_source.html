458 ImageRect* ImageRect::intersection(const ImageRect* rect1, const ImageRect* rect2)
459 {
460  FAIL_CONDITION( rect1 == NULL,
461  kNullRectangleError,
462  "NULL ImageRect parameter in ImageRect::intersection");
463  FAIL_CONDITION( rect2 == NULL,
464  kNullRectangleError,
465  "NULL ImageRect parameter in ImageRect::intersection");
466 
467  ImageRect* inter = NULL;
468 
469  // if the rectangles don't overlap, we return NULL
470  if (rect1->overlaps(rect2))
471  {
472  const int xMin1 = rect1->getLeft(),
473  xMax1 = rect1->getRight(),
474  yMin1 = rect1->getTop(),
475  yMax1 = rect1->getBottom();
476  const int xMin2 = rect2->getLeft(),
477  xMax2 = rect2->getRight(),
478  yMin2 = rect2->getTop(),
479  yMax2 = rect2->getBottom();
480  const int xMin = xMin1 > xMin2 ? xMin1 : xMin2;
481  const int yMin = yMin1 > yMin2 ? yMin1 : yMin2;
482  const int w = xMax1 > xMax2 ? xMax2 - xMin + 1: xMax1 - xMin + 1;
483  const int h = yMax1 > yMax2 ? yMax2 - yMin + 1: yMax1 - yMin +1;
484 
485  inter = new ImageRect(xMin, yMin, w, h);
486  }
487 
488  return inter;
489 }
