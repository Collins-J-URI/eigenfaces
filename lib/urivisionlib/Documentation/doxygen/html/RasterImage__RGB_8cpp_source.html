420 void RasterImage_RGB::writeInto_(RasterImage* imgOut, const ImageRect* copyRect,
421  const ImagePoint* origin) const
422 {
423  // The GWorld only works with unshifted rasters so we need to compute
424  // the "true" copy rectangles first
425  ImageRect* sourceRect = new ImageRect(copyRect->getLeft() - getLeft(),
426  copyRect->getTop() - getTop(),
427  copyRect->getWidth(), copyRect->getHeight());
428  ImageRect* destRect = origin == NULL ?
429  new ImageRect(copyRect->getLeft() - imgOut->getLeft(),
430  copyRect->getTop() - imgOut->getTop(),
431  copyRect->getWidth(), copyRect->getHeight())
432  :
433  new ImageRect(origin->getX() - imgOut->getLeft(),
434  origin->getY() - imgOut->getTop(),
435  copyRect->getWidth(), copyRect->getHeight());
436 
437  // Subclasses should override this function, but in case they don't, we need to
438  // do the conversion through the base rgb raster.
439  if (!gworldIsReference_ && referenceHasChanged_)
440  updateRaster_( );
441 
442  switch( imgOut->getBaseRasterType() )
443  {
444  case kBinaryRaster:
445  // The GWorld only works with unshifted rasters so we need to compute
446  // the "true" copy rectangles first
447  // At this point we are ready to perform the copy/conversion and we use
448  // the CopyBits function of QuickDraw/QuickTime
449  gworld_->copyInto(imgOut->gworld_, sourceRect, destRect, true);
450 
451  imgOut->gworldIsReference_ = true;
452  imgOut->referenceHasChanged_ = true;
453  break;
454 
455  case kGrayRaster:
456  // are we dealing with a gray or gray_F image?
457  //
458  // The output image is a "regular" gray image [unsigned char]
459  if (!imgOut->hasFloatRaster())
460  {
461  // At this point we are ready to perform the copy/conversion and we use
462  // the CopyBits function of QuickDraw/QuickTime
463  gworld_->copyInto(imgOut->gworld_, sourceRect, destRect, true);
464 
465  imgOut->gworldIsReference_ = true;
466  imgOut->referenceHasChanged_ = true;
467  }
468  // if the image has a floating point raster, we write into it.
469  // why???
470  else
471  {
472  const int iLowSource = copyRect->getTop(),
473  iHighSource = copyRect->getBottom(),
474  jLowSource = copyRect->getLeft();
475  int iLowDest, jLowDest;
476  //
477  if (origin==NULL)
478  {
479  iLowDest = iLowSource;
480  jLowDest = jLowSource;
481  }
482  else
483  {
484  iLowDest = origin->getY();
485  jLowDest = origin->getX();
486  }
487  const int nbCols = getWidth();
488 
489  const unsigned char* const* rgbSource = getShiftedRGB2D();
490  float* const* fltGrayDest= (static_cast<RasterImage_gray_F*>(imgOut))->getShiftedGrayF2D(R_W_ACCESS);
491  for (int i=iLowSource, iDest=iLowDest; i<=iHighSource; i++, iDest++)
492  convertToGray(rgbSource[i]+jLowSource, fltGrayDest[iDest]+jLowDest,
493  nbCols, defaultGrayConv_);
494 
495  (static_cast<RasterImage_gray_F*>(imgOut))->setFloatConversionMode(kSaturatedPositive);
496  imgOut->gworldIsReference_ = false;
497  imgOut->referenceHasChanged_ = true;
498  }
499  break;
500 
501  case kRGB24Raster:
502  // At this point we are ready to perform the copy/conversion and we use
503  // the CopyBits function of QuickDraw/QuickTime
504  gworld_->copyInto(imgOut->gworld_, sourceRect, destRect, false);
505 
506  imgOut->gworldIsReference_ = true;
507  imgOut->referenceHasChanged_ = true;
508  break;
509 
510  case kRGBa32Raster:
511  // are we dealing with a RGB or RGB_F image?
512  //
513  // The output image is a "regular" RGB image [unsigned char]
514  if (!imgOut->hasFloatRaster())
515  {
516  // At this point we are ready to perform the copy/conversion and we use
517  // the CopyBits function of QuickDraw/QuickTime
518  gworld_->copyInto(imgOut->gworld_, sourceRect, destRect, false);
519 
520  imgOut->gworldIsReference_ = true;
521  imgOut->referenceHasChanged_ = true;
522  }
523  // imgOut has float type
524  else
525  {
526  const int iLowSource = copyRect->getTop(),
527  iHighSource = copyRect->getBottom(),
528  jLowSource = copyRect->getLeft(),
529  jHighSource = copyRect->getRight();
530  int iLowDest, jLowDest;
531  //
532  if (origin==NULL)
533  {
534  iLowDest = iLowSource;
535  jLowDest = jLowSource;
536  }
537  else
538  {
539  iLowDest = origin->getY();
540  jLowDest = origin->getX();
541  }
542 
543 
544  const unsigned char* const* rgbSource = getShiftedRGB2D();
545  float* const* fltRedDest= (static_cast<RasterImage_RGBa_F*>(imgOut))->getShiftedRedF2D(R_W_ACCESS);
546  float* const* fltGreenDest= (static_cast<RasterImage_RGBa_F*>(imgOut))->getShiftedGreenF2D(R_W_ACCESS);
547  float* const* fltBlueDest= (static_cast<RasterImage_RGBa_F*>(imgOut))->getShiftedBlueF2D(R_W_ACCESS);

549  for(int j=jLowSource, k=3*jLowSource, jDest=jLowDest; j<=jHighSource; j++, jDest++)
550  {
551  fltRedDest[iDest][jDest] = rgbSource[i][k++];

561

565  "Invalid image type in RasterImage_gray::writeInto_");
566  break;
567 
568  }
569

586 void RasterImage_RGB::convertToGray(const unsigned char* theSource,
587  unsigned char* theDestination,
588  long nbPixels,
589  ConversionToGrayMode myGrayConv)
590 {
591  const unsigned char *source = theSource;
592  unsigned char *dest = theDestination;
593  int sum;
594  float sumF;
595 
596  // how do we convert from rgb to gray?
597  switch (myGrayConv)
598  {
599 
600  case kMaxFieldConversion:
601 
602  for (long j=0; j<nbPixels; j++)
603  {
604  // optimization "in the small" left for a more clever programmer
605  unsigned char red = *(source++),
606  green = *(source++),
607  blue = *(source++);
608  *(dest++) = MAX(red, MAX(green, blue));
609  }
610  break;
611 
612  case kAverageConversion:
613 
614  for (long j=0; j<nbPixels; j++)
615  {
616  sum = *(source++);
617  sum += *(source++);
618  sum += *(source++);
619  *(dest++) = static_cast<unsigned char>(sum/3);
620  }
621  break;
622 
623  case kWeightedAverageConversion:
624 
625  for (long j=0; j<nbPixels; j++)
626  {
627  sumF = defaultConvWeight_[0] ** (source++);
628  sumF += defaultConvWeight_[1] ** (source++);
629  sumF += defaultConvWeight_[2] ** (source++);
630  *(dest++) = static_cast<unsigned char>(sumF);
631  }
632  break;
633 
634  default:
635  FAIL_CONDITION( true,
636  kRasterImageError,
637  "unknown conversion to gray mode encountered");
638  break;
639 
640  }
641 }

798 {

800  kAccessToLockedData,
801  "Attempt to resize a RasterImage with locked properties");

803  // if width and height have not changed, then the operation is just a translation.
804  // All we need to do is update the ImageRect member and update the "shifted" array
805  /// (no re-allocation needed)
806  if ((theWidth == getWidth()) && (theHeight == getHeight()))
807  {
808  if ((x != getLeft()) || (y != getTop()))
809  {
810  // Since the rasters have the same dimensions, there is no need
811  // to delete and reallocate the shifted rasters. Just apply a
812  // relative shift to them
813  const int iLow = getTop(),
814  iHigh = getBottom(),
815  jLow = getLeft(),
816  rowShift = y - iLow,
817  colShift = 3*(x - jLow);
818 
819 if(shiftedRgb2D_)

821  for (int i=iLow; i<=iHigh; i++)
822  shiftedRgb2D_[i] += colShift;
823  shiftedRgb2D_ += rowShift;
824 }
825 
826  }
827  // if nothing has changed, there is nothing to do
828  }
829  // if the dimensions of the rectangles are different, then we must delete
830  // and reallocate the GraphWorld as well as the 0-based 2D raster.
831 
832  else
