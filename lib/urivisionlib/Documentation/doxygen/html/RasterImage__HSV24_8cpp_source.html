118 void RasterImage_HSV24::convertFromRGBa_()
119 {
120  unsigned char maxVal, minVal,
121  pRed, pGreen, pBlue,
122  **rgba = getRGBa2D();
123 
124  int nbRows = getHeight(),
125  nbCols = getWidth();
126 
127  if (gworldIsReference_ && referenceHasChanged_)
128  {
129  for (int i=0; i<nbRows; i++)
130  {
131  for (int j=0, rgbaIndex=0; j<nbCols; j++, rgbaIndex++)
132  {
133  pRed = rgba[i][rgbaIndex++];
134  pGreen = rgba[i][rgbaIndex++];
135  pBlue = rgba[i][rgbaIndex++];
136  maxVal = MAX(MAX(pRed, pGreen), pBlue);
137  minVal = MIN(MIN(pRed, pGreen), pBlue);
138 
139  /* HueSaturationV */
140  val2D_[i][j] = (unsigned char) ((100 * maxVal) / 255);
141 
142  if (maxVal != minVal)
143  {
144  sat2D_[i][j] = ((maxVal-minVal)*100)/maxVal;
145 
146  if ( pRed == maxVal )
147  hue2D_[i][j] = (pGreen-pBlue)*42/(maxVal-minVal);
148  else if ( pGreen == maxVal )
149  hue2D_[i][j] = 83 + (pBlue-pRed)*43/(maxVal-minVal);
150  else
151  hue2D_[i][j] = 128 + (pRed-pGreen)*42/(maxVal-minVal);
152 
153  // we want the hue_ in the range [0, 255).
154  if ( hue2D_[i][j] < 0)
155  hue2D_[i][j] += 255;
156  }
157  else
158  {
159  sat2D_[i][j] = 0;
160  hue2D_[i][j] = 0;
161  }
162  }
163  }
164  }
165  referenceHasChanged_ = false;
